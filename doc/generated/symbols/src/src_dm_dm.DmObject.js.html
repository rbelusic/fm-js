 10 /**
 11 * Basic DM class. Provide groups. 
 12 * 
 13 * @class FM.DmObject
 14 * @extends FM.Object
 15 * @memberOf FM
 16 * @param {object} attrs list of attribute name and values
 17 * @param {object} [flds] allowed attributes
 18 */    
 19 FM.DmObject = FM.defineClass('DmObject',FM.Object);

 22 // methods
 23 FM.DmObject.prototype._init = function(attrs,flds) {            
 24     this.objectGroups = {};
 25     this.defaultGroup = '';               
 26 
 27     this._super("_init",attrs,flds);
 28     this.objectSubClass = "Object";
 29 }

 43 /**
 44  * Add object to group. 
 45  * 
 46  * @public     
 47  * @function 
 48  * @param {string} gname Group name.
 49  * @param {boolean} [callevent=false] Fire <i>onChange</i> event.
 50  * @returns {boolean} Returns <i>true</i> if object is successfully added to gruop. 
 51  */
 52 FM.DmObject.prototype.addGroup = function(gname,callevent) {
 53     if(!FM.isset(gname) || !gname || gname == '') return false;
 54 
 55     this.objectGroups[gname] = {
 56         name: gname
 57     }
 58 
 59     if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
 60     return true;
 61 }
 62 
 63 /**
 64  * Remove object from group. 
 65  * 
 66  * @public     
 67  * @function 
 68  * @param {string} gname Group name.
 69  * @param {boolean} [callevent=false] Fire <i>onChange</i> event.
 70  * @returns {boolean} Returns <i>true</i> if object is successfully removed. 
 71  */
 72 FM.DmObject.prototype.removeGroup = function(gname,callevent) {
 73     if(!FM.isset(gname) || !gname || !FM.isset(this.objectGroups[gname])) return false;
 74     var newglist = {};            
 75     FM.forEachAttr(this.objectGroups,function(name,value) {
 76         if(name != gname) {
 77             newglist[name] = value;
 78         }
 79         return true;
 80     });
 81 
 82     this.objectGroups = newglist;
 83     if(this.defaultGroup == gname) this.defaultGroup = '';
 84     if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
 85     return true;
 86 }

 99         true : false
100         );
101 }

154 /** 
155  * Set default object group.
156  * 
157  * @public
158  * @function 
159  * @param {string} gname Group name.
160  * @returns {boolean}
161  */
162 FM.DmObject.prototype.setDefaultGroup = function(s,callevent) {
163     s = FM.isset(s) && s ? s : '';
164     if(s == '' || FM.isset(this.objectGroups[s])) {
165         this.defaultGroup = s;
166         if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
167         return true;
168     }            
169     return false;
170 }

172 /**
173  * Returns default group.
174  * 
175  * @public     
176  * @function 
177  * @returns {Object} 
178  */
179 FM.DmObject.prototype.getDefaultGroup = function() {
180     if(this.defaultGroup != '') {
181         return FM.getAttr(this.objectGroups,this.defaultGroup,null);
182     }
183 
184     // ako nema def ili nije vidljiv
185     var defgrp = null;
186     FM.forEach(this.objectGroups,function(name,value) {
187         // prvi u listi
188         defgrp = value;
189         return false;
190     }); 
191     return defgrp;
192 }

201 /**
202 * Returns DmObject class function for <b>sctype</b> subclass type.
203 * @static
204 * @function    
205 * @param {object} app Application.
206 * @param {String} sctype Subclass type.
207 * @return {function} Class function.
208 */   
209 FM.DmObject.getConfiguration = function(app,sctype) {
210     var list = FM.DmObject.subClassTypes;
211 
212     app = FM.isset(app) && app ? app : null;
213     var appCls = app ? app.getSubClassName() : null;
214     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
215         
216     var obj = null;
217     if(appCfg && FM.isset(appCfg[sctype])) {
218         obj = appCfg[sctype];
219     } else if(app && FM.isArray(app.applicationObjectsSpace)) {
220         FM.forEach(app.applicationObjectsSpace,function(i,ns) {
221             if(FM.isset(list[ns]) && FM.isset(list[ns][sctype])) {
222                 obj = list[ns][sctype];
223                 return false;
224             }
225             return true;
226         });
227     }
228     
229     if(!obj && FM.isset(list['GLOBAL'][sctype])) {
230         obj = list['GLOBAL'][sctype];
231     }
232     
233     return obj;
234 }

236 /**
237 * Returns new instance of chosen <b>sctype</b> subclass type.
238 * @static
239 * @public
240 * @function    
241 * @param {object} app Application.
242 * @param {String} sctype Subclass type.
243 * @return {object} Class function.
244 */   
245 FM.DmObject.newObject = function(app,sctype, oAttrs) {    
246     var clsFn = FM.DmObject.getConfiguration(app,sctype);
247     return clsFn ? new clsFn(oAttrs) : null;
248 }

250 /**
251 * Returns application registered subclass type function.
252 * 
253 * @static
254 * @public
255 * @function    
256 * @param {string} [sctype="Object"] Subclass type.
257 * @param {string} [scapp="GLOBAL"] Application subclass type.
258 * @return {function} Class function or <i>FM.DmObject</i> if <i>sctype</i> 
259 *  is ommited, empty string or null.
260 */   
261 FM.DmObject.getSubClassType = function(sctype,scapp) {
262     if(!FM.isset(sctype) || sctype == '') {
263         return FM.DmObject;
264     } 
265     scapp = FM.isset(scapp) && scapp != '' ? scapp : 'GLOBAL';
266     if(!FM.isset(FM.DmObject.subClassTypes[scapp])) {
267         return null;
268     }
269     
270     var appcfg = FM.DmObject.subClassTypes[scapp];
271     
272     return FM.isset(appcfg[sctype]) ? appcfg[sctype] : null;
273 }

275 /**
276 * Register application subclass type.
277 * 
278 * @static
279 * @public
280 * @function    
281 * @param {string} sctype Subclass type.
282 * @param {function} clsFn DM Class to register.
283 * @param {string} [scapp="GLOBAL"] Application subclass.
284 */   
285 FM.DmObject.addSubClassType = function(sctype, clsFn,scapp) {
286     scapp = FM.isset(scapp) && FM.isString(scapp) && scapp != '' ? scapp : 'GLOBAL';
287     if(!FM.isset(FM.DmObject.subClassTypes[scapp])) {
288         FM.DmObject.subClassTypes[scapp]= {};
289     }
290     
291     FM.DmObject.subClassTypes[scapp][sctype] = clsFn;
292 }
