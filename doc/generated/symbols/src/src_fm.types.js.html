 65 /**
 66  * Determine if a variable is array.
 67  * 
 68  * @static
 69  * @function 
 70  * @param v The variable to be checked.
 71  * @returns {boolean} 
 72  */
 73 FM.isArray = function(v) {
 74     if (!FM.isset(v) || !v) {
 75         return false;
 76     }
 77     
 78     if (v.constructor.toString().indexOf("Array") == -1) {
 79         return false;
 80     }
 81 
 82     return FM.isset(v.length);
 83 }

100  * 
101  * @static

103  * @param {Object|Array} o The array or object to be checked.
104  * @returns {number} Number of elements in array or object.

106 FM.sizeOf = function(o) {
107     if (!FM.isset(o) || o == null)
108         return(-1);
109     var i = 0;
110     for (var id in o) {

214 /**
215  *  Returns a string with backslashes before characters that need to be quoted. 
216  *  These characters are single quote ('), double quote (") and backslash.
217  * 
218  * @static
219  * @function 
220  * @param {string} str The string to be escaped.
221  * @returns {string} 
222  */
223 FM.addslashes = function(str) {
224     str = str.replace(/\\/g, '\\\\');
225     str = str.replace(/\'/g, '\\\'');
226     str = str.replace(/\"/g, '\\"');
227     str = str.replace(/\0/g, '\\0');
228     return str;
229 }
230 
231 /**
232  *  Un-quotes a quoted string
233  * 
234  * @static
235  * @function 
236  * @param {string} s The input string.
237  * @returns {string} 
238  */
239 FM.stripslashes = function(str) {
240     str = str.replace(/\\'/g, '\'');
241     str = str.replace(/\\"/g, '"');
242     str = str.replace(/\\0/g, '\0');
243     str = str.replace(/\\\\/g, '\\');
244     return str;
245 }

247 /**
248  * 
249  * @ignore
250  */
251 FM.tokenize = function(argsstr) {
252     var i, instr;
253 
254     // napravi listu tokena
255     var elm_array = [];
256     var st_array = argsstr.split('"');
257 
258     instr = false;
259     for (i = 0; i < st_array.length; i++) {
260         // ako nisi u stringu
261         if (!instr) {
262             var e = st_array[i].split(/[\s,]+/);
263             for (var j = 0; j < e.length; j++) {
264                 if (e[j] != "")
265                     elm_array.push(e[j]);
266             }
267         } else {
268             elm_array.push(st_array[i]);
269         }
270 
271         instr = !instr;
272     }
273 
274     return elm_array;
275 }

278 /**
279  * Uppercase the first character of each word  in a input string except the first one.
280  * The definition of a word is any string of characters that is immediately 
281  * after a whitespace or minus sign.
282  * 
283  * @static
284  * @function 
285  * @param {string} input The input string.
286  * @returns {string} 
287  */
288 FM.camelCase = function(input) {
289     return input.replace(/-([a-z])/ig, function(all, letter) {
290         return letter.toUpperCase();
291     });
292 }
293 
294 /**
295  * Uppercase the first character of each word in a string.
296  * The definition of a word is any string of characters that is immediately 
297  * after a whitespace or minus sign.
298  * 
299  * @static
300  * @function 
301  * @param {string} input The input string.
302  * @returns {string} 
303  */
304 FM.ucwords = function(input) {
305     var words = input.split(/(\s|-)+/),
306         output = [];
307 
308     for (var i = 0, len = words.length; i < len; i += 1) {
309         output.push(words[i][0].toUpperCase() +
310             words[i].toLowerCase().substr(1));
311     }
312 
313     return output.join('');
314 }
