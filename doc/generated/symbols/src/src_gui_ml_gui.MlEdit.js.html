1 /**
  2  * ML observer value edit extensions class. 
  3  * 
  4  * @class FM.MlEdit
  5  * @extends FM.MlExtension
  6  * @param {object} [attrs] DOM node attributes
  7  * @param {DOMnode} node DOM node
  8  */
  9 FM.MlEdit = FM.defineClass('MlEdit', FM.MlExtension);
 10 
 11 // methods
 12 FM.MlEdit.prototype._init = function(attrs, node) {
 13     this._super("_init", attrs, node);
 14     this.objectSubClass = "Edit";
 15     this.triggerEvent = this.getAttr('data-fmml-update-condition', 'blur') +
 16         ".fmMlEdit"
 17         ;
 18     this.editWidget = null;
 19 }
 20 
 21 FM.MlEdit.prototype.run = function(obs) {
 22     this._super("run", obs);
 23 
 24     if (this.getObserver()) {
 25         this.getObserver().addRenderer(this);
 26     }
 27 }
 28 
 29 FM.MlEdit.prototype.dispose = function(obs) {
 30     if (this.getObserver()) {
 31         this.getObserver().removeRenderer(this);
 32     }
 33 
 34     this._super("dispose");
 35 }

 37 // methods
 38 FM.MlEdit.prototype._setNodeValue = function(node, value) {
 39     var doSelection =false;
 40     try {
 41         doSelection = node.selectionStart ? true : false;
 42     } catch(e){
 43         
 44     }
 45     var selStart = 0,selEnd = 0;
 46     if (doSelection) {
 47         selStart = node.selectionStart;
 48         selEnd = node.selectionEnd;
 49     }
 50     
 51     
 52     if (node.nodeName == 'INPUT' || node.nodeName == 'TEXTAREA') {
 53         if ($(node).is(':checkbox')) {
 54             if (value && value != '' && value.toLowerCase() != 'false') {
 55                 $(node).attr('checked', 'checked');
 56             } else {
 57                 $(node).removeAttr('checked');
 58             }
 59         } else if ($(this.node).is(':radio')) {
 60             $("input:radio[name ='" + $(node).attr("name") + "']").val([value]);
 61         } else {
 62             $(node).val(value);
 63         }
 64     } else if (node.nodeName == 'IMG') {
 65         value = $(node).attr("src");
 66     } else if (node.nodeName == 'A') {
 67         value = $(node).attr("href");
 68     } else if (FM.isset(node.value)) {
 69         $(node).val(value);
 70     } else {
 71         $(node).html(value);
 72     }
 73 
 74     // selection range restore
 75     if (doSelection) {
 76         node.setSelectionRange(selStart, selEnd);
 77     }
 78 }

 80 FM.MlEdit.prototype.widgetHide = function() {
 81     $(this.editWidget).hide();
 82     $(this.getNode()).show();
 83 }
 84 
 85 FM.MlEdit.prototype.widgetShow = function() {
 86     $(this.getNode()).hide();
 87     $(this.editWidget).show();
 88     this.getObserver().setNodeValue(true);
 89     $(this.editWidget).focus();
 90 
 91 }
 92 
 93 FM.MlEdit.prototype.widgetCb = function(node) {
 94     if (!this.isRendererEnabled()) {
 95         return true;
 96     }
 97 
 98     //var node = this.editWidget ? this.editWidget : this.getNode();
 99     if (!node) {
100         return true;
101     }
102     var value =
103         $(node).is(':checkbox') ?
104         ($(node).is(":checked") ? 'true' : 'false') : (
105             $(node).is(':radio') ? 
106             $("input:radio[name ='" + $(node).attr("name") + "']:checked").val() :
107             (FM.isset(node.value) ? $(node).val() : $(node).html())
108         );
109     if(FM.isArray(value) && value.length < 2) {
110         value = value.length==0 ? "" : value[0];
111     }
112     this.getObserver().setValue(value);
113     return true;
114 }

116 FM.MlEdit.prototype.initWidget = function() {
117     var domnode = this.getNode();
118     var me = this;
119     this.editWidget = null;
120     var nType = domnode.nodeName;
121 
122     if (nType == 'INPUT' || nType == 'TEXTAREA' || nType == 'SELECT') {
123         if ($(domnode).is(':radio')) {
124             $("input:radio[name ='" + $(domnode).attr("name") + "']").bind(this.triggerEvent, function() {
125                 return me.widgetCb(this);
126             });
127         } else {
128             $(domnode).bind(this.triggerEvent, function(a,b,c) {
129                 return me.widgetCb(this);
130             });
131         }
132     } else {
133         var edittype = this.getAttr('data-fmml-attr-edittype', 'input');
134 
135         if (edittype != 'input') {
136             this.editWidget = $("<" + edittype + " class='fmmlValueInput'></" + edittype + ">")[0];
137         } else {
138             this.editWidget = $("<input type='text' class='fmmlValueInput'>")[0];
139         }
140 
141         $(this.editWidget).bind(this.triggerEvent, function() {
142             return me.widgetCb(this);
143         });
144 
145         $(domnode).bind("click.fmMlEdit", function() {
146             return me.widgetShow();
147         });
148         $(this.editWidget).bind("blur.fmMlEdit", function() {
149             return me.widgetHide();
150         });
151         $(this.editWidget).hide();
152         $(this.getNode()).after(this.editWidget);
153     }
154 }
155 
156 FM.MlEdit.prototype.disposeWidget = function() {
157     if (this.editWidget) {
158         this.widgetHide();
159         $(this.editWidget).unbind(this.triggerEvent);
160         $(this.editWidget).unbind("blur.fmMlEdit");
161         $(this.getNode()).unbind("click.fmMlEdit");
162         $(this.editWidget).remove();
163         this.editWidget = null;
164     } else {
165         if ($(this.getNode()).is(':radio')) {
166             $("input:radio[name ='" + $(this.getNode()).attr("name") + "']").unbind(this.triggerEvent);
167         } else {
168             $(this.getNode()).unbind(this.triggerEvent);
169         }
170     }
171 }
172 
173 
174 FM.MlEdit.prototype.render = function(value) {
175     if (this.editWidget) {
176         this._setNodeValue(this.editWidget, value);
177     }
178     this._setNodeValue(this.node, value);
179 }
180 
181 FM.MlEdit.prototype.enableRenderer = function() {
182     if (!this.isRendererEnabled()) {
183         this.initWidget();
184     }
185     this._super("enableRenderer");
186 }
187 
188 FM.MlEdit.prototype.disableRenderer = function() {
189     if (this.isRendererEnabled()) {
190         this.disposeWidget();
191     }
192     this._super("disableRenderer");
193 }
