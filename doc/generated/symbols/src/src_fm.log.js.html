119 FM.logObjectMsgToArray = function(obj) {
120     if (!FM.isset(obj) || !obj)
121         return [];
122     if (FM.isset(obj.length))
123         return obj;
124     if (FM.isset(obj.toLogArray))
125         return(obj.toLogArray());
126     var ar = ['('];
127     for (var id in obj) {
128         ar.push(
129             '  ' + id + ":" + (obj[id] === null ? 'null' : (
130             FM.isset(obj[id].getClassName) ? obj[id].getClassName() :
131             (
132                 FM.isFunction(obj[id]) ? "function() {...}" :
133                 (
134                     FM.isArray(obj[id]) ? "[...]" :
135                     (
136                         FM.isObject(obj[id]) ? "{...}" :
137                         (
138                             obj[id]
139                             )
140                         )
141                     )
142                 )
143             ))
144             );
145     }
146     ar.push(")");
147     return ar;
148 }

157

174 FM.getCallerInfo = function(shift) {
175     var strace = FM.getStackTrace();
176     var pos1, pos2, name, file;
177     var lin = "<anonymous>";
178 
179     shift = FM.isset(shift) ? shift : 0;
180     // mi smo na 1
181     if (strace.length < 2 + shift)
182         return lin;
183 
184     lin = strace[1 + shift];
185     pos1 = lin.indexOf("at ");
186     if (pos1 < 0)
187         return lin;
188 
189     lin = lin.substr(pos1 + 3);
190     pos1 = lin.indexOf("(");
191     pos2 = lin.indexOf(")");
192     if (pos1 > -1 && pos2 > -1) {
193         name = lin.substr(0, pos1);
194         file = lin.substr(pos1 + 1, pos2 - pos1 - 1);
195         pos1 = file.lastIndexOf("/");
196         if (pos1 > -1) {
197             file = file.substring(pos1 + 1);
198         }
199         lin = name + "(" + file + ")";
200     }
201     return(lin);
202 }

205 /**
206  *  Log a message.
207  * 
208  * @static
209  * @function  
210  * @param {FM.Object|null} oObj Context.
211  * @param {string|Object|...} msg Variable to be logged.
212  * @param {number} [level] Log level.
213  * @param {string} [callerinfo] Description of context.
214  */
215 FM.log = function(oObj, msg, level, callerinfo) {
216     var minlevel =
217         oObj && FM.isset(oObj.getLogLevel) && oObj.getLogLevel() != null ?
218         oObj.getLogLevel() : FM.getLogLevel()
219         ;
220     if (!FM.isset(level))
221         level = FM.logLevels.info;
222     if (level < minlevel)
223         return false;
224     if (FM.isset(callerinfo) && callerinfo.indexOf('.') < 0) {
225         callerinfo = FM.getLogId(oObj) + '.' + callerinfo;
226     } else if (!FM.isset(callerinfo)) {
227         callerinfo = FM.getLogId(oObj) +
228             '.' +
229             (FM.isset(FM.log.caller.name) && FM.log.caller.name != '' ? FM.log.caller.name : '<unknown>');
230     }
231 
232     // formiraj header    
233     console.log(
234         (FM.isset(callerinfo) ? callerinfo : "Unknown") +
235         " [" + FM.getLogTypeName(level) + "]:" +
236         (msg && !FM.isString(msg) ? '' : msg)
237         );
238 
239     if (FM.isObject(msg) || FM.isArray(msg)) {
240         console.dir(msg);
241     }
242     return true;
243 }
