 11 FM.Object.prototype._init = function(attrs, flds) {
 12     // properties    
 13     this.objectSubClass = "Objects";
 14     this.id = null;
 15     this.objectLogLevel = null;
 16 
 17     this.enabled = true;
 18     this.listenersArr = {};
 19     this.undoList = {};
 20     this.options = {};
 21     this.strictOptions = false;
 22 
 23     // properties
 24     this.prop = {
 25         dirty: false,
 26         timestamp: new Date().getTime(),
 27         fetched: true // da li je new loc ili je fetchan
 28     },
 29     this.setAttr(false, FM.isset(flds) ? flds : {}, false);
 30     this.strictOptions = FM.isset(flds) ? true : false;
 31     if (FM.isset(attrs) && attrs) {
 32         if (FM.isString(attrs))
 33             attrs = FM.stringPtrToObject(attrs);
 34         if (FM.isFunction(attrs))
 35             attrs = attrs();
 36         this.setAttr(this.strictOptions, attrs, false);
 37     }
 38 
 39     this.setChanged(false, false);
 40 }

 42 /**
 43  * Get FM class name.
 44  * @public     
 45  * @function 
 46  * @returns {string} Returns name of the class.
 47  */
 48 FM.Object.prototype.getClassName = function() {
 49     var o = this;
 50 
 51     while (o && !FM.isset(o.constructor.className)) {
 52         o = o._parent ? o._parent : null;
 53     }
 54     return(o ? o.constructor.className : '');
 55 }

 92 /**
 93  * Add listener.
 94  * 
 95  * @public     
 96  * @function 
 97  * @param {FM.Object|object} oListener FM.Object to register as listener or object with event functions.
 98  * @param {object} [config] Additional options.
 99  */
100 FM.Object.prototype.addListener = function(oListener, config) {
101     // definicija listenera
102     var lstnrdef = {
103         listener: oListener,
104         config: FM.isset(config) ? config : {},
105         iscallback: !FM.isset(oListener.onEvent)
106     };
107 
108     // if the listener is not object
109     if (!FM.isset(oListener.getID)) {
110         var lid = '_CB_' + FM.generateNewID();
111         /** @ignore */
112         oListener.getID = function() {
113             return lid;
114         }
115     }
116     this.listenersArr[oListener.getID()] = lstnrdef;
117 }

119 /**
120  * Remove listener.
121  * 
122  * @public     
123  * @function 
124  * @param {FM.Object|object} oListener Listener to remove.
125  */
126 FM.Object.prototype.removeListener = function(oListener) {
127     if (!FM.isset(oListener) || !oListener || !FM.isset(oListener.getID))
128         return false;
129 
130     var nlist = {};
131     var objId = oListener.getID();
132     if (!objId)
133         return false;
134 
135     for (var id in this.listenersArr) {
136         if (objId != id)
137             nlist[id] = this.listenersArr[id];
138     }
139 
140     this.listenersArr = nlist;
141 
142     return true;
143 }

157 /**
158  * Event function.
159  * 
160  * @public     
161  * @function 
162  * @param {FM.Object} sender Sender of event
163  * @param {string} ev Event
164  * @param {} data Event data.
165  */
166 FM.Object.prototype.onEvent = function(sender, ev, data, calledlist) {
167     var cl = FM.isset(calledlist) ? calledlist : {};
168     if (!this.isEnabled())
169         return false;
170 
171     //  ako ima≈° event fn
172     if (FM.isset(this[ev])) {
173         this[ev](sender, data);
174         cl[this.getID()] = '1';
175         FM.setAttr(cl, '_executed', '1');
176         //return cl;
177     }
178 
179     // proslijedi dalje ako nemas ev fn
180     //return this.fireEvent(ev,data,cl);
181     return cl;
182 }
183 
184 /**
185  * Send event to all listeners.
186  * 
187  * @public     
188  * @function 
189  * @param {string} ev Event to send.
190  * @param {} evdata Event data.
191  * 
192  */
193 FM.Object.prototype.fireEvent = function(ev, evdata, calledlist) {
194     var cl = FM.isset(calledlist) ? calledlist : {};
195     if (FM.getAttr(cl, '_executed', '0') == '1')
196         return cl;
197 
198     cl[this.getID()] = '0';
199 
200     // obicni listeneri
201     var larr = this.listenersArr;
202     for (var id in larr) {
203         var ldef = larr[id];
204         if (!FM.isset(cl[id])) {
205             cl[id] = "0";
206             try {
207                 if (ldef.iscallback) {
208                     if (FM.isFunction(ldef.listener[ev])) {
209                         ldef.listener[ev](this, evdata);
210                         cl[ldef.listener.getID()] = '1';
211                         FM.setAttr(cl, '_executed', '1');
212                     }
213                 } else {
214                     cl = ldef.listener.onEvent(this, ev, FM.isset(evdata) ? evdata : {}, cl);
215                 }
216             } catch (err) {
217                 FM.log(null, err, FM.logLevels.error, 'Object.fireEvent');
218             }
219             //if(FM.getAttr(cl,'_executed','0') == '1') break;            
220         }
221     }
222 
223     // kraj
224     return cl;
225 }

252 /**
253  * Set attribute value.
254  * 
255  * @static
256  * @function 
257  * @param {string} key Attribute name.
258  * @param {string|number|...} val Value of attribute.
259  * @returns {boolean} <i>true</i> if value of attribute is changed, otherwise <i>false</i>.
260  */
261 FM.Object.prototype.setAttr = function(key, val, callevent) {
262     if (FM.setAttr(this.options, this.undoList, key, val)) {
263         this.setProperty('dirty', true);
264         this.setProperty('timestamp', new Date().getTime());
265         if (FM.isset(callevent) && callevent == true)
266             this.fireEvent("onChange", this);
267     }
268 }

284 /**
285  * Check if <i>key</i> attribute is changed.
286  * 
287  * @public
288  * @function 
289  * @param {string} key Attribute name.
290  * @returns {boolean} 
291  */
292 FM.Object.prototype.isChanged = function(key) {
293     if (FM.isset(key)) {
294         if (FM.isset(this.undoList[key]))
295             return true;
296         return false;
297     }
298     return this.getProperty('dirty');
299 }

301 /**
302  * Set FM object <i>changed</i> property,
303  * 
304  * @public
305  * @function 
306  * @param {boolean} v true or false
307  * @param {boolean} callevent Fire <i>onChange</i> event after object update (default is false).
308  */
309 FM.Object.prototype.setChanged = function(v, callevent) {
310     this.setProperty('dirty', v == true);
311     if (!this.getProperty('dirty'))
312         this.undoList = {};
313     if (FM.isset(callevent) && callevent == true)
314         this.fireEvent("onChange", this);
315 }
316 
317 /**
318  * Discard object attribute changes,
319  * 
320  * @public
321  * @function 
322  */
323 FM.Object.prototype.discardChanges = function() {
324     for (var option in this.undoList) {
325         this.setAttr(option, this.undoList[option]);
326     }
327     this.setChanged(false, true);
328 }

337 /**
338  * Return an unique string based on object data.
339  * 
340  * @public
341  * @function
342  */
343 FM.Object.prototype.getDataHash = function() {
344     var dataHash = '';
345     for (var dataName in this.getAttr()) {
346         var dataValue = this.d(dataName, '');
347 
348         dataHash += dataHash == '' ? dataValue : ',' + dataValue;
349     }
350     return FM.md5(dataHash);
351 }
