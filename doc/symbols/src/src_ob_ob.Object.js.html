 14 FM.Object.prototype._init = function(attrs, flds) {
 15     // properties    
 16     this.objectSubClass = "Objects";
 17     this.id = null;
 18     this.objectLogLevel = null;
 19 
 20     this.enabled = true;
 21     this.listenersArr = {};
 22     this.undoList = {};
 23     this.options = {};
 24     this.strictOptions = false;
 25 
 26     // properties
 27     this.prop = {
 28         dirty: false,
 29         timestamp: new Date().getTime(),
 30         fetched: true // da li je new loc ili je fetchan
 31     },
 32     this.setAttr(false, FM.isset(flds) ? flds : {}, false);
 33     this.strictOptions = FM.isset(flds) ? true : false;
 34     if (FM.isset(attrs) && attrs) {
 35         if (FM.isString(attrs))
 36             attrs = FM.stringPtrToObject(attrs);
 37         if (FM.isFunction(attrs))
 38             attrs = attrs();
 39         this.setAttr(this.strictOptions, attrs, false);
 40     }
 41 
 42     this.setChanged(false, false);
 43 }

 45 /**
 46  * Get  FM class name
 47  * @public     
 48  * @function 
 49  * @returns {string} Returns name of the object
 50  */
 51 FM.Object.prototype.getClassName = function() {
 52     var o = this;
 53 
 54     while (o && !FM.isset(o.constructor.className)) {
 55         o = o._parent ? o._parent : null;
 56     }
 57     return(o ? o.constructor.className : '');
 58 }

 93 /**
 94  * Add listener
 95  * @public     
 96  * @function 
 97  * @param {FM.Object|object} oListener FM.Object to register as listener or object with event functions
 98  * @param {object} [config] Additional options
 99  */
100 FM.Object.prototype.addListener = function(oListener, config) {
101 
102     // definicija listenera
103     var lstnrdef = {
104         listener: oListener,
105         config: FM.isset(config) ? config : {},
106         iscallback: !FM.isset(oListener.onEvent)
107     };
108 
109     // if the listener is not object
110     if (!FM.isset(oListener.getID)) {
111         var lid = '_CB_' + FM.generateNewID();
112         /** @ignore */
113         oListener.getID = function() {
114             return lid;
115         }
116     }
117     this.listenersArr[oListener.getID()] = lstnrdef;
118 }
119 
120 /**
121  * Remove listener
122  * @public     
123  * @function 
124  * @param {FM.Object|object} oListener 
125  */
126 FM.Object.prototype.removeListener = function(oListener) {
127     if (!FM.isset(oListener) || !oListener || !FM.isset(oListener.getID))
128         return false;
129 
130     var nlist = {};
131     var objId = oListener.getID();
132     if (!objId)
133         return false;
134 
135     for (var id in this.listenersArr) {
136         if (objId != id)
137             nlist[id] = this.listenersArr[id];
138     }
139 
140     this.listenersArr = nlist;
141 
142     return true;
143 }

156 /**
157  * Event function
158  * @public     
159  * @function 
160  * @param {FM.Object} sender Sender of event
161  * @param {String} ev Event
162  * @param {String} data Event data
163  * @param {Object} calledlist list of olready called listeners
164  */
165 FM.Object.prototype.onEvent = function(sender, ev, data, calledlist) {
166     var cl = FM.isset(calledlist) ? calledlist : {};
167     if (!this.isEnabled())
168         return false;
169 
170     //  ako imaš event fn
171     if (FM.isset(this[ev])) {
172         this[ev](sender, data);
173         cl[this.getID()] = '1';
174         FM.setAttr(cl, '_executed', '1');
175         //return cl;
176     }
177 
178     // proslijedi dalje ako nemas ev fn
179     //return this.fireEvent(ev,data,cl);
180     return cl;
181 }
182 
183 /**
184  * Send event to all listeners
185  * @public     
186  * @function 
187  * @param {String} ev Event
188  * @param {String} evdata Event data
189  */
190 FM.Object.prototype.fireEvent = function(ev, evdata, calledlist) {
191     var cl = FM.isset(calledlist) ? calledlist : {};
192     if (FM.getAttr(cl, '_executed', '0') == '1')
193         return cl;
194 
195     cl[this.getID()] = '0';
196 
197     // obicni listeneri
198     var larr = this.listenersArr;
199     for (var id in larr) {
200         var ldef = larr[id];
201         if (!FM.isset(cl[id])) {
202             cl[id] = "0";
203             try {
204                 if (ldef.iscallback) {
205                     if (FM.isFunction(ldef.listener[ev])) {
206                         ldef.listener[ev](this, evdata);
207                         cl[ldef.listener.getID()] = '1';
208                         FM.setAttr(cl, '_executed', '1');
209                     }
210                 } else {
211                     cl = ldef.listener.onEvent(this, ev, FM.isset(evdata) ? evdata : {}, cl);
212                 }
213             } catch (err) {
214                 FM.log(null, err, FM.logLevels.error, 'Object.fireEvent');
215             }
216             //if(FM.getAttr(cl,'_executed','0') == '1') break;            
217         }
218     }
219 
220     // kraj
221     return cl;
222 }

247 /**
248  * Set value of <i>key</i> attribute to <i>val</i> value
249  * @public
250  * @function
251  * @param {string} key Attribute name
252  * @param {string} val Value of attribute
253  * @param {boolean} callevent Fire <i>onChange</i> after object update (default is true)
254  */
255 FM.Object.prototype.setAttr = function(key, val, callevent) {
256     if (FM.setAttr(this.options, this.undoList, key, val)) {
257         this.setProperty('dirty', true);
258         this.setProperty('timestamp', new Date().getTime());
259         if (FM.isset(callevent) && callevent == true)
260             this.fireEvent("onChange", this);
261     }
262 }

280 /**
281  * Check if <i>key</i> attribute is changed
282  * @public
283  * @function 
284  * @param {string} key Attribute name
285  * @returns {boolean} Returns <i>true</i> if attribute is changed 
286  */
287 FM.Object.prototype.isChanged = function(key) {
288     if (FM.isset(key)) {
289         if (FM.isset(this.undoList[key]))
290             return true;
291         return false;
292     }
293     return this.getProperty('dirty');
294 }
295 
296 /**
297  * Set object <i>changed</i> property 
298  * @public
299  * @function 
300  * @param {boolean} v true or false
301  * @param {boolean} callevent Fire <i>onChange</i> after object update (default is false)
302  */
303 FM.Object.prototype.setChanged = function(v, callevent) {
304     this.setProperty('dirty', v == true);
305     if (!this.getProperty('dirty'))
306         this.undoList = {};
307     if (FM.isset(callevent) && callevent == true)
308         this.fireEvent("onChange", this);
309 }
310 
311 /**
312  * Discard object attributes changes
313  * @public
314  * @function 
315  */
316 FM.Object.prototype.discardChanges = function() {
317     for (var option in this.undoList) {
318         this.setAttr(option, this.undoList[option]);
319     }
320     this.setChanged(false, true);
321 }

327 /**
328  * Return an unique string based on object data
329  * 
330  * @public
331  * @function
332  */
333 FM.Object.prototype.getDataHash = function() {
334     var dataHash = '';
335     for (var dataName in this.getAttr()) {
336         var dataValue = this.d(dataName, '');
337 
338         dataHash += dataHash == '' ? dataValue : ',' + dataValue;
339     }
340     return FM.md5(dataHash);
341 }

370  * @param [boolean] callevent Fire <i>onChange</i> after object update (default is true)

372 FM.Object.prototype.setProperty = function(key, val, callevent) {
373     if (FM.setAttr(this.prop, null, key, val)) {
374         if (FM.isset(callevent) && callevent == true)
375             this.fireEvent("onChange", this);
376     } else {
377         return false;
378     }
379     return true;
380 }
