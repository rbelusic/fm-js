  5 * @extends FM.Object

  8 FM.UtTranslations = FM.defineClass('UtTranslations',FM.Object);
  9 FM.UtTranslations.className = "UtTranslations";

 11 
 12 FM.UtTranslations.prototype._init = function(attrs) {
 13     this._super("_init",attrs);
 14     
 15     // properties
 16     this.translationsLoaded = false;
 17     this.loadedTranslations = {};
 18     this.missingTranslations = {};
 19     this.objectSubClass = "UtTranslations";
 20 }
 21     
 22 FM.UtTranslations.fetchTranslations = function(app,cbfn) {
 23     var callbackFn = FM.isset(cbfn) && FM.isFunction(cbfn) ? cbfn : function() {};
 24     var dmlist = new FM.DmList({},'getTranslations',app);     
 25     var lstnr = {
 26         onListEnd: function(sender,data) {
 27             FM.UtTranslations.translationsLoaded = true;
 28             
 29             FM.forEach(data.Added,function(id, obj) {
 30                 // obj is generic value
 31                 var dataid = FM.md5(obj.getAttr('value.text',''));
 32                 
 33                 FM.UtTranslations.loadedTranslations[dataid] = {
 34                     id: dataid,
 35                     text: obj.getAttr('text',''),
 36                     translation: obj.getAttr('translation','')
 37                 };
 38                 return true;
 39             });
 40             dmlist.removeListener(lstnr);
 41             dmlist.dispose();
 42             callbackFn(true);
 43             return true;
 44         },
 45         onListError: function(sender,data) {
 46             FM.UtTranslations.translationsLoaded = true;
 47             
 48             dmlist.removeListener(lstnr);
 49             dmlist.dispose();
 50             callbackFn(false,new FM.DmGenericError({
 51                 messageId: -1,
 52                 text: data
 53             }));
 54             return true;
 55         }
 56     };
 57     dmlist.addListener(lstnr);
 58     dmlist.getData();    
 59 }
 60     
 61 _T = function() { // text,app, params
 62     if(arguments.length < 2) return('');
 63 
 64     // nadji hash i prevedeni string
 65     var hash = FM.md5(arguments[0]);
 66     var str;
 67     
 68     if(FM.isset(FM.UtTranslations.loadedTranslations[hash])) {
 69         str = FM.UtTranslations.loadedTranslations[hash].translation;
 70     } else {
 71         str = arguments[0];
 72         FM.UtTranslations.missingTranslations[hash]=arguments[0];
 73     }
 74 
 75     // ubaci podatke
 76     for(var i = 2; i < arguments.length; i++)  {
 77         str = str.replace("[:" + i + "]", arguments[i]);
 78     }
 79 
 80     // kraj
 81     return(str);    
 82 }
