  5 FM._findClassWithMethod = function(o, m) {
  6     while (
  7         FM.isset(o) && o &&
  8         (FM.getAttr(o, 'constructor.prototype.' + m, null) == FM.getAttr(o, '_parent.constructor.prototype.' + m, null))
  9         ) {
 10         o = o._parent;
 11     }
 12     return FM.isset(o) && o ? o : null;
 13 
 14 }

 17  * @ignore
 18  * 
 19  */
 20 FM._super_stack = function(me, method, on) {
 21     var mStack = FM.getAttr(me, '_parent_call_stack', []);
 22 
 23     if (on) {
 24         // only class with same method can call super
 25         var startcls = FM._findClassWithMethod(
 26             mStack.length == 0 || mStack[mStack.length - 1].m != method ?
 27             me :
 28             mStack[mStack.length - 1].oext,
 29             method
 30             );
 31 
 32 
 33         mStack[mStack.length] = {
 34             o: startcls,
 35             m: method,
 36             oext: FM._findClassWithMethod(startcls._parent, method)
 37         };
 38         me._parent_call_stack = mStack;
 39         return mStack;
 40 
 41         // new stack
 42         if (mStack.length == 0 || mStack[mStack.length - 1].m != method) {
 43             mStack[mStack.length] = {
 44                 o: me, //FM._findClassWithMethod(me._parent,method), 
 45                 m: method
 46             };
 47 
 48         } else { // old stack
 49             mStack[mStack.length] = {
 50                 o: mStack[mStack.length - 1].o, //FM._findClassWithMethod(mStack[mStack.length-1].o._parent,method), 
 51                 m: method
 52             }
 53         }
 54     } else {
 55         mStack = Array.prototype.slice.call(mStack, 0, mStack.length - 1);
 56     }
 57 
 58     me._parent_call_stack = mStack;
 59     return mStack;
 60 }

 62 /**
 63  * @ignore

 65  */
 66 FM._super = function() {
 67     var me = arguments[0]
 68     var callArgs = arguments[1];
 69     var method = callArgs[0];
 70 
 71     var mStack = FM._super_stack(me, method, true);
 72     try {
 73         // nadji klasu od koje polazis
 74         var fnThis = mStack && mStack.length > 0 && mStack[mStack.length - 1].oext ?
 75             FM._findClassWithMethod(mStack[mStack.length - 1].oext, method) :
 76             null
 77             ;
 78 
 79         //if(fnThis) var retc = fnThis[method].apply(me, Array.prototype.slice.call(callArgs, 1));
 80         var retc = fnThis ?
 81             fnThis[method].apply(me, Array.prototype.slice.call(callArgs, 1))
 82             : undefined;
 83 
 84         // makni stack
 85         FM._super_stack(me, method, false);
 86 
 87         return retc;
 88     } catch (e) {
 89         // makni stack
 90         FM._super_stack(me, method, false);
 91         return undefined;
 92     }
 93 
 94 
 95 }

113 FM.extend = function(oDest, oSrc, isclass) {
114     isclass = FM.isset(isclass) && isclass;
115     oDest = oDest ? oDest : {};
116     oSrc = oSrc ? oSrc : {};
117 
118     for (var property in oSrc) {
119         oDest[property] = oSrc[property];
120     }
121 
122     if (FM.isset(isclass) && isclass == true) {
123         oDest._super = function() {
124             return FM._super(this, arguments);
125         }
126     }
127 
128     return oDest;
129 }

131 /**
132  * Create new FM class.
133  * 
134  * @param {string} name Name of new class.
135  * @param {FM.Object} [ext=null] FM class to extend.
136  * @returns {function}
137  */
138 FM.defineClass = function(name, ext) {
139     name = FM.isset(name) && FM.isString(name) && name != '' ? name : 'Undefined';
140     ext = FM.isset(ext) && FM.isFunction(ext) ? ext : null;
141     var cls = function() {
142         if (this._init)
143             this._init.apply(this, arguments);
144     };
145 
146     FM.extendClass(cls, ext);
147     cls.className = name;
148 
149     return cls;
150 }

152 /**
153  * @ignore
154  * 
155  */
156 FM.extendClass = function(oDest, oSrc) {
157     if (oSrc) {
158         for (var property in oSrc.prototype) {
159             oDest.prototype[property] = oSrc.prototype[property];
160         }
161         oDest.prototype._parent = oSrc.prototype;
162     } else {
163         oDest.prototype._parent = null;
164     }
165 
166     oDest.prototype._super = function() {
167         return FM._super(this, arguments);
168     }
169 
170     return oDest;
171 }

177 FM.stringPtrToObject = function(objptr, lm, app) {
178     var akeys = objptr.split(".");
179     if (akeys.length < 1)
180         return null;
181 
182     var parent = akeys[0] == 'APP' ? app : (
183         akeys[0] == 'LM' ? lm : window
184         );
185     var startIndex = parent == window ? 0 : 1;
186 
187     for (var i = startIndex; i < akeys.length; i++) {
188         if (!FM.isset(parent[akeys[i]]))
189             return null;
190         parent = parent[akeys[i]];
191     }
192 
193     return parent;
194 }
