1 /**
  2  * ML list of values edit extensions class. 
  3  * 
  4  * @class FM.MlListOfValues
  5  * @extends FM.MlEdit
  6  * @param {object} [attrs] DOM node attributes
  7  * @param {DOMnode} node DOM node
  8  */
  9 /*
 10  * data-fmml-update-condition   - update event blur def
 11  * data-fmml-list               - dm list conf
 12  * data-fmml-list-id-attr       - list attr which correspond to attr of observer
 13  * data-fmml-list-id            - menu id attr
 14  * data-fmml-list-text-attr     - menu text attr
 15  * data-fmml-list-def-selected  - menu def value
 16  * data-fmml-list-def-selected-attr -
 17  * data-fmml-list-allow-null    - add empty row in menu
 18  * data-fmml-list-depends-of    - list of host attributes to send
 19  */
 20 FM.MlListOfValues = FM.defineClass('MlListOfValues', FM.MlEdit);
 21 
 22 // methods
 23 FM.MlListOfValues.prototype._init = function(attrs, node) {
 24     this._super("_init", attrs, node);
 25     this.objectSubClass = "ListOfValues";
 26 
 27     // LOV
 28     this.lastLovArguments = null;
 29     this.lovDependsOf = null;
 30     this.lovItemsTemplate = null;
 31     this.dmList = null;
 32 }
 33 
 34 FM.MlListOfValues.prototype.run = function(obs) {
 35     this._super("run", obs);
 36 
 37     // if dependsof attrs are not jet collected 
 38     var dependsOfStr = obs.getAttr('data-fmml-list-depends-of', '');
 39     this.lovDependsOf = dependsOfStr == '' ? [] : dependsOfStr.split(',');
 40 
 41     // create LOV
 42     var me = this;
 43     this.lovFetch(this.lovGetFetchArguments(), function(isok, list) {
 44         me.lovDisplay(isok, list);
 45     });
 46 }

 49 // fetch values for LOV from server
 50 FM.MlListOfValues.prototype.lovFetch = function(args, callback) {
 51 
 52     // get observer
 53     var obs = this.getObserver();
 54     if (!obs) {
 55         return false;
 56     }
 57 
 58     // create dmlist
 59     args = FM.isset(args) && args && FM.isObject(args) ? args : {};
 60     callback = FM.isset(callback) && callback && FM.isFunction(callback) ? callback : function() {
 61     };
 62     var dmconfName = obs.getAttr('data-fmml-list', '');
 63     if (this.dmList) {
 64         this.dmList.dispose();
 65     }
 66     this.dmList = new FM.DmList(args, dmconfName, obs.getApp());
 67     this.lastLovArguments = args;
 68 
 69     // prepare listener
 70     var me = this;
 71     var lstnr = {
 72         /** @ignore */
 73         onListStart: function(sender, data) {
 74             me.log("Ajax call started", FM.logLevels.info, 'MlListOfValues.onListStart');
 75             $(me.getNode()).addClass("fmmlWaitState");
 76             return true;
 77         },
 78         /** @ignore */
 79         onListEnd: function(sender, data) {
 80             me.log("Ajax call completed", FM.logLevels.info, 'MlListOfValues.onListEnd');
 81             $(me.getNode()).removeClass("fmmlWaitState");
 82             me.dmList.removeListener(lstnr);
 83             callback(true, me.dmList);
 84             return true;
 85         },
 86         /** @ignore */
 87         onListError: function(sender, data) {
 88             me.log(data, FM.logLevels.error, 'MlListOfValues.onListError');
 89             $(me.getNode()).removeClass("fmmlWaitState");
 90             me.dmList.removeListener(lstnr);
 91             callback(false, null);
 92             return true;
 93         }
 94     };
 95 
 96     me.dmList.addListener(lstnr);
 97     me.dmList.getData();
 98 
 99     return true;
100 }

102 FM.MlListOfValues.prototype.lovEmpty = function() {
103     $(this.editWidget ? this.editWidget : this.getNode()).empty();
104     var lovDesc = this.getAttr(
105         'data-fmml-lov-description', ''

108         $(this.editWidget).append('<legend>' + lovDesc + '</legend>');
109     }

112 FM.MlListOfValues.prototype.lovRenderItem = function(obj, issel, menuId, menuAttr, menuTempl, menuTemplName) {
113     var itmText = '';
114     var me = this;
115     var domnode = this.editWidget ? this.editWidget : this.getNode();
116     var attrs = obj ? obj.getAttr() : {};
117     attrs['lovSelected'] = issel ? 'selected' : '';
118     attrs['lovValue'] = FM.getAttr(attrs, menuId, '');
119     attrs['lovDataID'] = obj ? obj.getDataID() : '';
120 
121     if (menuTemplName != '') {
122         if (this.lovItemsTemplate == null) {
123             FM.UtTemplate.getTemplate(
124                 this.getApp(), attrs, menuTemplName, function(isok, templ) {
125                 if (isok) {
126                     me.lovItemsTemplate = templ;
127                 } else {
128                     me.lovItemsTemplate = '';
129                 }
130                 itmText = $(FM.applyTemplate(
131                     attrs, me.lovItemsTemplate, false, false
132                     ));
133                 $(itmText).attr("data-fmml-object-id", FM.getAttr(attrs, 'lovDataID', ''));
134                 $(domnode).append(itmText);
135             }
136             );
137             return;
138         } else {
139             itmText = FM.applyTemplate(attrs, this.lovItemsTemplate, false, false);
140             itmText = $(itmText);
141         }
142 
143     } else if (menuTempl != '') {
144         itmText = FM.applyTemplate(attrs, this.menuTempl, false, false);
145         itmText = $(itmText);
146     } else {
147         var nType = domnode.nodeName;
148 
149         if (nType == 'SELECT') {
150             itmText =
151                 '<option  class="fmmlOption"' +
152                 (issel ? ' selected="selected"' : '') +
153                 ' value="' + FM.getAttr(attrs, menuId, '') + '">' +
154                 FM.getAttr(attrs, menuAttr, '') +
155                 '</option>'
156                 ;
157             itmText = $(itmText);
158         } else {
159             var itmid = "fm_rg_" + this.getID();
160             itmText =
161                 '<label for="' + itmid + '" class="fmmlRadioLabel">' +
162                 FM.getAttr(attrs, menuAttr, '') + '</label>' +
163                 '<input type="radio" name="' + itmid + '" id="' + itmid + '"' +
164                 (issel ? ' checked="checked"' : '') +
165                 ' value="' + FM.getAttr(attrs, menuId, '') + '"/><br />'
166                 ;
167             itmText = $(itmText);
168             itmText.bind(this.triggerEvent, function() {
169                 return me.widgetCb(this);
170             });
171         }
172     }
173 
174 
175     itmText.attr("data-fmml-object-id", obj.getDataID());
176 
177     $(domnode).append(itmText);
178 }

180 FM.MlListOfValues.prototype.lovDisplay = function(isok) {
181     if (!this.isRendererEnabled() || !this.dmList) {
182         return;
183     }
184 
185     // clear list
186     this.lovEmpty();
187 
188     // get observer & dmobject & obs attribute name
189     var me = this;
190     var obs = this.getObserver();
191 
192     // LOV def data-fmml-list-id-attr > data-fmml-attr-name ref!
193     var menuId = obs.getAttr('data-fmml-list-id-attr', 'id');
194     var menuAttr = obs.getAttr('data-fmml-list-text-attr', '');
195     var menuTempl = obs.getAttr('data-fmml-list-text-template', '');
196     var menuTemplName = obs.getAttr('data-fmml-list-text-template-name', '');
197     var defSelValue = obs.getAttr('data-fmml-list-def-selected', '');
198     if (FM.isFunction(window[defSelValue]))
199         defSelValue = window[defSelValue](); // ref!
200     var defSelAttr = obs.getAttr('data-fmml-list-def-selected-attr', '');
201     var allowNulls = obs.getAttr('data-fmml-list-allow-null', 'false');
202 
203     // state
204     var def = null, first = null, cur = null;
205     var curVal = obs.getValue();
206 
207     // ok, display list
208     if (isok) {
209         // iterate list and find sel value
210         this.dmList.forEachListElement(function(index, obj) {
211             // def
212             if (defSelAttr != '' && defSelValue != '') {
213                 if (obj.getAttr(defSelAttr, '') == defSelValue) {
214                     def = obj;
215                 }
216             }
217             if (!first) {
218                 first = obj;
219             }
220             if (obj.getAttr(menuId, '') == curVal) {
221                 cur = obj;
222             }
223             return(true);
224         });
225 
226         // find selected (current > def > first > none
227         var lovSelected = cur ? cur : (def ? def : first);
228 
229         // null
230         if (allowNulls == 'true') {
231             this.lovRenderItem(null, false, "", "", menuTempl, menuTemplName);
232         }
233 
234         // fill list
235         this.dmList.forEachListElement(function(index, obj) {
236             me.lovRenderItem(
237                 obj, obj == lovSelected,
238                 menuId, menuAttr,
239                 menuTempl, menuTemplName
240                 );
241             return(true);
242         });
243 
244         // check if dm value and lov value are the same
245         this._runHost(true);
246         if (lovSelected) {
247             obs.setValue(lovSelected.getAttr(menuId, ''));
248         }
249     } else { // error
250         this.lovRenderItem(null, false, "", "Error occured", menuTempl, menuTemplName);
251     }
252 }

274 FM.MlListOfValues.prototype._runHost = function(doinit) {
275     var hostToRun = this.getAttr("data-fmml-run-on-lov-change", '');
276     if (hostToRun != '' && this.dmList) {
277         var node = document.getElementById(hostToRun);
278         if (node) {
279             var menuId = this.getAttr('data-fmml-list-id-attr', 'id');
280             var curVal = this.getObserver().getValue();
281             var oObj = this.dmList.get(curVal, menuId);
282             doinit = doinit == true;
283 
284             if (!FM.isset(node.fmmlHost) || !node.fmmlHost) {
285                 if (!doinit || !$(node).attr('data-fmml-host'))
286                     return false;
287                 FM.MlHost.initChildNodes(this.getApp(), node, oObj, false);
288                 if (!FM.isset(node.fmmlHost) || !node.fmmlHost)
289                     return false;
290             }
291 
292             node.fmmlHost.run(oObj);
293             return true;
294         }
295     }
296 
297     return false;
298 }

300 FM.MlListOfValues.prototype.lovGetFetchArguments = function() {
301     var args = {};
302     var obs = this.getObserver();
303     var dmobj = obs.getDmObject();
304     if (dmobj)
305         for (var i = 0; i < this.lovDependsOf.length; i++) {
306             args[this.lovDependsOf[i]] = dmobj.getAttr(this.lovDependsOf[i], '');
307         }
308     return args;
309 }

312 FM.MlListOfValues.prototype.lovCheckDepedOfChanged = function() {
313     var obs = this.getObserver();
314     var dmobj = obs.getDmObject();
315 
316     // menu is not filled yet
317     if (!dmobj || this.lastLovArguments == null)
318         return true;
319 
320 
321     // check attributes
322     var depchanged = false;
323     for (var i = 0; i < this.lovDependsOf.length; i++) {
324         if (
325             FM.getAttr(this.lastLovArguments, this.lovDependsOf[i], '')
326             !=
327             dmobj.getAttr(this.lovDependsOf[i], '')
328             ) {
329             depchanged = true;
330             break;
331         }
332     }
333 
334     // end
335     return(depchanged);
336 }

340 FM.MlListOfValues.prototype.initWidget = function() {
341     var domnode = this.getNode();
342     var me = this;
343     this.editWidget = null;
344     var nType = domnode.nodeName;
345 
346     if (nType == 'SELECT') {
347         this.lovDisplay(true, this.dmList);
348         $(domnode).bind(this.triggerEvent, function() {
349             return me.widgetCb(this);
350         });
351     } else {
352         var edittype = this.getAttr(
353             'data-fmml-attr-edittype',
354             $(domnode).is(':radio') ? 'radio' : 'select'
355             );
356         if (edittype == 'select') {
357             this.editWidget = $("<select class='fmmlValueInput'></select>")[0];
358         } else if (edittype == 'radio') {
359             this.editWidget =
360                 $("<fieldset class='fmmlFieldset fmmlValueGroup'></fieldset>")[0];
361         } else {
362             this.editWidget = $("<" + edittype + " class='fmmlValueInput'></" + edittype + ">")[0];
363         }
364         $(domnode).after(this.editWidget);
365         if ($(domnode).is(':radio') || edittype == 'radio') {
366             this.widgetShow();
367         } else {
368             $(domnode).bind("click.fmMlEdit", function() {
369                 return me.widgetShow();
370             });
371             $(this.editWidget).bind("blur.fmMlEdit", function() {
372                 return me.widgetHide();
373             });
374             this.widgetHide();
375         }
376 
377         this.lovDisplay(true, this.dmList);
378         if (edittype != 'radio') {
379             $(this.editWidget).bind(this.triggerEvent, function() {
380                 return me.widgetCb(this);
381             });
382         }
383     }
384 }

390 FM.MlListOfValues.prototype.update = function(obs) {
391     this._super("update", obs);
392 
393     // if dep value are not changed
394     if (!this.lovCheckDepedOfChanged()) {
395         return true;
396     }
397 
398     var me = this;
399     this.lovFetch(this.lovGetFetchArguments(), function(isok, list) {
400         me.lovDisplay(isok, list);
401     });
402 }
