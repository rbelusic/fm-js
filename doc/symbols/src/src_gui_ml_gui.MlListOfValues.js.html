 33 // fetch values for LOV from server
 34 FM.MlListOfValues.prototype.lovFetch = function(args,callback) {
 35     
 36     // get observer
 37     var obs = this.getObserver();
 38     if(!obs) {
 39         return false;
 40     }
 41     
 42     // create dmlist
 43     args = FM.isset(args) && args && FM.isObject(args) ? args : {};
 44     callback = FM.isset(callback) && callback && FM.isFunction(callback) ? callback: function() {};
 45     var dmconfName = obs.getAttr('data-fmml-list','');
 46     if(this.dmList) {
 47         this.dmList.dispose();
 48     }
 49     this.dmList = new FM.DmList(args,dmconfName,obs.getApp());
 50     this.lastLovArguments = args;
 51     
 52     // prepare listener
 53     var me = this;
 54     var lstnr = {
 55         /** @ignore */
 56         onListStart: function(sender,data) {
 57             me.log("Ajax call started",FM.logLevels.info,'MlListOfValues.onListStart');
 58             $(me.getNode()).addClass("fmmlWaitState");
 59             return true;
 60         },
 61         /** @ignore */
 62         onListEnd: function(sender,data) {
 63             me.log("Ajax call completed",FM.logLevels.info,'MlListOfValues.onListEnd');
 64             $(me.getNode()).removeClass("fmmlWaitState");
 65             me.dmList.removeListener(lstnr);    
 66             callback(true,me.dmList);
 67             return true;
 68         },
 69         /** @ignore */
 70         onListError: function(sender,data) {
 71             me.log(data,FM.logLevels.error,'MlListOfValues.onListError');
 72             $(me.getNode()).removeClass("fmmlWaitState");
 73             me.dmList.removeListener(lstnr);    
 74             callback(false,null);
 75             return true;
 76         }
 77     };
 78     
 79     me.dmList.addListener(lstnr);
 80     me.dmList.getData();
 81 
 82     return true; 
 83 }

 86 FM.MlListOfValues.prototype.lovRenderItem = function(obj,issel,menuId,menuAttr,menuTempl,menuTemplName) {
 87     var itmText = '';
 88     var me = this;
 89     if(menuTemplName != '') {
 90         if(me.lovItemsTemplate == null) {
 91             FM.UtTemplate.getTemplate(this.getApp(),obj.getAttr(),menuTemplName,function(isok,templ) {            
 92                 if(isok) {
 93                     me.lovItemsTemplate = templ;
 94                 } else {
 95                     me.lovItemsTemplate = '';
 96                 }
 97                 var attrs = obj.getAttr();
 98                 attrs['lovSelected'] = issel ? 'selected' : '';
 99                 attrs['lovValue'] = obj.getAttr(menuId,'');
100                 attrs['lovDataID'] = obj.getDataID();
101                 itmText = $(FM.applyTemplate(attrs, me.lovItemsTemplate, false, false));
102                 $(itmText).attr("data-fmml-object-id",obj.getDataID());
103                 $(me.getNode()).append(itmText);  
104             });   
105             return;
106         } else {
107             var attrs = obj.getAttr();
108             attrs['lovSelected'] = issel ? 'selected' : '';
109             attrs['lovValue'] = obj.getAttr(menuId,'');
110             attrs['lovDataID'] = obj.getDataID();
111             itmText = FM.applyTemplate(attrs, this.lovItemsTemplate, false, false);
112         }
113     
114     } else if(menuTempl != '') {
115         itmText = 
116             '<option ' + 
117             (issel ? 'selected="selected"' : '') + 
118             ' value="' + obj.getAttr(menuId,'') + '">' +
119             FM.applyTemplate(obj.getAttr(), menuTempl, false, false) +
120             '</option>'
121         ;
122     } else {
123         itmText = 
124             '<option ' + 
125             (issel ? 'selected="selected"' : '') + 
126             ' value="' + obj.getAttr(menuId,'') + '">' +
127             obj.getAttr(menuAttr,'') +
128             '</option>'
129         ;
130     }
131 
132     itmText = $(itmText);
133     $(itmText).attr("data-fmml-object-id",obj.getDataID());
134     $(this.getNode()).append(itmText);        
135 }

137 FM.MlListOfValues.prototype.lovDisplay = function(isok, dmList) {    
138     // clear list
139     $(this.getNode()).empty();
140     
141     // get observer & dmobject & obs attribute name
142     var me = this;
143     var obs = this.getObserver();
144     var dmobj = this.getDmObject();
145     var attr = obs.getAttr('data-fmml-attr-name','');
146     
147     // LOV def data-fmml-list-id-attr > data-fmml-attr-name ref!
148     var menuId = obs.getAttr('data-fmml-list-id-attr','id');
149     var menuAttr = obs.getAttr('data-fmml-list-text-attr','');        
150     var menuTempl = obs.getAttr('data-fmml-list-text-template','');        
151     var menuTemplName = obs.getAttr('data-fmml-list-text-template-name','');        
152     var defSelValue = obs.getAttr('data-fmml-list-def-selected','');
153     if(FM.isFunction(window[defSelValue])) defSelValue = window[defSelValue](); // ref!
154     var defSelAttr = obs.getAttr('data-fmml-list-def-selected-attr','');
155     var allowNulls = obs.getAttr('data-fmml-list-allow-null','false');
156 
157     // state
158     var def=null,first = null,cur=null;    
159     var curVal = dmobj ? dmobj.getAttr(attr,'') : '';
160     
161     // ok, display list
162     if(isok) {
163         // null
164         if(allowNulls == 'true') {
165             $(thisgetNode()).append('<option value=""></option>');
166         }
167 
168         // iterate list and find sel value
169         dmList.forEachListElement(function(index,obj) {
170             // def
171             if(defSelAttr != '' && defSelValue!='') {
172                 if(obj.getAttr(defSelAttr,'') == defSelValue) {
173                     def = obj;
174                 }
175             }
176             if(!first) {
177                 first = obj;
178             }
179             if(obj.getAttr(menuId,'') == curVal) {
180                 cur = obj;
181             }
182             return(true);
183         });
184         
185         // find selected (current > def > first > none
186         var lovSelected = cur ? cur : (def ? def : first);
187 
188         // fill list
189         dmList.forEachListElement(function(index,obj) {
190             me.lovRenderItem(obj,obj == lovSelected,menuId,menuAttr,menuTempl,menuTemplName);
191             return(true);
192         });
193         
194         // start listener
195         $(me.getNode()).change(function() {
196             // attribute to set
197             var dmobj = obs.getDmObject();
198             var value = $(me.getNode()).val();
199             if(FM.isset(attr) && attr && attr != '' && dmobj && dmobj.getAttr(attr) !== value) {
200                 dmobj.setAttr(attr,value,true);
201             }
202             
203             // host to run
204             var hostToRun = me.getAttr("data-fmml-run-on-lov-change",'');
205             if(hostToRun != '') {
206                 var node = document.getElementById(hostToRun);
207                 if( node) {
208                     var optNode = $(me.getNode()).find(":selected");
209                     optNode = optNode.length > 0 ? optNode[0] : null;
210                     if(optNode) {
211                         var lovObj = dmList.get($(optNode).attr("data-fmml-object-id"));
212                         me._runHost(node,lovObj,true);
213                     }
214                 }
215             }
216         });
217         $(me.getNode()).change();
218         
219         // check if dm value and lov value are the same                
220         if(lovSelected) {
221             if(dmobj && dmobj.getAttr(attr,'') != lovSelected.getAttr(menuId,'')) {
222                 dmobj.setAttr(attr,lovSelected.getAttr(menuId,''),true);            
223             }
224         }
225            
226     } else { // error
227         $(me.getNode()).append('<option value=""><i>Error occured</i></option>');        
228     }    
229 }

231 FM.MlListOfValues.prototype._runHost = function(node,oObj,doinit) {    
232     if(!FM.isset(node) || !node) return false;
233     doinit = doinit == true;
234     
235     if(!FM.isset(node.fmmlHost) || !node.fmmlHost) {
236         if(!doinit || !$(node).attr('data-fmml-host')) return false;
237         FM.MlHost.initChildNodes(this.getApp(),node,oObj,false);
238         if(!FM.isset(node.fmmlHost) || !node.fmmlHost) return false;
239     }
240     
241     node.fmmlHost.run(oObj);
242     return true;
243 }

245 FM.MlListOfValues.prototype.lovGetFetchArguments = function() {
246     var args = {};
247     var obs = this.getObserver();    
248     var dmobj = obs.getDmObject();
249     if(dmobj) for(var i = 0; i < this.lovDependsOf.length; i++) {
250         args[this.lovDependsOf[i]] = dmobj.getAttr(this.lovDependsOf[i],'');
251     }    
252     return args;    
253 }

256 FM.MlListOfValues.prototype.lovCheckDepedOfChanged = function() {    
257     var obs = this.getObserver();    
258     var dmobj = obs.getDmObject();
259 
260     // menu is not filled yet
261     if(!dmobj || this.lastLovArguments == null) return true;
262     
263     
264     // check attributes
265     var depchanged = false;
266     for(var i = 0; i < this.lovDependsOf.length; i++) {
267         if(
268             FM.getAttr(this.lastLovArguments,this.lovDependsOf[i],'') 
269             != 
270             dmobj.getAttr(this.lovDependsOf[i],'')
271             ) {
272             depchanged = true;
273             break;
274         }
275     }
276     
277     // end
278     return(depchanged);
279 }

282 FM.MlListOfValues.prototype.run = function(obs) {
283     this._super("run",obs);
284     
285     // if dependsof attrs are not jet collected 
286     var dependsOfStr = obs.getAttr('data-fmml-list-depends-of','');
287     this.lovDependsOf = dependsOfStr == '' ? [] : dependsOfStr.split(',');
288 
289     // create LOV
290     var me = this;
291     this.lovFetch(this.lovGetFetchArguments(), function(isok, list) {
292         me.lovDisplay(isok, list);
293     });
294 }

297 FM.MlListOfValues.prototype.dispose = function(obs) {
298     if(this.getNode()) $(this.getNode()).unbind();
299     this._super("dispose");
300 }

302 FM.MlListOfValues.prototype.update = function(obs) {
303     this._super("update",obs);
304 
305     // if dep value are not changed
306     if(!this.lovCheckDepedOfChanged()) {
307         return true;
308     }
309     
310     var me = this;
311     this.lovFetch(this.lovGetFetchArguments(), function(isok, list) {
312         me.lovDisplay(isok, list);
313     });
314 }
