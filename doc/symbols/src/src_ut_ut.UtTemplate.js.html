1 /**
  2 * Templates factory class. 
  3 * 
  4 * @class FM.UtTemplate
  5 * @extends FM.Object
  6 * @param {object} opt Options
  7 */    
  8 FM.UtTemplate = FM.defineClass('UtTemplate',FM.Object);
  9 FM.UtTemplate.loadedTemplates = {};
 10 
 11 FM.UtTemplate.prototype._init = function(attrs) {
 12     this._super("_init",attrs);
 13     this.objectSubClass = "Template";
 14 }
 15 
 16 FM.UtTemplate.getTemplateArgs = function(attrlist) {
 17     var params = {};
 18     FM.forEach(attrlist, function(i,v) {
 19         if(FM.startsWith(i, "data-fmml-template-attr-")) {
 20             params[i.substring(24)] = v;
 21         }
 22         return true;
 23     });
 24     
 25     return params;
 26 }
 27 
 28 FM.UtTemplate.getTemplate = function(app,attrlist,tname,cbfn) {
 29     var callbackFn = FM.isset(cbfn) && FM.isFunction(cbfn) ? cbfn : function() {};
 30     attrlist = FM.isset(attrlist) && attrlist && FM.isObject(attrlist) ? attrlist : {};
 31     var params = FM.UtTemplate.getTemplateArgs(attrlist);
 32 
 33     FM.UtTemplate.fetchTemplate(app,tname,function(isok,templ) {
 34         if(isok) {
 35             callbackFn(true,FM.applyTemplate(params,templ,false,false));
 36         } else {
 37             callbackFn(false,templ);
 38         }
 39     });
 40 }    
 41     
 42 FM.UtTemplate.getLoadedTemplate = function(app,name) {
 43     var list = FM.UtTemplate.loadedTemplates;
 44 
 45     app = FM.isset(app) && app ? app : null;
 46     var appCls = app ? app.getSubClassName() : null;
 47     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
 48         
 49     var obj = null;
 50     if(appCfg && FM.isset(appCfg[name])) {
 51         obj = appCfg[name];
 52     } else if(app && FM.isArray(app.applicationObjectsSpace)) {
 53         FM.forEach(app.applicationObjectsSpace,function(i,ns) {
 54             if(FM.isset(list[ns]) && FM.isset(list[ns][name])) {
 55                 obj = list[ns][name];
 56                 return false;
 57             }
 58             return true;
 59         });
 60     }
 61     
 62     if(!obj && FM.isset(list['GLOBAL'][name])) {
 63         obj = list['GLOBAL'][name];
 64     }
 65     
 66     return obj;
 67 }

 69 FM.UtTemplate.addTemplate = function(tname,tdata,appCls) {
 70     appCls = FM.isset(appCls) && FM.isString(appCls) && appCls != '' ? appCls : 'GLOBAL';
 71     if(!FM.isset(tdata) || !tdata) return false;
 72     if(!FM.isset(tname) || !tname || tname == '') return false;
 73     if(!FM.isset(FM.UtTemplate.loadedTemplates[appCls])) {
 74         FM.UtTemplate.loadedTemplates[appCls]= {};
 75     }
 76     FM.UtTemplate.loadedTemplates[appCls][tname] = tdata;
 77     return true;
 78 }
 79 
 80 FM.UtTemplate.fetchTemplate = function(app,tname,cbfn) {
 81     var callbackFn = FM.isset(cbfn) && FM.isFunction(cbfn) ? cbfn : function() {};
 82     var templData = FM.UtTemplate.getLoadedTemplate(app,tname);
 83     if(templData) {
 84         callbackFn(true,templData);
 85         return;
 86     }
 87     
 88     // check in DOM
 89     var templNode = $("[data-fmml-template-def='" + tname + "']");
 90     if(templNode.length > 0) {
 91         callbackFn(true,$(templNode[0]).html());
 92         return;
 93     }
 94     
 95     var res = tname.replace(/\./g,'/');    
 96     res = '/' + res.replace('/html','.html');    
 97     var dmlist = new FM.DmList({
 98         fm_templates_path: app.getAttr(
 99             "fm_templates_path",
100             FM.getAttr(
101                 FM,"templates_path",
102                 FM.getAttr(
103                     window,"FM_TEMPLATES_PATH",
104                     "/resources/templates"
105                 )
106             )
107         )
108     },'getTemplate',app); 
109     var lurl = dmlist.getProperty('config.url','');
110     lurl += res;
111     dmlist.setProperty('config.url', lurl);
112     FM.log(app,"Loading template " + lurl + " ...",FM.logLevels.debug);
113     
114     var lstnr = {
115         onListEnd: function(sender,data) {
116             var oData = null;
117             FM.forEach(data.Added,function(id, obj) {
118                 oData = obj;
119                 return false;
120             });
121             dmlist.removeListener(lstnr);
122             dmlist.dispose();
123             if(oData) {
124                 FM.UtTemplate.addTemplate(tname,oData.getAttr("value",""),app.getSubClassName());
125                 callbackFn(true,oData.getAttr("value"));
126             } else {
127                 callbackFn(false,new FM.DmGenericError({
128                     messageId: -1,
129                     text: 'Data error: invalid response.'
130                 }));
131             }
132             return true;
133         },
134         onListError: function(sender,data) {
135             dmlist.removeListener(lstnr);
136             dmlist.dispose();
137             callbackFn(false,new FM.DmGenericError({
138                 messageId: -1,
139                 text: data
140             }));
141             return true;
142         }
143     };
144     dmlist.addListener(lstnr);
145     dmlist.getData();    
146 }
