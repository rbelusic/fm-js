 44 FM.MlHost.prototype.run = function(dmObj) {
 45     if(this.getAttr('data-fmml-run-maximized','false') == 'true') {
 46         this.onMaximize();
 47     }
 48 
 49     var id = this.getAttr("data-fmml-object-id",'');
 50     var className = this.getAttr("data-fmml-object-class",'');
 51     var objRef = this.getAttr("data-fmml-object-ref",'');
 52     this.setProperty('dmObjectCreated',this.getAttr("data-fmml-object-destroy-on-dispose",'true'));
 53     
 54     // object
 55     if(FM.isset(dmObj) && dmObj && (className == '' || dmObj.getSubClassName() == className)){
 56         this.setProperty('dmObjectCreated','false');
 57         this.setDmObject(dmObj);
 58     } else if(objRef != '' && FM.startsWith(objRef,'@')) {
 59         dmObj = FM.resolveAttrValue(null,"-",objRef,{
 60             A: this.getApp(),
 61             H: this
 62         });
 63         this.setProperty('dmObjectCreated','false');
 64         this.setDmObject(dmObj);
 65     } else {
 66         if(this.getAttr('data-fmml-master-host','') != '') {
 67             var mhostid = this.getAttr('data-fmml-master-host','');
 68             if(mhostid == 'true') {
 69                 this.masterHost = FM.findNodeWithAttr(this.getNode().parentNode, "fmmlHost");
 70             } else {            
 71                 var mhostnode = document.getElementById(mhostid);
 72                 if(mhostnode && FM.isset(mhostnode.fmmlHost) && mhostnode.fmmlHost) {
 73                     this.masterHost = mhostnode.fmmlHost;
 74                 }
 75             }
 76             if(this.masterHost) {
 77                 this.masterHost.addListener(this);
 78                 this.masterHostDm = this.masterHost.getDmObject();
 79                 if(this.masterHostDm) this.masterHostDm.addListener(this);
 80             }
 81         }
 82         
 83         this._checkMasterReload();        
 84     }
 85 
 86     this._super("run");     
 87     this.executed = true;
 88     
 89     var obsrv = this.listOfObservers;
 90     for(var id in obsrv) {
 91         try {
 92             obsrv[id].run();            
 93         } catch(e) {
 94             this.log(e,FM.logLevels.error,'MlHost.run');
 95         }
 96     }
 97 }

100 FM.MlHost.prototype._checkMasterReload = function() {
101     var id = this.getAttr("data-fmml-object-id",'');
102     var className = this.getAttr("data-fmml-object-class",'');
103     
104     if(this.getAttr('data-fmml-use-global-args') == 'true'){
105         var args = FM.getArgs();
106         if(id == '') id = FM.getAttr(args,'id','');
107         if(className == '') className = FM.getAttr(args,'cls','');        
108     }
109     
110     //FM.resolveAttrValue = function(options,attrName,def,context)
111     if(this.masterHost) {
112         id = FM.resolveAttrValue(null,"-",id,{
113             A: this.masterHost.getApp(),
114             H: this.masterHost, 
115             D: this.masterHostDm
116         });
117         className = FM.resolveAttrValue(null,"-",className,{
118             A: this.masterHost.getApp(),
119             H: this.masterHost, 
120             D: this.masterHostDm
121         });
122     } else {
123         id = FM.resolveAttrValue(null,"-",id,{
124             A: this.getApp(),
125             H: this, 
126             D: this.getDmObject()
127         });
128         className = FM.resolveAttrValue(null,"-",className,{
129             A: this.getApp(),
130             H: this, 
131             D: this.getDmObject()
132         });
133     }
134 
135     if(id == this.clsParams.id && className == this.clsParams.className) return;
136     
137     this.clsParams = {
138         id: id,
139         className: className
140     }
141     if(id != '') {
142         var fnName = 'get' + className;
143         if(id != '' && className != '' && FM.isset(this.app[fnName])) {
144             var me = this;
145             
146             this.app[fnName](id,function(isok,oObj) {
147                 if(isok) {
148                     me.setDmObject(oObj);
149                     me.setProperty('dmObjectCreated','true');
150                 }
151                 else {
152                     me.setDmObject(null);
153                 }
154             });
155         }
156     } else {
157         if(className != '') {
158             var attrs = {};
159             this.forEachAttr(function(n,v) {
160                 if(FM.startsWith(n,'data-fmml-object-attr-')) {
161                     attrs[n.substr(22)] = v;
162                 }
163                 return true;
164             });
165             var oObj = FM.DmObject.newObject(this.getApp(),className, attrs);
166             this.setProperty('dmObjectCreated','true');
167             this.setDmObject(oObj);
168         }
169     }
170 }

172 FM.MlHost.prototype.dispose = function() {
173     if(this.masterHost) {
174         this.masterHost.removeListener(this);
175         if(this.masterHostDm) this.masterHostDm.removeListener(this);
176     }
177     
178     this.app.removeListener(this);
179     if(this.node) this.node.fmmlHost = null;
180     var obsrv = this.listOfObservers;
181     for(var id in obsrv) {
182         try {
183             obsrv[id].dispose();
184             
185         } catch(e) {
186             this.log(e,FM.logLevels.error,'MlHost.dispose');
187         }
188     }
189     this.listOfObservers = [];
190     this.setDmObject();
191     this.executed = false;
192     
193     this._super("dispose");
194     return true;
195 }

225 FM.MlHost.prototype.addObserver = function(o) {
226     if(!FM.isset(o)  || !o || !FM.isset(o.getID)) return false;
227     this.listOfObservers[o.getID()] = o;
228     if(this.executed) o.run();
229     this.updateObserver(o);
230     return true;
231 }

233 FM.MlHost.prototype.removeObserver = function(o) {
234     if(!FM.isset(o)  || !o || !FM.isset(o.getID)) return false;
235 
236     var nlist = {};
237     var objId = o.getID();
238     if(!objId) return false;
239 
240     for(var id in this.listOfObservers) {
241         if(objId != id) nlist[id] = this.listOfObservers[id];
242     }
243 
244     this.listOfObservers = nlist;
245 
246     return true;
247 }

317 FM.MlHost.prototype.setLastError = function(oErr) {
318     var errhost = this._getErrorHost();
319     if(!errhost) {
320         return this.getApp() ? this.getApp().setLastError(oErr) : oErr;
321     }
322     
323     
324     if(!FM.isset(oErr) || !oErr || !FM.isObject(oErr)) {        
325         oErr = new FM.DmGenericError();
326     }
327     
328     if(!FM.isset(oErr) || !oErr || !FM.isObject(oErr)) {
329         oErr = new FM.DmGenericError();
330     }
331     
332     if(!errhost.isExecuted()) {
333         errhost.run(oErr);
334     } else {
335         var dmobj = errhost.getDmObject();
336         if(!dmobj) {
337             errhost.setDmObject(oErr);
338         } else {
339             dmobj.forEachAttr(function(attr,value) {
340                 dmobj.setAttr(attr,oErr.getAttr(attr,null));
341                 return true;
342             });   
343             dmobj.setChanged(true,true);
344             oErr = dmobj;
345         }
346     }    
347     
348     return oErr;
349 }

352 FM.MlHost.prototype.onChange = function(sender,obj) {
353     if(sender == this.masterHostDm) {
354         this._checkMasterReload();
355     } else if(sender == this.getDmObject()) {
356         this.updateAllObservers();
357     }
358     
359     var hostToRun =  this.getAttr('data-fmml-run-on-update','');
360     if(hostToRun != '') {

362         if( node && FM.isset(node.fmmlHost) && node.fmmlHost) {
363             node.fmmlHost.run(this.getDmObject());
364         }
365     }
366 
367     // kraj
368     return true;
369 }

391 FM.MlHost.prototype.onEvent = function(sender,ev,data,calledlist) {
392     var cl = FM.isset(calledlist) ? calledlist : {};
393 
394     if(!this.isEnabled()) return false;
395     
396     var done = false;
397     
398     //  ako ima event fn
399     if(FM.isset(this[ev])) {
400         this[ev](sender,data);
401         cl[this.getID()] = '1';
402         FM.setAttr(cl,'_executed','1'); 
403         done = true;
404     }
405 
406     //  ako ima event fn u app
407     if(!done && FM.isset(this.app[ev])) {
408         this.app[ev](sender,data);
409         cl[this.app.getID()] = '1';
410         FM.setAttr(cl,'_executed','1'); 
411         done = true;
412     }
413     if(!done) {
414         if(!this.sendEventToObservers(sender,ev, data)) {
415         // proslijedi dalje ako nemas ev fn
416         //cl = this.fireEvent(ev,data,cl);        
417         }
418         done = true;
419     }
420     
421     // ako ima def trigger u layoutu
422     var evtrg = this.getAttr('data-fmml-host-event-' + ev.toLowerCase(),'');
423     if(evtrg !== '') {
424         FM.resolveAttrValue(null,"-",evtrg,{
425             A: this.getApp(),
426             H: this, 
427             D: this.getDmObject()
428         });
429     }
430     
431     return cl;
432 }

437 /**
438  * Returns MlHost <b>config</b> class function for <b>config</b> subclass type
439  * @static
440  * @function    
441  * @param {object} app Application
442  * @param {String} name Configuration name
443  * @return {object} host configuration or null if not found
444  */   
445 FM.MlHost.getConfiguration = function(app,name) {
446     var list = FM.MlHost.hostTypes;
447 
448     app = FM.isset(app) && app ? app : null;
449     var appCls = app ? app.getSubClassName() : null;
450     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
451         
452     var obj = null;
453     if(appCfg && FM.isset(appCfg[name])) {
454         obj = appCfg[name];
455     } else if(app && FM.isArray(app.applicationObjectsSpace)) {
456         FM.forEach(app.applicationObjectsSpace,function(i,ns) {
457             if(FM.isset(list[ns]) && FM.isset(list[ns][name])) {
458                 obj = list[ns][name];
459                 return false;
460             }
461             return true;
462         });
463     }
464     
465     if(!obj && FM.isset(list['GLOBAL'][name])) {
466         obj = list['GLOBAL'][name];
467     }
468     
469     return obj;
470 }

472 FM.MlHost.newHost = function(app,attrs,node,type,oObj) {
473     var clsFn = FM.MlHost.getConfiguration(app,type);
474     var obj = clsFn ? new clsFn(app,attrs,node) : null;
475     if(obj && obj.getAttr('data-fmml-run-on-init','true') != 'false') {
476         obj.run(oObj); 
477     }
478     return obj;

482     appCls = FM.isset(appCls) && FM.isString(appCls) && appCls != '' ? appCls : 'GLOBAL';

488     FM.MlHost.hostTypes[appCls][type] = hostfn;
489     return true;

492 FM.MlHost.translateNode = function(app,node) {
493     // what to translate
494     var txtattrs = $(node).attr('data-fmml-translate');
495     var attrs = txtattrs.split(",");
496     var txttotranslate;
497     FM.forEach(attrs, function(i,name) {
498         name = FM.trim(name);
499         if(name == 'body') {
500             txttotranslate = $(node).text();

502                 $(node).text(_T(txttotranslate,app));
503             }
504         } else {
505             txttotranslate = $(node).attr(i); 
506             if(FM.isset(txttotranslate)) {
507                 $(node).attr(name,_T(txttotranslate,app));
508             }
509         }
510         return true;
511     });

514 // radi update child nodova - promjena na dom nodu koji je nosioc
515 FM.MlHost.initChildNodes = function(app,checknode,oObj,childsOnly) {
516     childsOnly = FM.isset(childsOnly) && childsOnly == false ? false : true;
517     checknode = FM.isset(checknode) && checknode ? checknode : $('body')[0];
518     var appsc = app.getSubClassName();
519     var nodeList = childsOnly ? $(checknode).children() : $(checknode);    
520     
521     nodeList.each(function(index) {
522         var domobj = this;        
523         var jqobj = $(this);
524         if(!jqobj) return;
525         if(
526             (!FM.isset(domobj.fmmlHost)) &&
527             (!FM.isset(domobj.fmmlObserver)) &&                
528             (
529                 jqobj.attr('data-fmml-host') || 
530                 jqobj.attr('data-fmml-observer') || 
531                 jqobj.attr('data-fmml-extensions') ||
532                 jqobj.attr('data-fmml-template') ||
533                 jqobj.attr('data-fmml-translate')
534                 )
535             ) {
536             var mlAppCls = jqobj.attr('data-fmml-app');
537             if(!mlAppCls && !app.strictApplicationObjectsSpace) {
538                 mlAppCls = app.getSubClassName();
539             }
540             if(mlAppCls == appsc) {  
541                 // pokupi atribute 
542                 var attrlist = {};
543                 $.each(this.attributes, function(i, attrib){
544                     if(FM.isString(attrib.value) && FM.startsWith(attrib.value,'@@')) { 
545                         try {
546                             attrlist[attrib.name] = FM.resolveAttrValue(null,"-",attrib.value,{
547                                 A: app,
548                                 D: oObj
549                             });
550                         } catch(e) {
551                             FM.log(null,e,FM.logLevels.error,'FM.MlHost.initChildNodes');
552                         }
553                     } else {
554                         attrlist[attrib.name] = attrib.value;
555                     }
556                 });
557                 
558                 
559                 // ako je observer
560                 if(jqobj.attr('data-fmml-observer')) { 
561                     try {
562                         obs = FM.MlObserver.newObserver(app,attrlist,domobj,jqobj.attr('data-fmml-observer'));
563                         if(obs) {
564                             if(obs.getHost()) {
565                                 obs.getHost().addObserver(obs);
566                             }
567                         }
568                     } catch(e) {
569                         FM.log(null,e,FM.logLevels.error,'FM.MlHost.initChildNodes');
570                     };
571                 }
572                 
573                 // extenzije
574                 if(jqobj.attr('data-fmml-extensions')) { // ako je observer
575                     var extarr = jqobj.attr('data-fmml-extensions').split(" ");
576                     for(var i = 0; i < extarr.length; i++) {
577                         var otype = extarr[i].toString();
578                         try {
579                             var oExt = FM.MlExtension.newExtension(app,attrlist,domobj,otype);
580                             if(oExt && domobj.fmmlObserver) domobj.fmmlObserver.addExtension(oExt);
581                         } catch(e) {
582                             FM.log(null,e,FM.logLevels.error,'FM.MlHost.initChildNodes');
583                         };
584                     }
585                 }
586 
587                 // ako je host
588                 if(jqobj.attr('data-fmml-host')) {
589                     try {
590                         FM.MlHost.newHost(app,attrlist,domobj,jqobj.attr('data-fmml-host'),oObj);
591                     } catch(e) {
592                         FM.log(null,e,FM.logLevels.error,'FM.MlHost.initChildNodes');
593                     }                  
594                 } 
595 
596                 // ako je translation
597                 if(jqobj.attr('data-fmml-translate')) {
598                     try {
599                         FM.MlHost.translateNode(app,domobj);
600                     } catch(e) {
601                         FM.log(null,e,FM.logLevels.error,'FM.MlHost.initChildNodes');
602                     }                  
603                 } 
604 
605                 // ako je template                
606                 if(jqobj.attr('data-fmml-template')) { // begin
607                     try {
608                         FM.MlTemplate.newTemplate(app,attrlist,domobj,oObj);
609                     } catch(e) {
610                         FM.log(null,e,FM.logLevels.error,'FM.MlHost.initChildNodes');
611                     }                  
612                 }
613                 // end
614             }            
615         }
616         
617         // napravi isto na child nodovima
618         FM.MlHost.initChildNodes(app,this,oObj);
619     });
620     
621     return true;
622 }

625 FM.MlHost.disposeChildNodes = function(checknode,childsOnly) {
626     if(FM.isString(checknode)) {
627         checknode = $('#' + checknode);
628         checknode = checknode.length > 0 ? checknode[0] : null;
629     }    
630     childsOnly = FM.isset(childsOnly) && childsOnly == false ? false : true;
631     var nodes = $(checknode).find("[data-fmml-host],[data-fmml-template]");
632     if(!childsOnly && $(checknode).is("[data-fmml-host],[data-fmml-template]")) {
633         nodes = nodes.add(checknode);
634     }
635     
636     nodes.each(function(i, n){
637         if(FM.isset(n.fmmlTemplate) && n.fmmlTemplate) {
638             n.fmmlTemplate.dispose();
639         }        
640         if(FM.isset(n.fmmlHost) && n.fmmlHost) {
641             n.fmmlHost.dispose();
642         }        
643     });
644 }
