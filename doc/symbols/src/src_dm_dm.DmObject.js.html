1 /**
  2 * Basic DM class. Provide groups
  3 * 
  4 * @class FM.DmObject
  5 * @extends FM.Object
  6 * @memberOf FM
  7 * @param {object} attrs list of attribute name and values
  8 * @param {object} [flds] allowed attributes
  9 */    
 10 FM.DmObject = FM.defineClass('DmObject',FM.Object);
 11 
 12 
 13 // methods
 14 FM.DmObject.prototype._init = function(attrs,flds) {            
 15     this.objectGroups = {};
 16     this.defaultGroup = '';               
 17 
 18     this._super("_init",attrs,flds);
 19     this.objectSubClass = "Object";
 20 }
 21 
 22 FM.DmObject.prototype.getDataID = function() {
 23     return this.getID();
 24 }
 25 
 26 // groups
 27 FM.DmObject.prototype.addGroup = function(s,callevent) {
 28     if(!FM.isset(s) || !s || s == '') return false;
 29 
 30     this.objectGroups[s] = {
 31         name: s
 32     }
 33 
 34     if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
 35     return true;
 36 }
 37 
 38 FM.DmObject.prototype.removeGroup = function(s,callevent) {
 39     if(!FM.isset(s) || !s || !FM.isset(this.objectGroups[s])) return false;
 40     var newglist = {};            
 41     FM.forEachAttr(this.objectGroups,function(name,value) {
 42         if(name != s) {
 43             newglist[name] = value;
 44         }
 45         return true;
 46     });
 47 
 48     this.objectGroups = newglist;
 49     if(this.defaultGroup == s) this.defaultGroup = '';
 50     if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
 51     return true;
 52 }
 53 
 54 FM.DmObject.prototype.isInGroup = function(s) {
 55     return (
 56         FM.isset(s) && s && FM.isset(this.objectGroups[s]) ?
 57         true : false
 58         );
 59 }
 60 
 61 FM.DmObject.prototype.removeAllGroups = function(callevent) {
 62     this.objectGroups = {};
 63     this.defaultGroup = '';
 64     if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
 65     return true;
 66 }
 67 
 68 
 69 FM.DmObject.prototype.getGroups = function() {
 70     return FM.cloneObject(this.objectGroups);
 71 }
 72 
 73 FM.DmObject.prototype.getGroupsCount = function() {
 74     return FM.sizeOf(this.objectGroups);
 75 }
 76 
 77 FM.DmObject.prototype.forEachGroup = function(doFn) {
 78     return FM.forEach(this.objectGroups,doFn);
 79 }
 80 
 81 FM.DmObject.prototype.setDefaultGroup = function(s,callevent) {
 82     s = FM.isset(s) && s ? s : '';
 83     if(s == '' || FM.isset(this.objectGroups[s])) {
 84         this.defaultGroup = s;
 85         if(FM.isset(callevent) && callevent == true) this.fireEvent("onChange", this);
 86         return true;
 87     }            
 88     return false;
 89 }
 90 
 91 FM.DmObject.prototype.getDefaultGroup = function() {
 92     if(this.defaultGroup != '') {
 93         return FM.getAttr(this.objectGroups,this.defaultGroup,null);
 94     }
 95 
 96     // ako nema def ili nije vidljiv
 97     var defgrp = null;
 98     FM.forEach(this.objectGroups,function(name,value) {
 99         // prvi u listi
100         defgrp = value;
101         return false;
102     }); 
103     return defgrp;
104 }

106 FM.DmObject.subClassTypes = {
107     GLOBAL: {}

110 /**
111 * Returns DmObject <b>config</b> class function for <b>config</b> subclass type
112 * @static
113 * @function    
114 * @param {object} app Application
115 * @param {String} name Configuration name
116 * @return {object} object configuration or null if not found
117 */   
118 FM.DmObject.getConfiguration = function(app,name) {
119     var list = FM.DmObject.subClassTypes;
120 
121     app = FM.isset(app) && app ? app : null;
122     var appCls = app ? app.getSubClassName() : null;
123     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
124         
125     var obj = null;
126     if(appCfg && FM.isset(appCfg[name])) {
127         obj = appCfg[name];
128     } else if(app && FM.isArray(app.applicationObjectsSpace)) {
129         FM.forEach(app.applicationObjectsSpace,function(i,ns) {
130             if(FM.isset(list[ns]) && FM.isset(list[ns][name])) {
131                 obj = list[ns][name];
132                 return false;
133             }
134             return true;
135         });
136     }
137     
138     if(!obj && FM.isset(list['GLOBAL'][name])) {
139         obj = list['GLOBAL'][name];
140     }
141     
142     return obj;
143 }
144 
145 FM.DmObject.newObject = function(app,clsname, oAttrs) {    
146     var clsFn = FM.DmObject.getConfiguration(app,clsname);
147     return clsFn ? new clsFn(oAttrs) : null;
148 }
149 
150 FM.DmObject.getSubClassType = function(clsname,app) {
151     return clsname == "" ? 
152     FM.DmObject : (
153         FM.isset(FM.DmObject.subClassTypes[clsname]) ? 
154         FM.DmObject.subClassTypes[clsname] : 
155         null
156         );
157 }
158 
159 FM.DmObject.addSubClassType = function(subclsname, clsfn,appCls) {
160     appCls = FM.isset(appCls) && FM.isString(appCls) && appCls != '' ? appCls : 'GLOBAL';
161     if(!FM.isset(FM.DmObject.subClassTypes[appCls])) {
162         FM.DmObject.subClassTypes[appCls]= {};
163     }
164     
165     FM.DmObject.subClassTypes[appCls][subclsname] = clsfn;
166 }
