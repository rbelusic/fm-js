 57 /**
 58  * Determine if a variable is array.
 59  * 
 60  * @static
 61  * @function 
 62  * @param v The variable to be checked.
 63  * @returns {boolean} 
 64  */
 65 FM.isArray = function(v) {
 66     if (!FM.isset(v) || !v) {
 67         return false;
 68     }
 69     
 70     if (v.constructor.toString().indexOf("Array") == -1) {
 71         return false;
 72     }
 73 
 74     return FM.isset(v.length);
 75 }

 90 /**
 91  * Count all elements in an array or object.
 92  * 
 93  * @static
 94  * @function 
 95  * @param {Object|Array} o The array or object to be checked.
 96  * @returns {number} Number of elements in array or object.
 97  */
 98 FM.sizeOf = function(o) {
 99     if (!FM.isset(o) || o == null)
100         return(-1);
101     var i = 0;
102     for (var id in o) {
103         i++;
104     }
105     return(i);

111  * 
112  * @static
113  * @function 
114  * @param {string} s The string that will be trimmed.
115  * @returns {string}

206 /**
207  *  Returns a string with backslashes before characters that need to be quoted. 
208  *  These characters are single quote ('), double quote (") and backslash.
209  * 
210  * @static
211  * @function 
212  * @param {string} str The string to be escaped.
213  * @returns {string} 
214  */
215 FM.addslashes = function(str) {
216     str = str.replace(/\\/g, '\\\\');
217     str = str.replace(/\'/g, '\\\'');
218     str = str.replace(/\"/g, '\\"');
219     str = str.replace(/\0/g, '\\0');
220     return str;
221 }
222 
223 /**
224  *  Un-quotes a quoted string
225  * 
226  * @static
227  * @function 
228  * @param {string} s The input string.
229  * @returns {string} 
230  */
231 FM.stripslashes = function(str) {
232     str = str.replace(/\\'/g, '\'');
233     str = str.replace(/\\"/g, '"');
234     str = str.replace(/\\0/g, '\0');
235     str = str.replace(/\\\\/g, '\\');
236     return str;
237 }

239 /**
240  * 
241  * @ignore
242  */
243 FM.tokenize = function(argsstr) {
244     var i, instr;
245 
246     // napravi listu tokena
247     var elm_array = [];
248     var st_array = argsstr.split('"');
249 
250     instr = false;
251     for (i = 0; i < st_array.length; i++) {
252         // ako nisi u stringu
253         if (!instr) {
254             var e = st_array[i].split(/[\s,]+/);
255             for (var j = 0; j < e.length; j++) {
256                 if (e[j] != "")
257                     elm_array.push(e[j]);
258             }
259         } else {
260             elm_array.push(st_array[i]);
261         }
262 
263         instr = !instr;
264     }
265 
266     return elm_array;
267 }

270 /**
271  * Uppercase the first character of each word  in a input string except the first one.
272  * The definition of a word is any string of characters that is immediately 
273  * after a whitespace or minus sign.
274  * 
275  * @static
276  * @function 
277  * @param {string} input The input string.
278  * @returns {string} 
279  */
280 FM.camelCase = function(input) {
281     return input.replace(/-([a-z])/ig, function(all, letter) {
282         return letter.toUpperCase();
283     });
284 }
285 
286 /**
287  * Uppercase the first character of each word in a string.
288  * The definition of a word is any string of characters that is immediately 
289  * after a whitespace or minus sign.
290  * 
291  * @static
292  * @function 
293  * @param {string} input The input string.
294  * @returns {string} 
295  */
296 FM.ucwords = function(input) {
297     var words = input.split(/(\s|-)+/),
298         output = [];
299 
300     for (var i = 0, len = words.length; i < len; i += 1) {
301         output.push(words[i][0].toUpperCase() +
302             words[i].toLowerCase().substr(1));
303     }
304 
305     return output.join('');
306 }
