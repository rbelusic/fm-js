1 /**
  2 * Ajax class. 
  3 * 
  4 * @class FM.UtAjax
  5 * @extends FM.Object
  6 * @param {object} config Options
  7 * 
  8 * Valid options:
  9 *
 10 * cache: true/false
 11 * url: url,
 12 * method: POST/GET/...
 13 * contentType: application/x-www-form-urlencoded
 14 * auth {
 15 *  username: '',
 16 *  password:''
 17 * }
 18 * responseFormat: JSON/TEXT
 19 * params: { // _body: za for post body
 20 *  a: true, 
 21 *  b:true
 22 *  } 
 23 * headers:{}
 24  
 25 */    
 26 
 27 FM.UtAjax = FM.defineClass('UtAjax',FM.Object);
 28 FM.UtAjax.className = "UtAjax";
 29 
 30 // methods
 31 FM.UtAjax.prototype._init = function(config) {            
 32     this._super("_init",config);
 33     this.objectSubClass = "UtAjax";
 34 
 35     this.http = null;
 36     this.lastStatusCode = "0";
 37 }
 38 
 39 FM.UtAjax.prototype.send = function(args) {
 40     var url = this.getAttr('url','');
 41     var params = this.getAttr('params',{});
 42     var headers = this.getAttr('headers',{});
 43     
 44     var pline = "";
 45     
 46     if(FM.isObject(args)) {
 47         var val;
 48         for(var pname in args) {
 49             if(FM.isset(params[pname])) {
 50                 val = FM.getAttr(args,pname,'');
 51                 if(pname === "_body" && this.getAttr("method","POST") == 'POST') {
 52                     pline = val;
 53                     break;
 54                 } else {
 55                     pline = pline + (pline == "" ? "" : "&") + pname + "=" + encodeURIComponent(val);
 56                 }
 57             }
 58         }
 59     }
 60     
 61     this.http = this.executeCall(url, pline, headers);
 62 
 63     return true;
 64 }     
 65 
 66 FM.UtAjax.prototype.getLastStatus = function() {
 67     return this.lastStatusCode;
 68 }
 69 
 70 FM.UtAjax.prototype.executeCall = function(url, params, headers) {
 71     var ajaxOptions = {
 72         global: false,
 73         cache: this.getAttr("cache","false").toLowerCase() == 'true',
 74         accepts: {},
 75         beforeSend: function(jqXHR, settings) {
 76             return this.fireEvent("onAjaxStateStart",params);            
 77         },
 78         error: function(jqXHR,textStatus,errorThrown) {
 79             this.lastStatusCode = textStatus;
 80             return this.fireEvent("onAjaxStateError", errorThrown);
 81         },
 82         success: function(data,textStatus, jqXHR) {
 83             this.lastStatusCode = textStatus;
 84             return this.fireEvent(
 85                 "onAjaxStateEnd", data
 86             );
 87         },
 88         url: url,
 89         type: this.getAttr("method","POST").toLowerCase(),
 90         contentType: this.getAttr('contentType',"application/x-www-form-urlencoded"),
 91         context: this,
 92         data: params,
 93         headers: headers,
 94         username: this.getAttr('auth.username',''),
 95         password: this.getAttr('auth.password',''),
 96         dataType: this.getAttr('responseFormat','TEXT').toLowerCase() 
 97     };
 98     
 99     return $.ajax(ajaxOptions);
100 }
