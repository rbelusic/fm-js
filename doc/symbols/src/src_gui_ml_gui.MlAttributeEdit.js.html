1 /**
  2 * ML attribute edit extensions class. 
  3 * 
  4 * @class FM.MlAttributeEdit
  5 * @extends FM.MlExtension
  6  * @param {object} [attrs] DOM node attributes
  7  * @param {DOMnode} node DOM node
  8 */
  9 FM.MlAttributeEdit = FM.defineClass('MlAttributeEdit',FM.MlExtension);

 11 // methods
 12 FM.MlAttributeEdit.prototype._init = function(attrs,node) {
 13     this._super("_init",attrs,node);
 14     this.objectSubClass = "AttributeEdit";
 15     this.triggerEvent = null;
 16     this.editWidget = null;    
 17 }
 18 
 19 // methods
 20 FM.MlAttributeEdit.prototype._cleanup = function(obs) {
 21     if(FM.isset(this.timer) && this.timer) {
 22         window.clearInterval(this.timer);
 23         this.timer = null;
 24     }
 25     
 26     if(this.editWidget) {
 27         this.editWidget.unbind(this.triggerEvent);
 28         this.triggerEvent = null;
 29         $(this.node).unbind('click');
 30         $(obs.node).html("");
 31         this.editWidget = null;
 32     } else {
 33         if(this.triggerEvent) {
 34             $(this.node).unbind(this.triggerEvent);
 35             this.triggerEvent = null;
 36         }        
 37     }    
 38 }
 39 
 40 FM.MlAttributeEdit.prototype.run = function(obs) {
 41     this._cleanup(obs);
 42     this._super("run",obs);
 43     
 44     var domnode = this.node;
 45     this.triggerEvent = obs.getAttr('data-fmml-update-condition','blur');
 46     if(!FM.isset($(domnode)[this.triggerEvent])) {
 47         this.triggerEvent = 'blur';
 48     }
 49     this.editWidget = null;
 50     var me = this;
 51     if(domnode.nodeName == 'INPUT' || domnode.nodeName == 'TEXTAREA') {    
 52         $(domnode)[this.triggerEvent]/*.change | blur*/(function() {
 53             var value = 
 54                 $(this).is(':checkbox') ?
 55                 ($(this).is(":checked") ? 'true' : 'false'):
 56                 $(this).val()
 57             ;   
 58             var savevalue = value;
 59             var obs = domnode.fmmlObserver;
 60 
 61             $(domnode).html("");
 62             $(domnode).text(value);
 63 
 64             if(obs && obs.getDmObject()) {
 65                 var attr = obs.getAttr('data-fmml-attr-name','');
 66                 var vtype = obs.getAttr('data-fmml-attr-type','string');                   
 67                 var dmobj = obs.getDmObject();
 68 
 69                 if(vtype == 'date') {
 70                     var isutc = obs.getAttr('data-fmml-date-is-utc','true') != 'false';
 71                     savevalue = FM.dateToString(FM.parseLocalDateString(value),isutc);
 72                 }
 73 
 74                 if(FM.isset(attr) && attr && attr != '') {
 75                     dmobj.setAttr(attr,savevalue,true);
 76                 }
 77             }
 78             return true;
 79         }); 
 80         if(obs.getAttr('data-fmml-attr-value-from-dom','false') == 'true') {
 81             this.timer = window.setInterval(function() {
 82                 if(FM.isset(domnode.fmmlValueSync) && obs.getNodeValue() != domnode.fmmlValueSync) {
 83                     var attr = obs.getAttr('data-fmml-attr-name','');
 84                     var dmobj = obs.getDmObject();
 85                     dmobj.setAttr(attr,obs.getNodeValue(),true);
 86                 }
 87             },300);
 88         }
 89         
 90     } else if(domnode.nodeName == 'SELECT') {
 91         $(domnode)[this.triggerEvent]/*.change | blur*/(function() {
 92             var value = $(this).val();
 93             var savevalue = value;             
 94             var obs = domnode.fmmlObserver;
 95 
 96             if(obs && obs.getDmObject()) {
 97                 var attr = obs.getAttr('data-fmml-attr-name','');
 98                 var vtype = obs.getAttr('data-fmml-attr-type','string');                      
 99                 var dmobj = obs.getDmObject();
100                 if(vtype == 'date') {
101                     var isutc = obs.getAttr('data-fmml-date-is-utc','true') != 'false';
102                     savevalue = FM.dateToString(FM.parseLocalDateString(value),isutc);
103                 }
104                 if(FM.isset(attr) && attr && attr != '') {
105                     dmobj.setAttr(attr,savevalue,true);
106                 }
107             }
108             return true;
109         });
110     } else {
111         var edittype = obs.getAttr('data-fmml-attr-edittype','input');
112         
113         if (edittype == 'textarea') {
114             this.editWidget = $("<textarea class='fmmlValueInput'></textarea>");    
115         }
116         else {
117             this.editWidget = $("<input type='text' class='fmmlValueInput'>");
118         }
119         
120         
121         $(domnode).click(function() {
122             var value = $(this).val();
123             var savevalue = value;                      
124             var obs = domnode.fmmlObserver;
125             if(!obs || !obs.getDmObject()) {
126                 return true;
127             }
128             if($(this).children().length < 1) {
129                 var val = $(this).text();
130 
131                 if (me.editWidgetAdded != true) {
132                     me.editWidget.hide();
133                     $(this).after(me.editWidget);    
134                 }
135                 $(this).hide();
136                 
137                 if (edittype == 'textarea') {
138                     me.editWidget.text(val);
139                 }
140                 else {
141                     me.editWidget.val(val);
142                 }
143             
144                 if (me.editWidgetAdded != true) {
145                     me.inputAdded(obs);        
146                 }
147             };
148             me.editWidget.show();
149             me.editWidget.focus();
150             me.editWidgetAdded = true;
151             /*
152             var defevobsnodeid = me.getAttr('data-fmml-update-def-event-observer','');            
153             if(defevobsnodeid != '') me.editWidget.keypress(function(e) {                
154                 if(e.keyCode == 13) {
155                     me.editWidget[me.triggerEvent]();
156                     $('#' + defevobsnodeid).click();             
157                 }
158             });*/
159             
160             me.editWidget[me.triggerEvent]/*.change | blur*/(function() {
161                 var value = $(this).val();
162                 var savevalue = value;             
163 
164                 $(domnode).text(value);
165                 $(this).hide();
166                 $(domnode).show();
167                 var obs = domnode.fmmlObserver;
168 
169                 if(obs && obs.getDmObject()) {
170 
171                     var attr = obs.getAttr('data-fmml-attr-name','');
172                     var vtype = obs.getAttr('data-fmml-attr-type','string');                      
173                     var dmobj = obs.getDmObject();
174                     if(vtype == 'date') {
175                         var isutc = obs.getAttr('data-fmml-date-is-utc','true') != 'false';
176                         savevalue = FM.dateToString(FM.parseLocalDateString(value),isutc);
177                     }
178 
179                     if(FM.isset(attr) && attr && attr != '') {
180                         dmobj.setAttr(attr,savevalue,true);
181                     }
182                 }
183 
184                 return false;
185             });
186 
187             return false;
188         });
189     }
190         
191 }
