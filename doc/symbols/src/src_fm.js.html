 37 /**
 38  * For each element in <i>ar</i> 
 39  * call function <i>doFn(id,elm)</i> 
 40  * until end of list or <i>false</i> is returned.
 41  * 
 42  * @static
 43  * @function 
 44  * @param {Object} [ar={}] 
 45  * @param {function} [doFn=function(){}]
 46  * 
 47  */
 48 FM.forEach = function(ar, doFn) {
 49     ar = FM.isset(ar) ? ar : {};
 50     for (var aname in ar) {
 51         if (!doFn(aname, ar[aname]))
 52             return(aname);
 53     }
 54     return null;
 55 }

 57 /**
 58  * Returns new array orobject without <i>index</i> element. 
 59  * 
 60  * @static
 61  * @function 
 62  * @param {Array|Object} arr  
 63  * @param {string|number} index
 64  * 
 65  */
 66 FM.removeArrayElement = function(arr, index) {
 67     var newarr = [];
 68 
 69     for (var ai in arr) {
 70         if (ai != index)
 71             newarr.push(arr[ai]);
 72     }
 73 
 74     return newarr;
 75 }

 78 /**
 79  * Serialize given object. If object has <i>serialize()</i>method, 
 80  * this method will be called, othervise 
 81  * generates JSON representing <i>obj</i> object. 
 82  * 
 83  * @static
 84  * @function 
 85  * @param {Object} [obj=null] Object to serialize.  
 86  * @param {string} [def=""] Default response in case of serialization error. 
 87  * 
 88  */
 89 FM.serialize = function(obj, def) {
 90     def = FM.isset(def) ? def : '';
 91     if (!FM.isset(obj) || !obj)
 92         return def;
 93 
 94     try {
 95         return FM.isFunction(obj.serialize) ? obj.serialize() : JSON.stringify(obj);
 96     } catch (e) {
 97         FM.log(null, e, FM.logLevels.error, 'FM.serialize');
 98     }
 99     return def;
100 }

102 /**
103  * Unserialize given object. If <i>unsMethod</i> method is passed to function, 
104  * this method will be called, othervise 
105  * JSON parsing will occur.
106  * 
107  * @static
108  * @function 
109  * @param {string} str String to deserialize.  
110  * @param {Object} [def=null] Default object to return in case of deserialization error. 
111  * @param {Function} [unsMethod=null] Deserialization method. 
112  * 
113  */
114 FM.unserialize = function(str, def,unsMethod) {
115     def = FM.isset(def) ? def : null;
116     if (!FM.isset(str) || !str)
117         return def;
118     unsMethod = FM.isFunction(unsMethod) ? unsMethod : JSON.parse;
119     try {
120         return unsMethod(str);
121     } catch (e) {
122         FM.log(null, e, FM.logLevels.error, 'FM.unserialize');
123     }
124     return def;
125 }

143 FM._T = function() {
144     if (arguments.length < 1)
145         return('');
146 
147     if (false /*T_messages_loaded */) {
148         // nadji hash i prevedeni string
149         var hash = md5(arguments[0]);
150         var str;
151 
152         if (isset(T_messages[hash])) {
153             str = T_messages[hash];
154         } else {
155             str = arguments[0];
156             if (T_missing_messages == null)
157                 T_missing_messages = {};
158             T_missing_messages[hash] = str;
159         }
160     } else {
161         str = arguments[0];
162     }
163 
164     // ubaci podatke
165     for (var i = 1; i < arguments.length; i++) {
166         str = str.replace("[:" + i + "]", arguments[i]);
167     }
168 
169     // kraj
170     return(str);
171 }
