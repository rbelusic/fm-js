 65                     eq = FM.isEqual(value,v2[name],maxlvl, _l);

 70

 73                 if(v1.length != v2.length) return false;
 74                 
 75                 for(var i=0; i < v1.length; i++) {
 76                     if(!FM.isEqual(v1[i],v2[i],maxlvl, _l)) return false;
 77                 }

354     /**
355 * Set attribute of object
356 * @static
357 * @function 
358 * @param {object} options Object with attributes
359 * @param {string} undoList Undo list 
360 * @param {string} key Attribute name
361 * @param {string} val Value of attribute
362 * @returns {boolean} <i>true</i> if value of attribute is changed, otherwise <i>false</i>
363 */
364     FM.setAttr = function(options,undoList,key,val) {
365         var i,k,aname;
366         var dirty = false;
367 
368         if(!options) return dirty;
369     
370         if(!FM.isString(key)) {
371             for(k in val) {
372                 if(FM.isFunction(val[k]) || !key || FM.isset(options[k])) {
373                     dirty = dirty | FM.setAttr(options,undoList,k,val[k]);
374                 }
375             }
376             return dirty;
377         }
378 
379         var akeys = key.split(".");
380         var ar = options;    
381 
382         for(i = 0; i < akeys.length-1; i++) {
383             k = akeys[i].toString();
384             if(
385                 !FM.isObject(ar) || !FM.isset(ar[k]) || ar[k] == null || (FM.isString(ar[k]) && ar[k] == '')
386                 ) {
387                 return dirty;
388             }
389             ar = ar[k];
390         }
391         aname = akeys[akeys.length-1].toString();
392         if(!FM.isObject(ar)) return dirty;
393 
394         if(FM.isset(ar[aname]) && ar[aname] === val) return dirty;
395         if(undoList && FM.isset(options[akeys[0].toString()]) && !FM.isset(undoList[akeys[0].toString()]))  {
396             undoList[akeys[0].toString()] = options[akeys[0].toString()];
397         }
398         ar[aname] = val;
399         dirty = true;
400         return dirty;
401     }
