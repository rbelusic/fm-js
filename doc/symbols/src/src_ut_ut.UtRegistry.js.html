  9 FM.UtRegistry = FM.defineClass('UtRegistry',FM.Object);
 10 FM.UtRegistry.className = "UtRegistry";

 12 FM.UtRegistry.prototype._init = function(opt) {            
 13     this.cookieName = '';
 14     this.registry = null;
 15 
 16     this._super("_init",opt);
 17     this.objectSubClass = "Registry";
 18 
 19     this.cookieName = this.getAttr('cookieName','fmRegistry');
 20     this.registry = null;
 21 }
 22 
 23 FM.UtRegistry.prototype.set = function(pkey,val) {
 24     if(this.registry == null) {
 25         var cookie = FM.loadCookie(this.cookieName);
 26         if(FM.isset(cookie.reg)) this.registry = FM.unserialize(cookie.reg);
 27         if(this.registry == null || !FM.isObject(this.registry)) this.registry = {};
 28     }
 29     pkey = FM.trim(pkey);
 30     if(FM.startsWith(pkey, "/")) {
 31         pkey = pkey.substr(1);
 32     }
 33     if(FM.endsWith(pkey, "/")) {
 34         pkey = pkey.substr(0,pkey.length -1);
 35     }
 36     var keyArr = pkey.split('/');
 37     var keyIndex = keyArr.length -1;
 38     var cur = this.registry;
 39     FM.forEach(keyArr,function(i,k) {       
 40        var c = FM.isset(cur[k]) ? cur[k] : null;       
 41        if(!c || !FM.isObject(c) || i == keyIndex) {
 42            cur[k] = i != keyIndex ? {} : val;
 43        }
 44        cur = cur[k];
 45        return true;
 46     });
 47     FM.saveCookie(this.cookieName,{reg: FM.serialize(this.registry)});
 48 
 49     return true;
 50 }

 52 FM.UtRegistry.prototype.get = function(pkey,defv) {
 53     if(this.registry == null) {
 54         var cookie = FM.loadCookie(this.cookieName);
 55         if(FM.isset(cookie.reg)) this.registry = FM.unserialize(cookie.reg);
 56         if(this.registry == null || !FM.isObject(this.registry)) this.registry = {};
 57     }
 58     
 59     pkey = FM.trim(pkey);
 60     if(FM.startsWith(pkey, "/")) {
 61         pkey = pkey.substr(1);
 62     }
 63     if(FM.endsWith(pkey, "/")) {
 64         pkey = pkey.substr(0,pkey.length -1);
 65     }
 66     pkey = pkey.replace(/\//g,'.');
 67     var val = FM.getAttr(this.registry,pkey,defv);
 68 
 69     return val;
 70 }
 71 
 72 FM.UtRegistry.prototype.remove = function(pkey) {
 73     if(this.registry == null) {
 74         var cookie = FM.loadCookie(this.cookieName);
 75         if(FM.isset(cookie.reg)) this.registry = FM.unserialize(cookie.reg);
 76         if(this.registry == null || !FM.isObject(this.registry)) this.registry = {};
 77     }
 78 
 79     var ndef = FM.UtRegistry.findKey(this.registry,pkey);
 80     if(!ndef.node) return false;
 81 
 82     var nnode = {};
 83     for(var en in ndef.node) {
 84         if(en != ndef.keyName) {
 85             nnode[en] = ndef.node[en];
 86         }
 87     }
 88 
 89     if(ndef.parent) ndef.parent[ndef.nodeKey] = nnode;
 90     else this.registry = nnode;
 91 
 92     FM.saveCookie(this.cookieName,{reg: FM.serialize(this.registry)});
 93     return true;
 94 }

 96 FM.UtRegistry.prototype.findKey = function(key,force) {
 97     return FM.UtRegistry(this.registry,key,force);
 98 }

100 // static
101 FM.UtRegistry.findKey = function(reg,key,force) {
102     
103     var retc = {found: false, node: null, nodeKey: '', parent: null, keyName: '', keyValue: null};
104     force = FM.isset(force) && force == true ? true: false;
105 
106     if(!FM.isset(reg) || !reg || !FM.isObject(reg)) return retc;
107     if(!FM.isset(key) || !key || key == '') return retc;
108 
109     var apath_f = key.split("/");
110     var apath = [];
111     for(var k=0; k < apath_f.length; k++) {
112         if(apath_f[k] != null && apath_f[k] != '') {
113             apath.push(apath_f[k]);
114         }
115     }
116     if(apath.length < 1) return retc;
117 
118     var ndirs = apath.length -1;
119 
120     retc.keyName = apath[apath.length -1];
121     retc.node = reg;
122     for(var i = 0; i < ndirs; i++) {
123         var nname = apath[i];
124         if(!FM.isset(retc.node[nname]) || !FM.isObject(retc.node[nname])) {
125             if(force) retc.node[nname] = {};
126             else return retc;
127         }
128         retc.parent = retc.node;
129         retc.nodeKey = nname;
130         retc.node = retc.node[nname];
131     }
132 
133     if(FM.isset(retc.node[retc.keyName])) {
134         retc.found = true;
135         retc.keyValue = retc.node[retc.keyName];
136     }
137 
138     return retc;
139 }
