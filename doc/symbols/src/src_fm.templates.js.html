 10 /**
 11  * Apply <i>attrs</i> to template.
 12  * 
 13  * @static
 14  * @function 
 15  * @param {Object} attrs Object with properties to apply.
 16  * @param {string} template Template text.
 17  * @param {boolean} [escapeValues=true] If set to <i>true</i> all values will be escaped before applyinig to template.
 18  * @param {boolean} [encodeValues=false] If set to <i>true</i> all values will be encoded before applyinig to template.
 19  * @param {string} [prefix=""] Prefix of bind variables in template.
 20  * 
 21  * @returns {string} 
 22  */
 23 FM.applyTemplate = function(attrs, template, escapeValues, encodeValues, prefix) {
 24     var templ = FM.isset(template) && template ? template : "";
 25     var pref = FM.isset(prefix) && prefix ? "" + prefix + "." : "";
 26     var val;
 27     var me = this;
 28 
 29     // ako imas dmobject
 30     if (attrs) {
 31         FM.forEach(attrs, function(name, value) {
 32             if (!FM.isset(value) || !value)
 33                 value = '';
 34             if (!FM.isFunction(value) && !FM.isObject(value) && !FM.isArray(value)) {
 35                 if (FM.isset(encodeValues) && encodeValues == true) {
 36                     val = FM.urlEncode(value.toString());
 37                 } else {
 38                     val = value;
 39                 }
 40                 if (FM.isset(escapeValues) && escapeValues != false) {
 41                     val = FM.escapeStr(val);
 42                 }
 43 
 44                 templ = templ.replace(new RegExp("\\[\\:" + pref + name + "\\]", "g"), val);
 45             } else if ((FM.isObject(value) || FM.isArray(value)) && (pref.split(".").length - 1 < 2)) {
 46                 templ = FM.applyTemplate(value, templ, escapeValues, encodeValues, pref + name);
 47             }
 48             return true;
 49         });
 50     }
 51 
 52     // kraj
 53     return(templ);
 54 }
