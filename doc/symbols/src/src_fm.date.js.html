 29 /**
 30  * Basic date and time function.
 31  * Derived from <a href="https://github.com/slevithan">Steven Levithan</a> date formating function.
 32  * 
 33  * @static
 34  * @function 
 35  * @see <a href="http://blog.stevenlevithan.com/archives/date-time-format">Steven Levithan blog article</a> for full list of options.
 36  */
 37 FM.dateFormat = function() {
 38     var token = /d{1,4}|m{1,4}|yy(?:yy)?|([HhMsTt])\1?|[LloSZ]|"[^"]*"|'[^']*'/g,
 39         timezone = /\b(?:[PMCEA][SDP]T|(?:Pacific|Mountain|Central|Eastern|Atlantic) (?:Standard|Daylight|Prevailing) Time|(?:GMT|UTC)(?:[-+]\d{4})?)\b/g,
 40         timezoneClip = /[^-+\dA-Z]/g,
 41         pad = function(val, len) {
 42         val = String(val);
 43         len = len || 2;
 44         while (val.length < len)
 45             val = "0" + val;
 46         return val;
 47     };
 48 
 49     // Regexes and supporting functions are cached through closure
 50     return function(date, mask, utc) {
 51         var dF = FM.dateFormat;
 52 
 53         // You can't provide utc if you skip other args (use the "UTC:" mask prefix)
 54         if (arguments.length == 1 && Object.prototype.toString.call(date) == "[object String]" && !/\d/.test(date)) {
 55             mask = date;
 56             date = undefined;
 57         }
 58 
 59         // Passing date through Date applies Date.parse, if necessary
 60         date = date ? new Date(date) : new Date;
 61         if (isNaN(date))
 62             throw SyntaxError("invalid date");
 63 
 64         mask = String(dF.masks[mask] || mask || dF.masks["default"]);
 65 
 66         // Allow setting the utc argument via the mask
 67         if (mask.slice(0, 4) == "UTC:") {
 68             mask = mask.slice(4);
 69             utc = true;
 70         }
 71 
 72         var _ = utc ? "getUTC" : "get",
 73             d = date[_ + "Date"](),
 74             D = date[_ + "Day"](),
 75             m = date[_ + "Month"](),
 76             y = date[_ + "FullYear"](),
 77             H = date[_ + "Hours"](),
 78             M = date[_ + "Minutes"](),
 79             s = date[_ + "Seconds"](),
 80             L = date[_ + "Milliseconds"](),
 81             o = utc ? 0 : date.getTimezoneOffset(),
 82             flags = {
 83             d: d,
 84             dd: pad(d),
 85             ddd: dF.i18n.dayNames[D],
 86             dddd: dF.i18n.dayNames[D + 7],
 87             m: m + 1,
 88             mm: pad(m + 1),
 89             mmm: dF.i18n.monthNames[m],
 90             mmmm: dF.i18n.monthNames[m + 12],
 91             yy: String(y).slice(2),
 92             yyyy: y,
 93             h: H % 12 || 12,
 94             hh: pad(H % 12 || 12),
 95             H: H,
 96             HH: pad(H),
 97             M: M,
 98             MM: pad(M),
 99             s: s,
100             ss: pad(s),
101             l: pad(L, 3),
102             L: pad(L > 99 ? Math.round(L / 10) : L),
103             t: H < 12 ? "a" : "p",
104             tt: H < 12 ? "am" : "pm",
105             T: H < 12 ? "A" : "P",
106             TT: H < 12 ? "AM" : "PM",
107             Z: utc ? "UTC" : (String(date).match(timezone) || [""]).pop().replace(timezoneClip, ""),
108             o: (o > 0 ? "-" : "+") + pad(Math.floor(Math.abs(o) / 60) * 100 + Math.abs(o) % 60, 4),
109             S: ["th", "st", "nd", "rd"][d % 10 > 3 ? 0 : (d % 100 - d % 10 != 10) * d % 10]
110         };
111 
112         return mask.replace(token, function($0) {
113             return $0 in flags ? flags[$0] : $0.slice(1, $0.length - 1);
114         });
115     };
116 }();

170 FM.dateToString = function(dat, utc) {
171     dat = FM.isset(dat) ? dat : new Date();
172     
173     try {
174         var s = FM.dateFormat(dat, (utc ? 'fmUtcDateTime' : 'fmDateTime'));
175         return s;
176     } catch(e) {
177         return '';
178     }
179 }

181 FM.isDateString = function(sdate) { 
182     if (!FM.isString(sdate)) {
183         return false;
184     }
185 
186     try {
187         var ds = FM.dateFormat(sdate);
188         var d = new Date(ds);
189         return d ? true : false;
190     } catch(e) {
191         return false;
192     }
193 }

195 FM.parseLocalDateString = function(sdate) {
196     if (!FM.isset(sdate) || sdate == null || sdate == '')
197         return(null);
198 
199     try {
200         var s = FM.dateFormat(sdate);
201         var d = new Date(s);
202         return d;
203     } catch(e) {
204         return null;
205     }
206 }

208 FM.parseDateString = function(sdate, utc) { 
209     if (!FM.isset(sdate) || sdate == null || sdate == '') {
210         return(null);
211     }
212 
213     try {
214         var s = FM.dateFormat(sdate + (utc ? 'Z' : ''));
215         var d = new Date(s);
216         return d;
217     } catch(e) {
218         return null;
219     }
220 }

242 /* calculate time difference in seconds between two dates 
243  */

245 FM.timeBetween = function(d1, d2) {
246     if (
247         !FM.isset(d1) || !d1 || !FM.isset(d1.getTime) || 
248         !FM.isset(d2) || !d2 || !FM.isset(d2.getTime)
249     ) {
250         return false;    
251     }
252     // Calculate the difference in milliseconds
253     return (d2.getTime()/1000 - d1.getTime()/1000);
254 }

257 /* calculate time difference in seconds between two dates and returns descriptive string
258  */
259 FM.strTimeBetween = function(d1, d2) {
260     var dif = FM.timeBetween(d1,d2);    
261     if (dif == false) {
262         return '';    
263     }
264     var dif_abs = Math.abs(dif);
265     
266 
267     // The number of milliseconds in one day
268     var ONE_DAY = 24;
269     var ONE_HOUR = 60;
270     var ONE_MINUTE = 60;
271 
272     var ret;
273 
274 
275     if (dif < 60) {
276         ret = FM._T(dif > 0 ? "[:1] seconds ago" : "In [:1] seconds", dif_abs);
277     } else { // ONE_MINUTE
278         dif_abs = Math.round(dif_abs / ONE_MINUTE);
279         if (dif_abs < 60) {
280             ret = FM._T(dif > 0 ? "[:1] minutes ago" : "In [:1] minutes", dif_abs);
281         } else { // ONE_HOUR
282             dif_abs = Math.round(dif_abs / ONE_HOUR);
283             if (dif_abs < 24) {
284                 ret = FM._T(dif > 0 ? "[:1] hours ago" : "In [:1] hours", dif_abs);
285             } else { // ONE_DAY
286                 dif_abs = Math.round(dif_abs / ONE_DAY);
287 
288                 if (dif_abs == 1) {
289                     ret = FM._T(dif > 0 ? "Yesterday" : "Tomorow", dif_abs);
290                 }
291                 else {
292                     ret = FM._T(dif > 0 ? "[:1] days ago" : "In [:1] days", dif_abs);
293                 }
294             }
295         }
296     }
297 
298 
299     // kraj
300     return(ret);
301 }

303 FM.dateLocalFormat = function(d) {
304     if (!FM.isset(d) || d == null || d == '')
305         return('');
306 
307     try {
308         var s = d.toLocaleString();
309     } catch (err) {
310         alert(err)
311     }
312     ;
313     var i = s.indexOf("GMT");
314     if (i >= 0)
315         s = s.substr(0, i);
316     return(s);
317 }
