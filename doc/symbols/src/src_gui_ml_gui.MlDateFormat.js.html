1 /**
  2 * ML date fromating extensions class.

  9 FM.MlDateFormat = FM.defineClass('MlDateFormat',FM.MlExtension);
 10 
 11 // methods
 12 FM.MlDateFormat.prototype._init = function(attrs,node) {
 13     this._super("_init",attrs,node);
 14     this.objectSubClass = "DateFormat";
 15 }
 16 
 17 FM.MlDateFormat.prototype.run = function(obs) {
 18     this._super("run",obs);
 19     var attr = $(this.node).attr('data-fmml-date-format','');
 20 
 21     if(this.node.nodeName == 'INPUT' || this.node.nodeName == 'TEXTAREA') {
 22         $(this.node).watermark(attr);
 23     } 
 24 }
