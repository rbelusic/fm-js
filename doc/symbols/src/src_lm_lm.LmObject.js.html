1 /**
  2 * Basic LM class. 
  3 * 
  4 * @class FM.LmObject
  5 * @extends FM.Object
  6 * @memberOf FM
  7 * @param {FM.AppObject} app application object
  8 * @param {object} [options] Options
  9 */    
 10 FM.LmObject = FM.defineClass('LmObject',FM.Object);
 11 
 12 FM.LmObject.prototype._init = function(app,opt) {            
 13     this.setExecuted(false);
 14     this.setApp(app);
 15 
 16     this._super("_init",opt);
 17     this.objectSubClass = "Object";
 18 }
 19 
 20 FM.LmObject.prototype.run = function() {
 21     this._super("run");
 22     this.setExecuted(true);
 23 }
 24 
 25 FM.LmObject.prototype.dispose = function() { 
 26     // reset
 27     this.setExecuted(false);
 28     this.setDmObject();
 29     this.setApp();            
 30     this._super("dispose");
 31 }
 32 
 33 FM.LmObject.prototype.isExecuted = function() {
 34     return this.executed;
 35 }
 36 FM.LmObject.prototype.setExecuted = function(e) {
 37     this.executed = FM.isset(e) && e == true;
 38 }
 39 
 40 FM.LmObject.prototype.getApp = function() {
 41     return this.app;
 42 }
 43 
 44 FM.LmObject.prototype.setApp = function(a) {
 45     this.app = FM.isset(a) && a ? a : null;
 46 }
 47 
 48 FM.LmObject.prototype.getDmObject = function() {
 49     return this.dmObject;
 50 }
 51 
 52 FM.LmObject.prototype.setDmObject = function(o,addListener) {
 53     if(o && o === this.dmObject) return;
 54     
 55     if(this.dmObject) {
 56         this.dmObject.removeListener(this);
 57         if(this.getProperty('dmObjectCreated','true') === 'true') {
 58             this.dmObject.dispose();
 59             this.dmObject = null;
 60         }
 61     }
 62     
 63     this.dmObject = FM.isset(o) && o ? o : null;
 64     if(this.dmObject && FM.isset(addListener) && addListener) {
 65         this.dmObject.addListener(this);
 66     }    
 67 }
 68 
 69 FM.LmObject.prototype.getRegistryRoot = function() {
 70     return  "/" + (this.app ? this.app.getSubClassName() : "APP") + 
 71             "/sClass" +
 72             "/" + this.getSubClassName()
 73     ;    
 74 }
 75 
 76 FM.LmObject.prototype.getRegistryValue = function(key,dval) {
 77     return this.app ? this.app.appRegistry.get(
 78          this.getRegistryRoot() +
 79          key,
 80          dval
 81      ) : dval;    
 82 }
 83 
 84 FM.LmObject.prototype.setRegistryValue = function(key,val) {
 85     if(this.app) this.app.appRegistry.set(
 86          this.getRegistryRoot() +
 87          key,
 88          val
 89      );        
 90 }
