 19 FM.MlTypeaheadMenu.prototype._createList = function(obs) {
 20     var dmconfName = this.getAttr('data-fmml-list','');
 21     var listOpt = {};
 22     var pref = 'data-fmml-list-attr-';
 23     var prefLen = pref.length;
 24     
 25     this.forEachAttr(function(pname,value) {
 26         if(FM.startsWith(pname,pref)) {
 27             listOpt[pname.substring(prefLen)] = value;
 28         }                    
 29         return true;
 30     });
 31     
 32     // input param
 33     var oObj = obs.getDmObject();
 34     
 35     if(oObj && FM.isset(oObj.forEachAttr)) { // fm obect
 36         oObj.forEachAttr(function(pname,value) {
 37             listOpt[pname] = value;
 38             return true;
 39         });
 40         listOpt["objectClass"] = oObj.getSubClassName();
 41     } else if(FM.isArray(oObj) || FM.isObject(oObj)) { // array
 42         FM.forEach(oObj, function(pname,value) {
 43             listOpt[pname] = value;
 44             return true;
 45         });        
 46     }
 47     
 48     return new FM.DmList(listOpt,dmconfName,obs.getApp());
 49 }

 51 FM.MlTypeaheadMenu.prototype._getItemDataID = function(itm) {
 52     var srch = "<div id='typeahead-";
 53     
 54     var itext = itm.indexOf(srch);
 55     if(itext < 0) {
 56         return '';
 57     }
 58     var txt = itm.substr(itext + srch.length);
 59     itext = txt.indexOf("'");
 60     if(itext < 0) {
 61         return '';
 62     }
 63     return txt.substr(0,itext);
 64 }

 66 FM.MlTypeaheadMenu.prototype.run = function(obs) {
 67     this._super("run",obs);
 68     var itmtemplate = 
 69         "<div id='typeahead-[:data_id]' class=''>" +
 70         "  <div class='fmmlCover pull-left'>" +
 71         "    <img class='media-object' src='[:cover_url]'></img>" +
 72         " </div>" +
 73         " <div class='media-body'>[:description]</div>" + 
 74         "</div>"
 75     ;
 76     this.listCache = {};
 77     var me = this;
 78     var _obs = obs;
 79     
 80     $(_obs.getNode()).typeahead({
 81         updater: function(itm) {
 82             /*
 83             var srch = "<div class='media-body'>";
 84             var qry = this.query;
 85             var itext = itm.indexOf(srch);
 86             if(itext < 0) {
 87                 return false;
 88             }
 89             var btxt = itm.substr(0,itext + srch.length);
 90             var txt = itm.substr(itext + srch.length);
 91             itext = txt.indexOf("</div>");
 92             if(itext < 0) {
 93                 return false;
 94             }
 95             var tag = txt.substr(0,itext);
 96             */
 97             // run host if required
 98             var hostid = me.getAttr('data-fmml-typeahead-add-to-on-select','');
 99             if(hostid != '') {
100                 var node = document.getElementById(hostid);
101                 if( node && FM.isset(node.fmmlHost) && node.fmmlHost && 
102                     FM.isset(node.fmmlHost.onAddObjectToList)
103                 ) {
104                     var o = FM.getAttr(me.listCache,itm,null);
105                     if(o) node.fmmlHost.onAddObjectToList(this,{object: o});
106                 }
107             }
108             return ''; //tag;
109         },
110         highlighter: function(itm) {
111             var o = FM.getAttr(me.listCache,itm,null);
112             if(!o) return itm;
113             var attr = me.getAttr('data-fmml-attr-name','');
114             var menuText = me.getAttr('data-fmml-list-text-attr',attr);
115             var menuImg = me.getAttr('data-fmml-img-attr','icon_url');
116             
117             var itmstr = FM.applyTemplate({
118                     data_id: FM.md5(itm),
119                     cover_url: o.getAttr(menuImg),
120                     description: o.getAttr(menuText,'...')
121                 }, 
122                 itmtemplate, 
123                 false, false
124             );
125             var inode = $(itmstr)[0];
126 
127             return inode;
128             
129             var srch = "<div class='media-body'>";
130             var qry = this.query;
131             var itext = itm.indexOf(srch);
132             if(itext < 0) {
133                 return false;
134             }
135             var btxt = itm.substr(0,itext + srch.length);
136             var txt = itm.substr(itext + srch.length);
137             itext = txt.indexOf("</div>");
138             if(itext < 0) {
139                 return false;
140             }
141             var atxt = txt.substr(itext);
142             txt = txt.substr(0,itext).replace(qry,'<strong>' + qry + "</strong>");
143             return btxt + txt + atxt;
144             
145         },
146         matcher: function(itm) {
147             var dmobj = FM.getAttr(me.listCache,itm,null);
148             var attr = me.getAttr('data-fmml-attr-name','');
149             var menuText = me.getAttr('data-fmml-list-text-attr',attr);
150             var txt = dmobj.getAttr(menuText,'');
151             var qry = this.query;
152 /*            
153             var srch = "<div class='media-body'>";
154             var itext = itm.indexOf(srch);
155             if(itext < 0) {
156                 return false;
157             }
158             var txt = itm.substr(itext + srch.length);
159             itext = txt.indexOf("</div>");
160             if(itext < 0) {
161                 return false;
162             }
163             txt = txt.substr(0,itext);
164 */
165             return FM.isString(txt) && FM.isString(qry) && txt.toLowerCase().indexOf(qry.toLowerCase()) > -1;
166         },
167         source: function(val,cbFn){
168             var attr = me.getAttr('data-fmml-attr-name','');
169             var lstattr = me.getAttr('data-fmml-list-attr-name',attr);
170             if(attr != '') {
171                 var dmList = me._createList(_obs);
172                 dmList.setAttr(lstattr,val);
173                 var lstnr = {
174                     /** @ignore */
175                     onListEnd: function(sender,data) {
176                         dmList.removeListener(lstnr);
177                         var menuText = obs.getAttr('data-fmml-list-text-attr',attr);
178                         var menuImg = obs.getAttr('data-fmml-img-attr','icon_url');
179                         var retc = [];
180                         dmList.forEachListElement(function(i,o) {
181                             var itmstr = FM.applyTemplate({
182                                     data_id: FM.md5(o.getDataID()),
183                                     cover_url: o.getAttr(menuImg),
184                                     description: o.getAttr(menuText,'...')
185                                 }, 
186                                 itmtemplate, 
187                                 false, false
188                             );
189                             var inode = $(itmstr)[0];
190                             inode.fmmlMenuDmObject = o;
191                             retc.push(o.getDataID());
192                             me.listCache[o.getDataID()] = o;
193                             return true;
194                         });
195                         dmList.dispose();
196                         cbFn(retc);
197                     },
198                     onListError: function(sender,data) {
199                         dmList.removeListener(lstnr);
200                         dmList.dispose();
201                         cbFn([]);
202                     }
203                 };
204                 dmList.addListener(lstnr);
205                 dmList.getData(false);
206             }
207         },
208         items: parseInt(me.getAttr('data-fmml-list-size','7')),
209         minLength: parseInt(me.getAttr('data-fmml-typeahead-chars','2'))
210     });
211     
212     return true;
213 }
