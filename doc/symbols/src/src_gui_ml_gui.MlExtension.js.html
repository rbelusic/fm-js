 56 // static
 57 FM.MlExtension.extensionTypes = {
 58     GLOBAL: {}
 59 };
 60 
 61 FM.MlExtension.addExtensionType = function(type,fn,appCls) {    
 62     if(!FM.isset(fn) || !FM.isFunction(fn)) return false;
 63     appCls = FM.isset(appCls) && FM.isString(appCls) && appCls != '' ? appCls : 'GLOBAL';
 64     if(!FM.isset(FM.MlExtension.extensionTypes[appCls])) {
 65         FM.MlExtension.extensionTypes[appCls]= {};
 66     }
 67     
 68     FM.MlExtension.extensionTypes[appCls][type] = fn;
 69     return true;
 70 }

 74 /**
 75 * Returns MlExtension <b>config</b> class function for <b>config</b> subclass type
 76 * @static
 77 * @function    
 78 * @param {object} app Application
 79 * @param {String} name Configuration name
 80 * @return {object} extension configuration or null if not found
 81 */   
 82 FM.MlExtension.getConfiguration = function(app,name) {
 83     var list = FM.MlExtension.extensionTypes;
 84 
 85     app = FM.isset(app) && app ? app : null;
 86     var appCls = app ? app.getSubClassName() : null;
 87     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
 88         
 89     var obj = null;
 90     if(appCfg && FM.isset(appCfg[name])) {
 91         obj = appCfg[name];
 92     } else if(app && FM.isArray(app.applicationObjectsSpace)) {
 93         FM.forEach(app.applicationObjectsSpace,function(i,ns) {
 94             if(FM.isset(list[ns]) && FM.isset(list[ns][name])) {
 95                 obj = list[ns][name];
 96                 return false;
 97             }
 98             return true;
 99         });
100     }
101     
102     if(!obj && FM.isset(list['GLOBAL'][name])) {
103         obj = list['GLOBAL'][name];
104     }
105     
106     return obj;
107 }

109 FM.MlExtension.newExtension = function(app,attrs,node,type) {
110     var clsFn = FM.MlExtension.getConfiguration(app,type);
111     return clsFn ? new clsFn(attrs,node) : null;
112 }
