  5 /**
  6  * Determine if a variable containing e-mail address.
  7  * 
  8  * @static
  9  * @function 
 10  * @param email The variable to be checked.
 11  * @returns {boolean} 
 12  */
 13 FM.validateEmail = function(email) {
 14     var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
 15     return re.test(email);
 16 }
