  9 /**
 10  * Check if attribute exists in object.
 11  * 
 12  * @static
 13  * @function 
 14  * @param {Object} options Object with attributes.
 15  * @param {string} key Attribute name.
 16  * @returns {boolean}
 17  */
 18 FM.isAttr = function(options, key) {
 19     if (!FM.isset(options) || !options || !FM.isset(key))
 20         return false;
 21 
 22     var akeys = key.split(".");
 23     var ar = options;
 24     var val = null;
 25     for (var i = 0; i < akeys.length; i++) {
 26         var k = akeys[i].toString();
 27         if (!FM.isObject(ar) || !FM.isset(ar[k]))
 28             return false;
 29         ar = ar[k];
 30     }
 31     return true;
 32 }

 34 /**
 35  * Get attribute value from object.
 36  * 
 37  * @static
 38  * @function 
 39  * @param {Object} options Object with attributes.
 40  * @param {string} key Attribute name.
 41  * @param {...} [defv=""] Default value of attribute.
 42  * @returns {..,} Value of attribute or default value.
 43  */
 44 FM.getAttr = function(options, key, defv) {
 45     if (!options)
 46         return options;
 47 
 48     if (FM.isset(key)) {
 49         var akeys = key.split(".");
 50         var ar = options;
 51         var val = null;
 52         for (var i = 0; i < akeys.length; i++) {
 53             var k = akeys[i].toString();
 54             if (
 55                 (!FM.isObject(ar) && !FM.isFunction(ar)) || !FM.isset(ar[k]) || ar[k] == null || (FM.isString(ar[k]) && ar[k] == '')
 56                 ) {
 57                 return(FM.isset(defv) ? defv : '');
 58             }
 59             ar = ar[k];
 60             val = ar;
 61         }
 62 
 63         return val == null ? (FM.isset(defv) ? defv : '') : val;
 64     } else {
 65         return FM.cloneObject(options);
 66     }
 67 }

 69 /**
 70  * Set attribute of object
 71  * 
 72  * @static
 73  * @function 
 74  * @param {Object} options Object with attributes.
 75  * @param {Object} undoList Undo list (or<i>null</i>).
 76  * @param {string} key Attribute name.
 77  * @param {string|number|...} val Value of attribute.
 78  * @returns {boolean} <i>true</i> if value of attribute is changed, otherwise <i>false</i>.
 79  */
 80 FM.setAttr = function(options, undoList, key, val) {
 81     var i, k, aname;
 82     var dirty = false;
 83 
 84     if (!options)
 85         return dirty;
 86 
 87     if (!FM.isString(key)) {
 88         for (k in val) {
 89             if (FM.isFunction(val[k]) || !key || FM.isset(options[k])) {
 90                 dirty = dirty | FM.setAttr(options, undoList, k, val[k]);
 91             }
 92         }
 93         return dirty;
 94     }
 95 
 96     var akeys = key.split(".");
 97     var ar = options;
 98 
 99     for (i = 0; i < akeys.length - 1; i++) {
100         k = akeys[i].toString();
101         if (
102             !FM.isObject(ar) || !FM.isset(ar[k]) || ar[k] == null || (FM.isString(ar[k]) && ar[k] == '')
103             ) {
104             return dirty;
105         }
106         ar = ar[k];
107     }
108     aname = akeys[akeys.length - 1].toString();
109     if (!FM.isObject(ar))
110         return dirty;
111 
112     if (FM.isset(ar[aname]) && ar[aname] === val)
113         return dirty;
114     if (undoList && FM.isset(options[akeys[0].toString()]) && !FM.isset(undoList[akeys[0].toString()])) {
115         undoList[akeys[0].toString()] = options[akeys[0].toString()];
116     }
117     ar[aname] = val;
118     dirty = true;
119     return dirty;
120 }

126 FM.resolveAttrValue = function(options, attrName, def, context) {
127     options = FM.isset(options) && options && FM.isObject(options) ? options : {};
128     context = FM.isset(context) && context && FM.isObject(context) ? context : {};
129     var v = FM.getAttr(options, attrName, def);
130 
131     // eval ?
132     if (FM.isString(v) && FM.startsWith(v, '@')) {
133 
134         v = v.substring(FM.startsWith(v, '@@') ? 2 : 1);
135         context._fn = function() {
136             return eval(v);
137         }
138         try {
139             var retv = context._fn();
140         } catch (e) {
141             FM.log(context, e, FM.logLevels.error, 'FM.resolveAttrValue');
142 
143             return undefined;
144         }
145         v = retv;
146     }
147 
148     return v;
149 }

157     context = context && FM.isObject(context) ? context : {};
158     var v = attrName;

162         v = attrName.substring(FM.startsWith(attrName, '@@') ? 2 : 1);
163         context._fn = function() {
164             return eval(v);
165         }
166         try {
167             var retv = context._fn();
168         } catch (e) {
169             FM.log(context, e, FM.logLevels.error, 'FM.resolveAttrName');
170             return undefined;
171         }
172         v = retv;
173     } else {
174         v = FM.getAttr(options, attrName, def);
175     }
176 
177     return v;
