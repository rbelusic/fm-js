1 /**
  2  * ML date  edit extensions class. 
  3  * 
  4  * @class FM.MlDateEdit
  5  * @extends FM.MlExtension
  6  * @param {object} [attrs] DOM node attributes
  7  * @param {DOMnode} node DOM node
  8  * @requires http://api.jqueryui.com/datepicker/
  9  * @description You can use other plugins instead jquery ui by replacing initWidget() method
 10  */
 11 FM.MlDateEdit = FM.defineClass('MlDateEdit', FM.MlEdit);
 12 
 13 FM.MlDateEdit.prototype._init = function(attrs, node) {
 14     this._super("_init", attrs, node);
 15     this.objectSubClass = "DateEdit";
 16     this.editWidgetDate =  null;
 17 }
 18 
 19 // methods
 20 FM.MlDateEdit.prototype.initWidget = function() {
 21     this._super("initWidget");
 22     this.editWidgetDate = $("<input style='display:none;'></input>")[0];
 23     $(this.editWidget? this.editWidget : this.node).after(this.editWidgetDate);
 24     var me = this;
 25         
 26     $(this.editWidgetDate).val(this.getObserver().getValue());
 27     
 28     var opts = {
 29         dateFormat: "yy-mm-dd",
 30          showOn: "both",
 31         onSelect: function(dateText, oDp) {
 32             if (obs) {
 33                 me.getObserver().setValue(dateText);
 34             }
 35             return true;
 36         }
 37     };
 38     
 39     this.forEachAttr(function(n,v) {
 40         if(FM.startsWith(n,'data-fmml-wg-option-')) {
 41             opts[FM.camelCase(n.substr(20))] = v;
 42         }
 43         return true;
 44     });
 45 
 46     $(this.editWidgetDate).datepicker(opts);
 47 }
 48 
 49 FM.MlDateEdit.prototype.disposeWidget = function() {    
 50     if (this.editWidgetDate) {
 51         $(this.editWidgetDate).remove();        
 52     }
 53     this.editWidgetDate = null;
 54     
 55     this._super("disposeWidget");    
 56 }
 57 
 58 FM.MlDateEdit.prototype.render = function(value) {
 59     this._super("render",value);
 60     
 61     if (this.editWidgetDate) {
 62         $(this.editWidgetDate).datepicker( 
 63             "setDate", this.getObserver().getValue()
 64         );
 65     }
 66 }
 67 
 68 FM.MlExtension.addExtensionType('DateEdit', FM.MlDateEdit);
