1 /**
  2 * Timer class. 
  3 * 
  4 * @class FM.UtTimer

  6 
  7 FM.UtTimer = FM.defineClass('UtTimer',FM.Object);
  8 
  9 FM.UtTimer.minPeriod = 1;
 10 FM.UtTimer.timeoutHandle = null;
 11 FM.UtTimer.jobsList = [];
 12 FM.UtTimer.suspended = false;
 13 
 14 
 15 FM.UtTimer.__checklist__ = function() {
 16     if(!FM.UtTimer.isQueueSuspended()) {
 17         var nlist = [];
 18         for(var i=0; i < FM.UtTimer.jobsList.length; i++) {
 19             var job = FM.UtTimer.jobsList[i];
 20             if(
 21                 job.executecount != 0 && !job.isSuspended() &&
 22                 job.lastRun + job.period * 1000 < new Date().getTime()
 23             ) {
 24                 job.lastRun = new Date().getTime();
 25                 job.executecount--;
 26                 if(job.executecount != 0) nlist.push(job);
 27                 job.fireEvent(job.event, job.evdata);
 28             } else {
 29                 if(job.executecount != 0) nlist.push(job);
 30             }
 31         }
 32         FM.UtTimer.jobsList = nlist;
 33 
 34         if(FM.UtTimer.jobsList.length > 0) {
 35             FM.UtTimer.timeoutHandle = setTimeout(
 36                 "FM.UtTimer.__checklist__()",
 37                 FM.UtTimer.minPeriod * 1000
 38             );
 39         }else {
 40             FM.UtTimer.timeoutHandle = null;
 41         }
 42     } else { // za svaki slucaj
 43         FM.UtTimer.timeoutHandle = null;
 44     }
 45 }
 46 
 47 FM.UtTimer.removeFromQueue = function(job) {
 48     var nlist = [];
 49     for(var i=0; i < FM.UtTimer.jobsList.length; i++) {
 50         if(FM.UtTimer.jobsList[i] != job) {
 51             nlist.push(FM.UtTimer.jobsList[i]);
 52         }
 53     }
 54     FM.UtTimer.jobsList = nlist;    
 55 }
 56 
 57 FM.UtTimer.addToQueue = function(job) {
 58     FM.UtTimer.jobsList.push(job);
 59     if(!FM.UtTimer.timeoutHandle) {
 60         FM.UtTimer.__checklist__();
 61     }    
 62 }
 63 
 64 FM.UtTimer.isQueueSuspended = function() {
 65     return FM.UtTimer.suspended;
 66 }
 67 
 68 FM.UtTimer.suspendQueue = function() {
 69     FM.UtTimer.suspended = true;
 70 }
 71 
 72 FM.UtTimer.resumeQueue = function() {
 73     FM.UtTimer.suspended = false;
 74     FM.UtTimer.__checklist__();
 75 }
