 14 * @class FM.MlObserverAttribute

 17 * @param {DOMnode} node DOM node
 18 */    
 19 FM.MlObserverAttribute = FM.defineClass('MlObserverAttribute',FM.MlObserver);
 20 
 21 
 22 FM.MlObserverAttribute.prototype._init = function(app,attrs,node) {            
 23     this._super("_init",app,attrs,node);
 24     this.objectSubClass ="ObserverAttribute";
 25 }        
 26 
 27 FM.MlObserverAttribute.prototype.setNodeValue = function() {
 28     // conf
 29     var attrname = this.getAttr('data-fmml-attr-name','');
 30     var attrtype = this.getAttr('data-fmml-attr-type','string');
 31     var decplaces = parseInt(this.getAttr('data-fmml-attr-decimals','2'));
 32     var attrtemplate = this.getAttr('data-fmml-attr-template','');
 33     var dateIsUtc = this.getAttr('data-fmml-date-is-utc','true') != 'false';
 34     var dateFormat = this.getAttr('data-fmml-date-display-as','date');
 35     var defval = this.resolveAttrValue('data-fmml-attr-default-value','');
 36     var newlntobr = this.resolveAttrValue('data-fmml-attr-newln-br','false');
 37     var domprop = this.getAttr('data-fmml-observer-dom-property','');
 38     var dmobj = this.getDmObject();
 39     
 40     var alttext = this.getAttr('data-fmml-attr-hover-text','');
 41         
 42     // check defval    
 43     var value = 
 44         dmobj && attrname != '' ?
 45         dmobj.getAttr(attrname,'') :
 46         ''
 47     ;
 48     if(dmobj && value == '' && defval != '') {
 49         return dmobj.setAttr(attrname,defval,true);
 50     }
 51     
 52     // value
 53     if(value == '') value = defval;
 54     
 55     if (this.lastValue == value || dmobj == null) return;
 56     this.lastValue = value;
 57     
 58     // dates
 59     if(attrtype == "date") {
 60         if(FM.isDateString(value)) {
 61              var dateObj = FM.parseDateString(value,dateIsUtc);
 62             if(dateFormat == 'local') {
 63                 value = FM.dateLocalFormat(dateObj);
 64             } else if(dateFormat == 'date') {
 65                 value = FM.dateFormat(dateObj,'mediumDate');
 66             } else if(dateFormat == 'datetime') {
 67                 value = FM.dateFormat(dateObj,'mediumDateTime');
 68             } else if(dateFormat == 'time') {
 69                 value = FM.dateFormat(dateObj,'isoTime');
 70             } else if(dateFormat == 'ago') {
 71                 value = FM.strTimeBetween(dateObj,new Date());
 72             } 
 73         }  
 74     } else if(attrtype == "number") {
 75         value = parseFloat(0.0 + value);
 76         value = value.toFixed(decplaces);
 77     }
 78     if(attrtemplate != "") {
 79         var app = this.app;
 80         if (newlntobr == 'true') value = value.replace(/\n/g, "<br />");
 81         var attrlist = {
 82             id: value,
 83             alttext: alttext
 84         };
 85         var tname = attrtemplate;
 86         var me = this;
 87         
 88         FM.UtTemplate.getTemplate(app,attrlist,tname,function(isok,templ) {
 89             if(isok) {
 90                 var tmplstr = FM.applyTemplate(attrlist, templ, false, false);
 91                 var tmplnode = $(tmplstr);
 92                 if(FM.isset(tmplnode[0])) tmplnode = tmplnode[0];
 93 
 94                 me.node.innerHTML = '';
 95                 $(me.node).append(tmplnode);
 96                 FM.MlHost.initChildNodes(app, tmplnode);
 97             }
 98         });
 99     
100         return;
101     }
102     // set
103     if(domprop !== '') {
104         this.node.setAttribute(domprop,value);
105     } else if(attrtype == "url") {
106         this.node.innerHTML = '';
107 
108         if (value != '') {
109             var a = document.createElement('a');
110             a.setAttribute('href', value);
111             a.setAttribute('target', '_blank');
112             a.innerText = value;
113             
114             this.node.appendChild(a);
115         }
116         else {
117             this.node.innerHTML = ' ';
118         }
119     } else {    
120         if(FM.isset(this.node.value)) {
121             var selStart = this.node.selectionStart;
122             var selEnd = this.node.selectionEnd;
123             this.node.value = value;
124             if(FM.isset(this.node.setSelectionRange)) {
125                 this.node.setSelectionRange(selStart, selEnd);
126             }
127         } else if(this.node.nodeName == 'IMG') {
128             this.node.setAttribute("src",value);
129         } else if(this.node.nodeName == 'A') {
130             this.node.setAttribute("href",value);
131         } else {
132             if (newlntobr == 'true') value = value.replace(/\n/g, "<br />");
133             this.node.innerHTML = value == '' ? ' ' : value;
134         }
135         this.node.fmmlValueSync = value;
136     }
137 }
