 26 FM.MlObserver.prototype._init = function(app,attrs,node) {
 27     this._super("_init",app,attrs);
 28     this.objectSubClass = "Observer";
 29     this.executed = false;
 30     this.node = node;
 31     this.node.fmmlObserver = this;
 32     this.lastValue = null;
 33     this.extensions = [];
 34     
 35     
 36     // find error host
 37     this.errorObject = this.getAttr('data-fmml-error-host','') != '' ?            
 38     new FM.DmGenericError({
 39         id: '',
 40         text: ''
 41     }) : 
 42     null
 43     ;
 44     this.getHost();
 45     if(this.host) this.host.addObserver(this);
 46 }

 48 FM.MlObserver.prototype.run = function() {
 49     //if(this.executed) this.dispose();
 50     this._super("run");
 51 
 52     var attrname = this.getAttr('data-fmml-attr-name','');
 53     var dmobj = this.getDmObject();
 54 
 55     for(var i = 0; i < this.extensions.length; i++) {
 56         try {
 57             this.runExtension(this.extensions[i]);
 58         } catch(err) {
 59             this.log(err,FM.logLevels.error,'MlObserver.run');     
 60         }
 61     }
 62     this.executed = true;
 63     this.setNodeValue();
 64     return true;
 65 }

 68 FM.MlObserver.prototype.dispose = function() {
 69     var exts = FM.cloneObject(this.extensions);
 70     for(var i = 0; i < exts.length; i++) {
 71         var extObj = exts[i];
 72         if(FM.isset(extObj.dispose)) {
 73             extObj.dispose(this);
 74         }
 75     }
 76     this.extensions = [];
 77     
 78     if(this.node) {
 79         this.node.fmmlObserver = null;
 80     }
 81     if(this.host) {
 82         this.host.removeObserver(this);
 83     }
 84     
 85     if(this.errorObject) {
 86         this.errorObject.dispose();
 87         this.errorObject = null;
 88     }
 89         
 90     return this._super("dispose");
 91     this.executed = false;
 92 }

108 FM.MlObserver.prototype.setLastError = function(oErr) {
109     var errhost = this._getErrorHost();
110     if(!errhost) {
111         return oErr;
112     }
113     
114     if(!FM.isset(oErr) || !oErr || !FM.isObject(oErr)) {
115         oErr = new FM.DmGenericError();
116     }
117     
118     if(!errhost.isExecuted()) {
119         errhost.run(oErr);
120     } else {
121         var dmobj = errhost.getDmObject();
122         if(!dmobj) {
123             errhost.setDmObject(oErr);
124         } else {
125             dmobj.forEachAttr(function(attr,value) {
126                 dmobj.setAttr(attr,oErr.getAttr(attr,null));
127                 return true;
128             });   
129             dmobj.setChanged(true,true);
130             oErr = dmobj;
131         }
132     }    
133     
134     return oErr;
135 }

137 FM.MlObserver.prototype.isValid = function(force) {
138     var rules = this.getAttr("data-fmml-validation-rules",''); 
139     var response = true;
140     
141     if(rules != '') {        
142         var value = this.node ? 
143         (FM.isset(this.node.value) ? this.node.value : this.node.innerHTML) : ""
144         ;
145         force = FM.isset(force) ? force : false;
146 
147         if(force || value != "") {
148             var allRules = rules != null && rules != '' ? rules.split(";") : [];
149 
150             for (var i = 0; i < allRules.length; i++) {
151                 var invert = false;
152                 var rule = allRules[i];
153                 var ruleArr = rule != null && rule != '' ? rule.split("=") : [];
154                 var ruleOperator = ruleArr.length > 0 ? ruleArr[0] : '';
155                 var ruleParamStr = ruleArr.length > 1 ? ruleArr[1] : '';
156                 var ruleParams = ruleParamStr.split(",");
157                 if(FM.endsWith(ruleOperator,"!")) {
158                     ruleOperator = ruleOperator.substring(0,ruleOperator.length -1);
159                     invert = true;
160                 }
161                 var v = true;
162                 var fn = FM.MlObserver.getValidationRule(this.getApp(), ruleOperator);
163                 
164                 if(fn) {
165                     v = fn(this,ruleParams) == (invert ? false : true);
166                 }
167                 if(!v) {
168                     response = false;
169                     break;
170                 }
171             }
172         }
173     }
174     
175     if(response){
176         $(this.node).removeClass("fmmlInvalidValue");
177         this.setLastError(new FM.DmGenericError({
178             messageId: '',
179             text: ''
180         }));
181     } else {
182         $(this.node).addClass("fmmlInvalidValue"); 
183         this.setLastError(new FM.DmGenericError({
184             messageId: 'UIVALIDATION',
185             text: this.getAttr('data-fmml-validation-message','Invalid value')
186         }));
187     }
188     return response; //no rules
189 }

191 FM.MlObserver.prototype.update = function() {
192     if(!this.executed) return false;
193     
194     // sync node with dmobject
195     this.setNodeValue();
196     
197     // notify extensions
198     for(var i = 0; i < this.extensions.length; i++) {
199         var extObj = this.extensions[i];
200         if(FM.isset(extObj.update)) {
201             extObj.update(this);
202         }
203     }
204     
205     // check if obs is valid
206     if(this.isValid()) {
207         var hostToRun =  this.getAttr('data-fmml-run-on-update','');
208         if(hostToRun != '') {
209             var node = document.getElementById(hostToRun);
210             if( node && FM.isset(node.fmmlHost) && node.fmmlHost) {
211                 node.fmmlHost.run(this.getDmObject());
212             }
213         }
214         return true;
215     }
216     return false;
217 }

221     
222     // conf
223     var attrname = this.getAttr('data-fmml-attr-name','');
224     var host = this.getHost();

231

238 FM.MlObserver.prototype.getValue = function() {
239     // conf
240     var attrname = this.getAttr('data-fmml-attr-name','');
241     var defval = this.resolveAttrValue('data-fmml-attr-default-value','');
242     if(!this.executed) return defval;
243     
244     // value
245     var dmobj = this.getDmObject();
246     if(!dmobj) return defval;
247     
248     // get
249     return dmobj.getAttr(attrname,defval);
250 }

253 FM.MlObserver.prototype.getNodeValue = function() {
254     var value = '';
255     if(FM.isset(this.node.value)) {
256         value = this.node.value;
257     } else if(this.node.nodeName == 'IMG') {
258         value = this.node.getAttribute("src");
259     } else {
260         value = this.node.innerHTML;
261     }
262 
263     return value;
264 }

266 FM.MlObserver.prototype.setNodeValue = function() {
267     // conf
268     var attrname = this.getAttr('data-fmml-attr-name','');
269     if(attrname == '') return;
270     
271     if(FM.startsWith(this.getAttr('data-fmml-attr-default-value',''),'@')) {
272         var sss = 1;
273     }
274     var defval = this.resolveAttrValue('data-fmml-attr-default-value','');
275     
276     // value
277     var dmobj = this.getDmObject();
278     var value = 
279     dmobj && attrname != '' ?
280     dmobj.getAttr(attrname,'') :
281     defval
282     ;
283     
284     // not changed
285     if(value == this.lastValue) {
286         return;
287     }
288     
289     // set
290     this.lastValue = value;
291     if(FM.isset(this.node.value)) {
292         var selStart = this.node.selectionStart;
293         var selEnd = this.node.selectionEnd;
294         this.node.value = value;
295         if(FM.isset(this.node.setSelectionRange)) {
296             this.node.setSelectionRange(selStart, selEnd);
297         }
298     } else if(this.node.nodeName == 'IMG') {
299         this.node.setAttribute("src",value);
300     } else {
301         this.node.innerHTML = value;
302     }
303     this.node.fmmlValueSync = value;
304     
305 }

307 FM.MlObserver.prototype.addExtension = function(extObj) {
308     this.extensions.push(extObj);
309     if(this.executed) {
310         extObj.run(this);
311     }
312     return true;
313 }

315 FM.MlObserver.prototype.removeExtension = function(extObj) {
316     for(var i = 0; i < this.extensions.length; i++) {
317         if(extObj == this.extensions[i]) {
318             if(FM.isset(this.extensions[i].dispose)) {
319                 this.extensions[i].dispose(this);
320             }
321             delete this.extensions[i];
322             return true;
323         }
324     }
325     return false;
326 }

348 FM.MlObserver.prototype.onHostEvent = function(sender,ev,evdata) {
349     if(FM.isset(this[ev])) {
350         try {
351             this[ev](sender,evdata);
352             return true;
353         } catch(e) {
354             this.log(err,FM.logLevels.error,'MlObserver.onHostEvent');     
355             return true;
356         }        
357     } else {
358         var fnd = false;
359         
360         // notify extensions
361         for(var i = 0; i < this.extensions.length; i++) {
362             var extObj = this.extensions[i];
363             if(FM.isset(extObj[ev])) {
364                 try {
365                     extObj[ev](sender,evdata);
366                 } catch(e) {
367                     this.log(e,FM.logLevels.error,'MlObserver.onHostEvent');
368                 }
369                 fnd = true;
370             }
371         }
372         return fnd;
373     }
374 }

402 FM.MlObserver.addObserver = function(type,fn,appCls) {    
403     appCls = FM.isset(appCls) && FM.isString(appCls) && appCls != '' ? appCls : 'GLOBAL';
404     if(!FM.isset(fn) || !FM.isFunction(fn)) return false;
405     if(!FM.isset(type) || !type || type == '') return false;
406     if(!FM.isset(FM.MlObserver.observerTypes[appCls])) {
407         FM.MlObserver.observerTypes[appCls]= {};
408     }
409     
410     FM.MlObserver.observerTypes[appCls][type] = fn;
411     return true;
412 }

414 /**
415 * Returns MlObserver <b>config</b> class function for <b>config</b> subclass type
416 * @static
417 * @function    
418 * @param {object} app Application
419 * @param {String} name Configuration name
420 * @return {object} observer configuration or null if not found
421 */   
422 FM.MlObserver.getConfiguration = function(app,name) {
423     var list = FM.MlObserver.observerTypes;
424 
425     app = FM.isset(app) && app ? app : null;
426     var appCls = app ? app.getSubClassName() : null;
427     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
428         
429     var obj = null;
430     if(appCfg && FM.isset(appCfg[name])) {
431         obj = appCfg[name];
432     } else if(app && FM.isArray(app.applicationObjectsSpace)) {
433         FM.forEach(app.applicationObjectsSpace,function(i,ns) {
434             if(FM.isset(list[ns]) && FM.isset(list[ns][name])) {
435                 obj = list[ns][name];
436                 return false;
437             }
438             return true;
439         });
440     }
441     
442     if(!obj && FM.isset(list['GLOBAL'][name])) {
443         obj = list['GLOBAL'][name];
444     }
445     
446     return obj;
447 }

461             var value = observer.getValue();
462             if(ruleParams.length < 1) return false;
463         
464             for(var i = 0; i < ruleParams.length; i++) {
465                 if(FM.startsWith(ruleParams[i],'#')) {
466                     if(value != $(ruleParams[i]).val()) return false;
467                 } else {
468                     try {
469                         if(value != '' + eval(ruleParams[i])) {
470                             return false;
471                         }
472                     } catch(e) {                            
473                         return false;
474                     }
475                 }
476             }
477         			
478             return true;
479         },

481             var value = observer.getValue();
482             if(ruleParams.length < 1) return false;
483         
484             for(var i = 0; i < ruleParams.length; i++) {
485                 if(FM.startsWith(ruleParams[i],'#')) {
486                     if(value != $(ruleParams[i]).val()) return false;
487                 } else {
488                     try {
489                         if(parseFloat(value) > parseFloat(eval(ruleParams[i]))) {
490                             return true;
491                         }
492                     } catch(e) {                            
493                         return false;
494                     }
495                 }
496             }
497         			
498             return false;
499         },

502             if(ruleParams.length < 1) return false;
503

514                     }
515                 }
516             }
517         			
518             return false;
519         },
520         empty: function(observer, ruleParams,cbFn) {
521             var value = observer.getValue();
522             if(value==null || value == '') {
523                 return true;
524             }
525             return false;
526         },

528             var value = observer.getValue();

531             }
