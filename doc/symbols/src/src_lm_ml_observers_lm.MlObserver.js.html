 33 FM.MlObserver.prototype._init = function(app, attrs, node) {
 34     this._super("_init", app, attrs);
 35     this.objectSubClass = "Observer";
 36 
 37     this.log(attrs, FM.logLevels.debug, 'MlObserver._init');
 38 
 39     this.executed = false;
 40     this.node = node;
 41     
 42     this.node.fmmlObserver = this;
 43     this.lastValue = null;
 44 
 45     this.extensions = [];
 46     this.renderers = {};
 47     this.currentRenderer = null;
 48 
 49 
 50     // find error host
 51     this.errorObject = this.getAttr('data-fmml-error-host', '') != '' ?
 52         new FM.DmGenericError({
 53         id: '',
 54         text: ''
 55     }) :
 56         null
 57         ;
 58 
 59     this.log("New observer created.", FM.logLevels.debug, 'MlObserver._init');
 60 }

 62 FM.MlObserver.prototype.run = function() {
 63     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.run');
 64 
 65     this._super("run");
 66 
 67     this.log("Starting all registred extensions ...", FM.logLevels.debug, 'MlObserver.run');
 68     for (var i = 0; i < this.extensions.length; i++) {
 69         try {
 70             this.runExtension(this.extensions[i]);
 71         } catch (err) {
 72             this.log(err, FM.logLevels.error, 'MlObserver.run');
 73         }
 74     }
 75     this.log("Set DOM node value ...", FM.logLevels.debug, 'MlObserver.run');
 76     this.setNodeValue();
 77 
 78     this.log("New observer started.", FM.logLevels.debug, 'MlObserver.run');
 79     return true;
 80 }

 83 FM.MlObserver.prototype.dispose = function() {
 84     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.dispose');
 85 
 86     this.log("Disposing all registred extensions ...", FM.logLevels.debug, 'MlObserver.dispose');
 87     var exts = FM.cloneObject(this.extensions);
 88     for (var i = 0; i < exts.length; i++) {
 89         var extObj = exts[i];
 90         if (FM.isset(extObj.dispose)) {
 91             try {
 92                 extObj.dispose(this);
 93             } catch (err) {
 94                 this.log(err, FM.logLevels.error, 'MlObserver.dispose');
 95             }
 96         }
 97     }
 98     this.extensions = [];
 99 
100     this.log("Removing observer from DOM node ...", FM.logLevels.debug, 'MlObserver.dispose');
101     if (this.node) {
102         this.node.fmmlObserver = null;
103     }
104 
105     this.log("Removing observer from host ...", FM.logLevels.debug, 'MlObserver.dispose');
106     if (this.host) {
107         this.host.removeObserver(this);
108     }
109 
110     if (this.errorObject) {
111         this.log("Disposing error object ...", FM.logLevels.debug, 'MlObserver.dispose');
112         this.errorObject.dispose();
113         this.errorObject = null;
114     }
115 
116     return this._super("dispose");
117     this.log("New observer disposed.", FM.logLevels.debug, 'MlObserver.dispose');
118 }

134 FM.MlObserver.prototype.setLastError = function(oErr) {
135     var errhost = this._getErrorHost();
136     if (!errhost) {
137         return oErr;
138     }
139 
140     if (!FM.isset(oErr) || !oErr || !FM.isObject(oErr)) {
141         oErr = new FM.DmGenericError();
142     }
143 
144     if (!errhost.isExecuted()) {
145         errhost.run(oErr);
146     } else {
147         var dmobj = errhost.getDmObject();
148         if (!dmobj) {
149             errhost.setDmObject(oErr);
150         } else {
151             dmobj.forEachAttr(function(attr, value) {
152                 dmobj.setAttr(attr, oErr.getAttr(attr, null));
153                 return true;
154             });
155             dmobj.setChanged(true, true);
156             oErr = dmobj;
157         }
158     }
159 
160     return oErr;
161 }

163 FM.MlObserver.prototype.isValid = function(force) {
164     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.isValid');
165 
166     var rules = this.getAttr("data-fmml-validation-rules", '');
167     var response = true;
168     var value = this.getValue();
169 
170     if (rules != '') {
171         force = FM.isset(force) ? force : 
172             (
173                 this.getAttr('data-fmml-force-validation','false') == 'true' ? 
174                 true : false
175             );
176         
177         if (force || value != "") {
178             var allRules = rules != null && rules != '' ? rules.split(";") : [];
179 
180             for (var i = 0; i < allRules.length; i++) {
181                 var invert = false;
182                 var rule = allRules[i];
183                 var ruleArr = rule != null && rule != '' ? rule.split("=") : [];
184                 var ruleOperator = ruleArr.length > 0 ? ruleArr[0] : '';
185                 var ruleParamStr = ruleArr.length > 1 ? ruleArr[1] : '';
186                 var ruleParams = ruleParamStr.split(",");
187                 if (FM.endsWith(ruleOperator, "!")) {
188                     ruleOperator = ruleOperator.substring(0, ruleOperator.length - 1);
189                     invert = true;
190                 }
191                 var v = true;
192                 var fn = FM.MlObserver.getValidationRule(this.getApp(), ruleOperator);
193 
194                 if (fn) {
195                     v = fn(this, ruleParams) == (invert ? false : true);
196                 }
197                 if (!v) {
198                     response = false;
199                     break;
200                 }
201             }
202         }
203     }
204 
205     if (response) {
206         $(this.node).removeClass("fmmlInvalidValue");
207         this.setLastError(new FM.DmGenericError({
208             messageId: '',
209             text: ''
210         }));
211     } else {
212         $(this.node).addClass("fmmlInvalidValue");
213         this.setLastError(new FM.DmGenericError({
214             messageId: 'UIVALIDATION',
215             text: this.getAttr('data-fmml-validation-message', 'Invalid value')
216         }));
217     }
218 
219     this.log(
220         response ? "Observer is valid" : "Validation failed: " + ruleOperator,
221         FM.logLevels.debug, 'MlObserver.isValid'
222         );
223     return response; //no rules
224 }

228 FM.MlObserver.prototype.update = function() {
229     if (!this.isExecuted()) {
230         return false;
231     }
232     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.update');
233 
234     var dmObj = this.getDmObject();
235     var dtstmp = dmObj ? dmObj.getProperty('timestamp', '0') : '0';
236 
237     if (dtstmp != '0' && dtstmp == this.getProperty('updateTimestamp', '0')) {
238         this.log("Aborting, processed updateTimestamp.", FM.logLevels.debug, 'MlObserver.update');
239         return false;
240     }
241     this.setProperty('updateTimestamp', dtstmp);
242 
243     // notify extensions
244     this.log("Updating all extensions ...", FM.logLevels.debug, 'MlObserver.update');
245     for (var i = 0; i < this.extensions.length; i++) {
246         var extObj = this.extensions[i];
247         if (FM.isset(extObj.update)) {
248             try {
249                 extObj.update(this);
250             } catch (err) {
251                 this.log(err, FM.logLevels.error, 'MlObserver.update');
252             }
253         }
254     }
255 
256     // sync node with dmobject
257     this.log("Set DOM node value ...", FM.logLevels.debug, 'MlObserver.update');
258     this.setNodeValue();
259 
260     // check if obs is valid and run update host
261     var retc = false;
262     if (this.isValid()) {
263         var hostToRun = this.getAttr('data-fmml-run-on-update', '');
264         if (hostToRun != '') {
265             this.log("Running [" + hostToRun + "] on update ...", FM.logLevels.debug, 'MlObserver.update');
266             var node = document.getElementById(hostToRun);
267             if (node && FM.isset(node.fmmlHost) && node.fmmlHost) {
268                 try {
269                     node.fmmlHost.run(this.getDmObject());
270                 } catch (err) {
271                     this.log(err, FM.logLevels.error, 'MlObserver.update');
272                 }
273             } else {
274                 this.log("Host [" + hostToRun + "] not found", FM.logLevels.warn, 'MlObserver.update');
275             }
276         }
277         retc = true;
278     }
279 
280     this.log("Done.", FM.logLevels.warn, 'MlObserver.update');
281     return retc;
282 }

284 FM.MlObserver.prototype.setValue = function(value) {
285     if (!this.isExecuted()) {
286         return false;
287     }
288     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.setValue');
289 
290     // conf
291     var attrname = this.getAttr('data-fmml-attr-name', '');
292     var host = this.getHost();
293     this.log("Set Observer attribute [" + attrname + "] to [" + value + "] ...", FM.logLevels.debug, 'MlObserver.setValue');
294 
295     // value
296     var dmobj = this.getDmObject();
297     if (!dmobj) {
298         this.log("DmObject not found", FM.logLevels.warn, 'MlObserver.setValue');
299         return false;
300     }
301 
302     // set
303     value = this._formatValue(value);
304     dmobj.setAttr(attrname, value, true);
305 
306     // end
307     this.log("Done.", FM.logLevels.debug, 'MlObserver.setValue');
308     return true;

311 FM.MlObserver.prototype.getValue = function() {
312     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.getValue');
313 
314     if (!this.isExecuted()) {
315         this.log("Observer is not executed,returning undefined.", FM.logLevels.warn, 'MlObserver.getValue');
316         return undefined;
317     }
318 
319 
320     // conf
321     var attrname = this.getAttr('data-fmml-attr-name', '');
322     var defval = this.resolveAttrValue('data-fmml-attr-default-value', '');
323     var dmobj = this.getDmObject();
324 
325     // value
326     var value = FM.resolveAttrName(dmobj ? dmobj.options : {}, attrname, defval, {
327         A: this.getApp(),
328         H: this.getHost(),
329         O: this,
330         D: this.getDmObject()
331     });
332 
333 
334     // end
335     this.log("Done.", FM.logLevels.debug, 'MlObserver.getValue');
336     return value;
337 }

339 FM.MlObserver.prototype._formatValue = function(value) {
340     var attrtype = this.getAttr('data-fmml-attr-type', 'string');
341     var decplaces = parseInt(this.getAttr('data-fmml-attr-decimals', '-1'));
342     var dateIsUtc = this.getAttr('data-fmml-date-is-utc', 'true') != 'false';
343     var dateFormat = this.getAttr('data-fmml-date-format', 
344         this.getApp().getAttr('fm_templates_date_format',undefined))
345     ;
346 
347 // dates
348     if (attrtype == "date") {
349         var dateObj = null;
350         if(FM.isObject(value) && FM.isset(value.getTime)) {
351             dateObj = value;
352         } else if (FM.isDateString(value)) {
353             dateObj = FM.parseDateString(value, dateIsUtc);
354         } else {
355             dateObj = FM.parseLocalDateString(value);
356         }
357 
358         if (dateObj) {
359             value = FM.dateToString(dateObj, dateIsUtc,dateFormat);
360         }
361     } else if (attrtype == "number") {
362         value = parseFloat(0.0 + value);
363         if (decplaces > -1) {
364             value = value.toFixed(decplaces);
365         }
366     }
367 
368     return value;
369 }

371 FM.MlObserver.prototype._formatValueForRendering = function(value) {
372     var attrtype = this.getAttr('data-fmml-attr-type', 'string');
373     var dateIsUtc = this.getAttr('data-fmml-date-is-utc', 'true') != 'false';
374     var dateFormat = this.getAttr(
375         'data-fmml-date-display-as', 
376         this.getApp().getAttr('fm_templates_date_display_as',undefined)
377     );
378 
379     var decplaces = parseInt(
380         this.getAttr(
381         'data-fmml-nuber-display-decimals',
382         this.getAttr('data-fmml-attr-decimals', '-1')
383         )
384         );
385     // var attrtemplate = this.getAttr('data-fmml-attr-template', '');
386 
387 // dates
388     if (attrtype == "date") {
389         var dateObj = null;
390 
391         if (FM.isDateString(value)) {
392             dateObj = FM.parseDateString(value, dateIsUtc);
393         } else {
394             dateObj = FM.parseLocalDateString(value);
395         }
396 
397         if (dateObj) {
398             if (dateFormat == 'local') {
399                 value = FM.dateLocalFormat(dateObj);
400             } else if (dateFormat == 'ago') {
401                 value = FM.strTimeBetween(dateObj, new Date());
402             } else {
403                 value = FM.dateFormat(dateObj, dateFormat);
404             }
405         }
406     } else if (attrtype == "number") {
407         value = parseFloat(0.0 + value);
408         if (decplaces > -1) {
409             value = value.toFixed(decplaces);
410         }
411     }
412 
413     return value;
414 }

416 FM.MlObserver.prototype.setNodeValue = function(force) {
417     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.setNodeValue');
418     force = FM.isset(force) && force == true ? true : false;
419     
420     // get value
421     var nfvalue = this.getValue();
422 
423     // formating
424     this.log("Formating value [" + nfvalue + "]...", FM.logLevels.debug, 'MlObserver.setNodeValue');
425     var value = this._formatValueForRendering(nfvalue);
426 
427     this.log("Formated value is [" + value + "].", FM.logLevels.debug, 'MlObserver.setNodeValue');
428 
429     // not changed
430     if (!force && value == this.lastValue) {
431         this.log("Aborting, formated value is not changed.", FM.logLevels.debug, 'MlObserver.setNodeValue');
432         return;
433     }
434 
435     // remember
436     this.lastValue = value;
437 
438     // render
439     this.log("Rendering value ...", FM.logLevels.debug, 'MlObserver.setNodeValue');
440     if (this.getCurrentRenderer()) {
441         this.getCurrentRenderer().render(value);
442         return;
443     }
444 
445     // def render
446     var doSelection =false;
447     try {
448         doSelection = this.node.selectionStart ? true : false;
449     } catch(e){
450         
451     }
452     var selStart = 0,selEnd = 0;
453     if (doSelection) {
454         selStart = this.node.selectionStart;
455         selEnd = this.node.selectionEnd;
456     }
457     
458     if (this.node.nodeName == 'INPUT' || this.node.nodeName == 'TEXTAREA') {
459         if ($(this.node).is(':checkbox')) {
460             if (value && value != '' && value.toLowerCase() != 'false') {
461                 $(this.node).attr('checked', 'checked');
462             } else {
463                 $(this.node).removeAttr('checked');
464             }
465         } else if ($(this.node).is(':radio')) {
466             $("input:radio[name ='" + $(this.node).attr("name") + "']").val([value]);
467         } else {
468             $(this.node).val(value);
469         }
470     } else if (this.node.nodeName == 'IMG') {
471         $(this.node).attr("src",value);
472     } else if (this.node.nodeName == 'A') {
473         $(this.node).attr("href",value);
474     } else if (FM.isset(this.node.value)) {
475         $(this.node).val(value);
476     } else {
477         $(this.node).html(value);
478     }
479 
480     // selection range restore
481     if (doSelection) {
482         this.node.setSelectionRange(selStart, selEnd);
483     }
484 
485     //end
486     this.log("Done.", FM.logLevels.debug, 'MlObserver.setNodeValue');
487 }

489 FM.MlObserver.prototype.addExtension = function(extObj) {
490     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.addExtension');
491     this.log("Adding extension:", FM.logLevels.debug, 'MlObserver.addExtension');
492     this.log(extObj, FM.logLevels.debug, 'MlObserver.addExtension');
493

495     if (this.isExecuted()) {
496         this.log("Running added extension ...", FM.logLevels.debug, 'MlObserver.addExtension');
497         try {
498             this.runExtension(extObj);
499         } catch (err) {
500             this.log(err, FM.logLevels.error, 'MlObserver.addExtension');
501         }
502     }
503     this.log("Done.", FM.logLevels.debug, 'MlObserver.addExtension');

507 FM.MlObserver.prototype.removeExtension = function(extObj) {
508     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.removeExtension');
509     this.log("Removing extension:", FM.logLevels.debug, 'MlObserver.removeExtension');
510     this.log(extObj, FM.logLevels.debug, 'MlObserver.removeExtension');
511 
512     for (var i = 0; i < this.extensions.length; i++) {
513         if (extObj == this.extensions[i]) {
514             if (FM.isset(this.extensions[i].dispose)) {
515                 this.extensions[i].dispose(this);
516             }
517             delete this.extensions[i];
518             this.log("Done.", FM.logLevels.debug, 'MlObserver.removeExtension');
519             return true;
520         }
521     }
522     this.log("Not found.", FM.logLevels.warn, 'MlObserver.removeExtension');
523     return false;
524 }

547 FM.MlObserver.prototype.onHostEvent = function(sender, ev, evdata) {
548     this.log(this.getNode(), FM.logLevels.debug, 'MlObserver.onHostEvent');
549     this.log("Event:" + ev, FM.logLevels.debug, 'MlObserver.onHostEvent');
550     this.log("Event data:", FM.logLevels.debug, 'MlObserver.onHostEvent');
551     this.log(evdata, FM.logLevels.debug, 'MlObserver.onHostEvent');
552 
553     var fnd = false;
554     if (FM.isset(this[ev])) {
555         fnd = true;
556         try {
557             this.log("Event is found, executing ...", FM.logLevels.debug, 'MlObserver.onHostEvent');
558             this[ev](sender, evdata);
559         } catch (e) {
560             this.log(err, FM.logLevels.error, 'MlObserver.onHostEvent');
561         }
562     } else {
563         this.log("Event is not found, checking extensions ...", FM.logLevels.debug, 'MlObserver.onHostEvent');
564 
565         // notify extensions
566         for (var i = 0; i < this.extensions.length; i++) {
567             var extObj = this.extensions[i];
568             if (FM.isset(extObj[ev])) {
569                 try {
570                     this.log("Executing event in extension:", FM.logLevels.debug, 'MlObserver.onHostEvent');
571                     this.log(extObj, FM.logLevels.debug, 'MlObserver.onHostEvent');
572                     extObj[ev](sender, evdata);
573                 } catch (e) {
574                     this.log(e, FM.logLevels.error, 'MlObserver.onHostEvent');
575                 }
576                 fnd = true;
577             }
578         }
579     }
580 
581     this.log("Done.", FM.logLevels.debug, 'MlObserver.onHostEvent');
582     return fnd;
583 }

604 FM.MlObserver.prototype._checkCurrentRenderer = function() {
605     for (var i = this.extensions.length - 1; i > -1; i--) {
606         var ext = this.extensions[i];
607         if (FM.isset(this.renderers[ext.getID()] && this.renderers[ext.getID()])) {
608             this.currentRenderer = ext;
609             return this.currentRenderer;
610         }
611     }
612 
613     return this.currentRenderer;
614 }

616 FM.MlObserver.prototype.addRenderer = function(r) {
617     if (!r)
618         return;
619 
620     var curr = this.getCurrentRenderer();
621     this.renderers[r.getID()] = r;
622     var newr = this._checkCurrentRenderer();
623     if (curr != newr) {
624         if (curr)
625             curr.disableRenderer();
626         if (newr)
627             newr.enableRenderer();
628     }
629 }
630 
631 FM.MlObserver.prototype.removeRenderer = function(r) {
632     if (!r)
633         return;
634 
635     var curr = this.getCurrentRenderer();
636     if (FM.isset(this.renderers[r.getID()])) {
637         this.renderers[r.getID()] = null;
638     }
639     var newr = this._checkCurrentRenderer();
640     if (curr != newr) {
641         if (curr)
642             curr.disableRenderer();
643         if (newr)
644             newr.enableRenderer();
645     }
646 }

662 FM.MlObserver.addObserver = function(type, fn, appCls) {
663     appCls = FM.isset(appCls) && FM.isString(appCls) && appCls != '' ? appCls : 'GLOBAL';
664     if (!FM.isset(fn) || !FM.isFunction(fn))
665         return false;
666     if (!FM.isset(type) || !type || type == '')
667         return false;
668     if (!FM.isset(FM.MlObserver.observerTypes[appCls])) {
669         FM.MlObserver.observerTypes[appCls] = {};
670     }
671 
672     FM.MlObserver.observerTypes[appCls][type] = fn;
673     return true;
674 }

676 /**
677  * Returns MlObserver <b>config</b> class function for <b>config</b> subclass type
678  * @static
679  * @function    
680  * @param {object} app Application
681  * @param {String} name Configuration name
682  * @return {object} observer configuration or null if not found
683  */
684 FM.MlObserver.getConfiguration = function(app, name) {
685     var list = FM.MlObserver.observerTypes;
686 
687     app = FM.isset(app) && app ? app : null;
688     var appCls = app ? app.getSubClassName() : null;
689     var appCfg = appCls && FM.isset(list[appCls]) ? list[appCls] : null;
690 
691     var obj = null;
692     if (appCfg && FM.isset(appCfg[name])) {
693         obj = appCfg[name];
694     } else if (app && FM.isArray(app.applicationObjectsSpace)) {
695         FM.forEach(app.applicationObjectsSpace, function(i, ns) {
696             if (FM.isset(list[ns]) && FM.isset(list[ns][name])) {
697                 obj = list[ns][name];
698                 return false;
699             }
700             return true;
701         });
702     }
703 
704     if (!obj && FM.isset(list['GLOBAL'][name])) {
705         obj = list['GLOBAL'][name];
706     }
707 
708     return obj;
709 }

712     var clsFn = FM.MlObserver.getConfiguration(app, type);
713     return clsFn ? new clsFn(app, attrs, node) : null;
714 }

724         equal: function(observer, ruleParams, cbFn) {
725             var value = observer.getValue();
726             if (ruleParams.length < 1)
727                 return false;
728 
729             for (var i = 0; i < ruleParams.length; i++) {
730                 if (FM.startsWith(ruleParams[i], '#')) {
731                     if (value != $(ruleParams[i]).val())
732                         return false;
733                 } else {
734                     try {
735                         if (value != '' + eval(ruleParams[i])) {
736                             return false;
737                         }
738                     } catch (e) {
739                         return false;
740                     }
741                 }
742             }
743 
744             return true;
745         },
746         gt: function(observer, ruleParams, cbFn) {
747             var value = observer.getValue();
748             if (ruleParams.length < 1)
749                 return false;
750 
751             for (var i = 0; i < ruleParams.length; i++) {
752                 if (FM.startsWith(ruleParams[i], '#')) {
753                     if (value != $(ruleParams[i]).val())
754                         return false;
755                 } else {
756                     try {
757                         if (parseFloat(value) > parseFloat(eval(ruleParams[i]))) {
758                             return true;
759                         }
760                     } catch (e) {
761                         return false;
762                     }
763                 }
764             }
765 
766             return false;

780                             return true;

784                     }
785                 }
786             }
787 
788             return false;
789         },
790         empty: function(observer, ruleParams, cbFn) {
791             var value = observer.getValue();
792             if (value == null || value == '') {
793                 return true;
794             }
795             return false;
