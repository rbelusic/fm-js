 72 /**
 73  * Set cookie.
 74  * 
 75  * @static
 76  * @function 
 77  * @param {string} name Name of the cookie.
 78  * @param {...} value Value of the cookie.
 79  * @param {number} [expiredays=3650] The time the cookie expires. 
 80  * @param {string} [domain] Cookie domain.
 81  * 
 82  */
 83 FM.saveCookie = function(name, value, expiredays, domain) {
 84     var daysahead, expires = null;
 85 
 86     if (FM.isset(expiredays)) {
 87         daysahead = parseInt(expiredays);
 88         if (daysahead <= 0)
 89             daysahead = 3650; // 10 godina = zauvijek
 90     } else {
 91         daysahead = 3650; // 10 godina = zauvijek
 92     }
 93     expires = new Date();
 94     expires.setDate(expires.getDate() + daysahead);
 95 
 96     if (!FM.isset(value) || value == null)
 97         value = {};
 98 
 99     document.cookie =
100         name + "=" +
101         escape(FM.isString(value) ? value : FM.arrayToUrl(value)) +
102         (FM.isset(domain) && domain ? ";domain=" + domain : "") +
103         "; path=/" +
104         ((expires == null) ? "" : "; expires=" + expires.toGMTString())
105         ;
106 
107     return document.cookie;
108 }

111 /**
112  * Get cookie value.
113  * 
114  * @static
115  * @function 
116  * @param {string} name Name of the cookie.
117  * @param {boolean} [asstring=false] Don't deserialize cookie value.
118  * @param {string} [domain] Cookie domain.
119  * 
120  */
121 FM.loadCookie = function(name, asstring) {
122     var dc = document.cookie;
123     var cname = name + "=";
124     var cbegin, cend, retstr = "";
125 
126     asstring = FM.isset(asstring) ? asstring == true : false;
127 
128     if (dc.length > 0) {
129         cbegin = dc.indexOf(cname);
130         if (cbegin != -1) {
131             cbegin += cname.length;
132             cend = dc.indexOf(";", cbegin);
133             if (cend == -1)
134                 cend = dc.length;
135             retstr = unescape(dc.substring(cbegin, cend));
136         }
137     }
138 
139     return asstring ? retstr : FM.urlToArray(retstr);
140 }

142 /**
143  * HTML escape string.
144  * 
145  * @static
146  * @function 
147  * @param {string} str String to escape.
148  * @returns {string} Escaped string.
149  * 
150  */
151 FM.escapeStr = function(str) {
152     str = str.replace(/&/g, "&");
153     str = str.replace(/>/g, ">");
154     str = str.replace(/</g, "<");
155     str = str.replace(/"/g, """);
156     str = str.replace(/'/g, "'");
157     return str;
158 }

178 /**
179  * URL encode string.
180  * 
181  * @static
182  * @function 
183  * @param {string} s String to encode.
184  * @returns {string} Encoded string.
185  * 
186  */
187 FM.urlEncode = function(s) {
188     return encodeURIComponent(s).replace(/\%20/g, '+').replace(/!/g, '%21').replace(/'/g, '%27').replace(/\(/g, '%28').replace(/\)/g, '%29').replace(/\*/g, '%2A').replace(/\~/g, '%7E');
189 }

191 /**
192  * URL decode string.
193  * 
194  * @static
195  * @function 
196  * @param {string} s String to decode.
197  * @returns {string} Decoded string.
198  * 
199  */
200 FM.urlDecode = function(s) {
201     return decodeURIComponent(s.replace(/\+/g, '%20').replace(/\%21/g, '!').replace(/\%27/g, "'").replace(/\%28/g, '(').replace(/\%29/g, ')').replace(/\%2A/g, '*').replace(/\%7E/g, '~'));
202 }

204 /**
205  * Encode all array elements or object properties as URL query arguments.
206  * 
207  * @static
208  * @function 
209  * @param {Array|Object} params Array or object to encode.
210  * @returns {string} Encoded string.
211  * 
212  */
213 FM.arrayToUrl = function(params) {
214     var ret = "";
215     var first = true;
216     for (var vname in params) {
217         if (first != true)
218             ret = ret + '&';
219         ret = ret + vname + '=' + FM.urlEncode(params[vname]);
220         first = false;
221     }
222 
223     return ret;
224 }

226 /**
227  * Decode URL encoded parameters to object properties.
228  * 
229  * @static
230  * @function 
231  * @param {string} url.
232  * @returns {Object}
233  * 
234  */
235 FM.urlToArray = function(url) {
236     //location.queryString = {};
237     var arr = {};
238 
239     var pairs = url.split("&");
240 
241     for (var i = 0; i < pairs.length; i++) {
242         var keyval = pairs[ i ].split("=");
243         arr[ keyval[0] ] = FM.isset(keyval[1]) ? FM.urlDecode(keyval[1]) : '';
244     }
245 
246     return arr;
247 }

251  * 
252  * @static
253  * @function 
254  * @param {string} s The string to be checked.
255  * @returns {boolean} 
256  */
257 FM.isURL = function(s) {
258     var regexp = /(ftp|http|https):\/\/(\w+:{0,1}\w*@)?(\S+)(:[0-9]+)?(\/|\/([\w#!:.?+=&%@!\-\/]))?/
259     return regexp.test(s);
260 }

329 /**
330  * Return requested page information, or object containing all informations.
331  * 
332  * @static
333  * @function 
334  * @param {string} [attr] The name of attrribute to return.
335  * @param {string} [def] Default return value.
336  * @returns {string|FM.PageInfo} 
337  */
338 FM.getPageInfo = function(attr,def) {
339 
340     var hash = FM.isset(window.location.hash) && window.location.hash ?
341         window.location.hash : ''
342     ;
343     if(FM.startsWith(hash, "#")) {
344         hash = hash.substring(1);
345     }
346 
347     var pnamearr = window.location.pathname.split("/");
348     var fullname = pnamearr.length > 0 ? pnamearr[pnamearr.length - 1] : window.location.pathname;
349     
350     var i = fullname.lastIndexOf(".");
351     var name = i > -1 ?
352         fullname.substring(0, i) :
353         fullname
354     ;
355 
356     var pi = {
357         query: window.location.search != '' ? window.location.search.substring(1) : '',
358         protocol: FM.getAttr(window.location,'protocol',''),
359         name: name,
360         fullname: fullname,
361         host: window.location.host != '' ?
362             window.location.protocol + "//" + window.location.host +
363             (window.location.port == '' ? '' : ":" + window.location.port) : '',
364         url: window.location.href,
365         path: window.location.pathname,
366         hash: hash
367     };
368 
369     return(FM.isset(attr) && attr ? FM.getAttr(pi,attr,def) : pi);
370 }

372 /**
373  * Return requested query parameter value, or object containing all query name - value pairs.
374  * 
375  * @static
376  * @function 
377  * @param {string} [attr] The name of attrribute to return.
378  * @param {string} [def] Default return value.
379  * @returns {...|Object} 
380  */
381 FM.getPageArgs = function(attr, def) {
382     var query_string = {};
383     var query = window.location.search.substring(1);
384     var vars = query.split("&");
385     for (var i = 0; i < vars.length; i++) {
386         var pair = vars[i].split("=");
387         pair[0] = FM.urlDecode(pair[0]);
388         pair[1] = FM.isset(pair[1]) ? FM.urlDecode(pair[1]) : '';
389 
390         // If first entry with this name
391         if (typeof query_string[pair[0]] === "undefined") {
392             query_string[pair[0]] = pair[1];
393             // If second entry with this name
394         } else if (typeof query_string[pair[0]] === "string") {
395             var arr = [query_string[pair[0]], pair[1]];
396             query_string[pair[0]] = arr;
397             // If third or later entry with this name
398         } else {
399             query_string[pair[0]].push(pair[1]);
400         }
401     }
402    
403     return FM.isset(attr) && attr ? FM.getAttr(query_string, attr, def) : query_string;
404 }

406 /**
407  * Return combined result of FM.getPageArgs and FM.getPageInfo functions. 
408  * FM.PageInfo is embeded as <i>_page</i> property in result of FM.getPageArgs function call.
409  * <b>This function is deprecated. Please use FM.getPageArgs and FM.getPageInfo functions.</b>
410  * 
411  * 
412  * @deprecated
413 
414  * @static
415  * @function 
416  * @see FM.getPageArgs
417  * @see FM.getPageInfo
418  * @returns {...|Object} 
419  */
420 FM.getArgs = function(attr, def) {
421     var pi = FM.getPageInfo();
422     var args = FM.getPageArgs();
423     args._page = pi;
424     
425     return(FM.isset(attr) && attr ? FM.getAttr(args,attr,def) : args);    
426 }
