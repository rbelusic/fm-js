 71  * @ignore
 72  *
 73  */
 74 FM.saveCookie = function(name, value, expiredays, domain) {
 75     var daysahead, expires = null;
 76 
 77     if (FM.isset(expiredays)) {
 78         daysahead = parseInt(expiredays);
 79         if (daysahead <= 0)
 80             daysahead = 3650; // 10 godina = zauvijek
 81     } else {
 82         daysahead = 3650; // 10 godina = zauvijek
 83     }
 84     expires = new Date();
 85     expires.setDate(expires.getDate() + daysahead);
 86 
 87     if (!FM.isset(value) || value == null)
 88         value = {};
 89 
 90     document.cookie =
 91         name + "=" +
 92         escape(FM.isString(value) ? value : FM.arrayToUrl(value)) +
 93         (FM.isset(domain) && domain ? ";domain=" + domain : "") +
 94         "; path=/" +
 95         ((expires == null) ? "" : "; expires=" + expires.toGMTString())
 96         ;
 97 
 98     return document.cookie;
 99 }

106 FM.loadCookie = function(name, asstring) {
107     var dc = document.cookie;
108     var cname = name + "=";
109     var cbegin, cend, retstr = "";
110 
111     asstring = FM.isset(asstring) ? asstring == true : false;
112 
113     if (dc.length > 0) {
114         cbegin = dc.indexOf(cname);
115         if (cbegin != -1) {
116             cbegin += cname.length;
117             cend = dc.indexOf(";", cbegin);
118             if (cend == -1)
119                 cend = dc.length;
120             retstr = unescape(dc.substring(cbegin, cend));
121         }
122     }
123 
124     return asstring ? retstr : FM.urlToArray(retstr);
125 }

127 /**
128  * HTML escape string.
129  * 
130  * @static
131  * @function 
132  * @param {string} str String to escape.
133  * @returns {string} Escaped string.
134  * 
135  */
136 FM.escapeStr = function(str) {
137     str = str.replace(/&/g, "&");
138     str = str.replace(/>/g, ">");
139     str = str.replace(/</g, "<");
140     str = str.replace(/"/g, """);
141     str = str.replace(/'/g, "'");
142     return str;
143 }

163 /**
164  * URL encode string.
165  * 
166  * @static
167  * @function 
168  * @param {string} s String to encode.
169  * @returns {string} Encoded string.
170  * 
171  */
172 FM.urlEncode = function(s) {
173     return encodeURIComponent(s).replace(/\%20/g, '+').replace(/!/g, '%21').replace(/'/g, '%27').replace(/\(/g, '%28').replace(/\)/g, '%29').replace(/\*/g, '%2A').replace(/\~/g, '%7E');
174 }

176 /**
177  * URL decode string.
178  * 
179  * @static
180  * @function 
181  * @param {string} s String to decode.
182  * @returns {string} Decoded string.
183  * 
184  */
185 FM.urlDecode = function(s) {
186     return decodeURIComponent(s.replace(/\+/g, '%20').replace(/\%21/g, '!').replace(/\%27/g, "'").replace(/\%28/g, '(').replace(/\%29/g, ')').replace(/\%2A/g, '*').replace(/\%7E/g, '~'));
187 }

189 /**
190  * Encode all array elements or object properties as URL query arguments.
191  * 
192  * @static
193  * @function 
194  * @param {Array|Object} params Array or object to encode.
195  * @returns {string} Encoded string.
196  * 
197  */
198 FM.arrayToUrl = function(params) {
199     var ret = "";
200     var first = true;
201     for (var vname in params) {
202         if (first != true)
203             ret = ret + '&';
204         ret = ret + vname + '=' + FM.urlEncode(params[vname]);
205         first = false;
206     }
207 
208     return ret;
209 }

211 /**
212  * Decode URL encoded parameters to object properties.
213  * 
214  * @static
215  * @function 
216  * @param {string} url.
217  * @returns {Object}
218  * 
219  */
220 FM.urlToArray = function(url) {
221     //location.queryString = {};
222     var arr = {};
223 
224     var pairs = url.split("&");
225 
226     for (var i = 0; i < pairs.length; i++) {
227         var keyval = pairs[ i ].split("=");
228         arr[ keyval[0] ] = FM.isset(keyval[1]) ? FM.urlDecode(keyval[1]) : '';
229     }
230 
231     return arr;
232 }

235  * Determine if a string is URL.
236  * 
237  * @static
238  * @function 
239  * @param {string} s The string to be checked.
240  * @returns {boolean} 
241  */
242 FM.isURL = function(s) {
243     var regexp = /(ftp|http|https):\/\/(\w+:{0,1}\w*@)?(\S+)(:[0-9]+)?(\/|\/([\w#!:.?+=&%@!\-\/]))?/
244     return regexp.test(s);
245 }

248 /**
249  * Return chosen page information 
250  * or object representing page info and all URL query parameters as name,value pairs.
251  * {
252  *   _page: {
253  *     host: "...",
254  *     url: "...",
255  *     path: "...",
256  *     name: "..."
257  *     fullname: "..."
258  *   },
259  *   arg1: v1,
260  *   arg2: v2,
261  *   ...
262  * }
263  * @static
264  * @function 
265  * @param {string} [attr] The name of attrribute to return.
266  * @param {string} [def] Default return value.
267  * @returns {string} 
268  */
269 FM.getArgs = function(attr, def) {
270     var query_string = {};
271     var query = window.location.search.substring(1);
272     var vars = query.split("&");
273     for (var i = 0; i < vars.length; i++) {
274         var pair = vars[i].split("=");
275         pair[0] = FM.urlDecode(pair[0]);
276         pair[1] = FM.isset(pair[1]) ? FM.urlDecode(pair[1]) : '';
277 
278         // If first entry with this name
279         if (typeof query_string[pair[0]] === "undefined") {
280             query_string[pair[0]] = pair[1];
281             // If second entry with this name
282         } else if (typeof query_string[pair[0]] === "string") {
283             var arr = [query_string[pair[0]], pair[1]];
284             query_string[pair[0]] = arr;
285             // If third or later entry with this name
286         } else {
287             query_string[pair[0]].push(pair[1]);
288         }
289     }
290 
291     var args = query_string;
292     var hash = FM.isset(window.location.hash) && window.location.hash ?
293         window.location.hash : '';
294     if(FM.startsWith(hash, "#")) {
295         hash = hash.substring(1);
296     }
297         
298     args._page = {
299         host: window.location.protocol + "//" + window.location.host +
300             (window.location.port == '' ? '' : ":" + window.location.port),
301         url: window.location.href,
302         path: window.location.pathname,
303         hash: hash
304     }
305 
306     var pnamearr = args._page.path.split("/");
307     args._page.fullname = pnamearr.length > 0 ? pnamearr[pnamearr.length - 1] : args._page.path;
308     var i = args._page.fullname.lastIndexOf(".");
309     args._page.name = i > -1 ?
310         args._page.fullname.substring(0, i) :
311         args._page.fullname
312         ;
313 
314     return FM.isset(attr) && attr ? FM.getAttr(args, attr, def) : args;
315 }
