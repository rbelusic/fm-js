1 /**
  2 * List index ML observer class for {@link FM.MlHostGenericList} host. 
  3 * 
  4 * @class FM.MlObserverListIndex
  5 * @extends FM.MlObserver
  6 * @param {object} [attrs] DOM node attributes
  7 * @param {DOMnode} node DOM node
  8 */ 
  9 FM.MlObserverListIndex = FM.defineClass('MlObserverListIndex',FM.MlObserver);
 10 
 11 FM.MlObserverListIndex.prototype._init = function(app,attrs,node) {            
 12     this._super("_init",app,attrs,node);
 13     this.objectSubClass ="ListIndex";
 14 }        
 15 
 16 FM.MlObserverListIndex.prototype.setNodeValue = function() {
 17     var me = this;
 18     var value = '';
 19     
 20     // get dmobj && dmlist
 21     var dmobj = this.getDmObject();
 22     var host = this.getHost();
 23     
 24     if(host) value = host.getAttr('data-fmml-list-index','');
 25     if(host && value == '') {    
 26         var dmlist = host ? host.getDmObject(null) : null;
 27 
 28         if(
 29             dmobj && dmlist && 
 30             host && FM.isset(host.cursorStartIndex) && 
 31             FM.isset(this.node) && FM.isset(this.node.fmmlObserverListIndex)
 32         ) {
 33             value = this.node.fmmlObserverListIndex + host.cursorStartIndex + 1;
 34             value = '' + value;
 35         }
 36     }
 37     
 38     if(FM.isset(this.node.value)) {
 39         this.node.value = value;
 40     } else if(me.node.nodeName == 'IMG') {
 41         me.node.setAttribute("src",value);
 42     } else {
 43         me.node.innerHTML = value;
 44     }
 45 }
 46 
 47 FM.MlObserver.addObserver("ListIndex",FM.MlObserverListIndex,'GLOBAL');
