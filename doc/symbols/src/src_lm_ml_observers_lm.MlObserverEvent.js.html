1 /**
  2 * Event ML observer class. Sends event to host.
  3 * 
  4 * Applicable on HTML nodes with <click> event.
  5 * 
  6 * <table>
  7 * <th>List of ML node attributes</th>
  8 * <tr><td>data-fmml-dom-event</td><td>DOM event which trigger FM event. Default is <b>click</b> event</td></tr>
  9 * <tr><td>data-fmml-event-type</td><td>type (name) of event. In case of event type is not defined, successful action processing is triggered immediately</td></tr>
 10 * <tr><td>data-fmml-event-host</td><td>DOM node id of host which will receive event instead of default host</td></tr>
 11 * <tr><td>data-fmml-event-data</td><td>instead current host dmObject send FM.GenericValue object with this value</td></tr>
 12 * <tr><td>data-fmml-verify-observers</td><td>verify all observers first</td></tr>
 13 * <tr><td>data-fmml-event-async</td><td>Signals that event is async (AJAX request wil be triggered)</td></tr>
 14 * <tr><td>data-fmml-exec-on-success</td><td>code to exec when event is successfully completed</td></tr>
 15 * <tr><td>data-fmml-exec-on-error</td><td>code to exec when event error occurred</td></tr>
 16 * <tr><td>data-fmml-redirect-on-success</td><td>redirect to specified URL when event is successfully completed</td></tr>
 17 * <tr><td>data-fmml-redirect-on-error</td><td>redirect to specified URL when event error occurred</td></tr>
 18 * <tr><td>data-fmml-run-on-success</td><td>run host on DOM node with specified id when event is successfully completed</td></tr>
 19 * <tr><td>data-fmml-run-on-error</td><td>run host on DOM node with specified id when event error occurred</td></tr>
 20 * <tr><td>data-fmml-send-form</td><td>send form on host node to URL specified in data-fmml-redirect-on-success node attribute. Host node type must be FORM</td></tr>
 21 * </table>
 22 *
 23 * On case of multiple actions defined on succes or on error, all of them will be executed in following order: 
 24 *  data-fmml-exec-on-*, data-fmml-run-on-*, data-fmml-redirect-on-*
 25 * 
 26 * <table>
 27 * <th>List of ML CSS classes</th>
 28 * <tr><td>fmmlWaitButton</td><td>Async event is in progress</td></tr>
 29 * </table>
 30 * 
 31 * @class FM.MlObserverEvent
 32 * @extends FM.MlObserver
 33 * @param {object} [attrs] DOM node attributes
 34 * @param {DOMnode} node DOM node
 35 */    
 36 FM.MlObserverEvent = FM.defineClass('MlObserverEvent',FM.MlObserver);
 37 
 38 FM.MlObserverEvent.prototype._init = function(app,attrs,node) {
 39     this._super("_init",app,attrs,node);
 40     this.objectSubClass ="Event";
 41 
 42     this.eventHost = null;
 43     this.eventHostNode = null;
 44 }
 45 
 46 
 47 FM.MlObserverEvent.prototype.run = function() {
 48     this._super("run");
 49     
 50     var me = this;
 51     
 52     // dom event
 53     var evtrigger = this.getAttr('data-fmml-dom-event','click');
 54 
 55     $(this.node)[evtrigger](function(event) {
 56         event.preventDefault();
 57         
 58         // blur form change processing (bad, bad, bad)
 59         if(document.activeElement != this) { // this -> ev node
 60             document.activeElement.blur();
 61         }
 62         
 63         // my host & host node
 64         var myhost = me.getHost();
 65         var myhostnode = myhost ? myhost.getNode() : null;
 66         
 67         // obs verify
 68         if(myhost && me.getAttr('data-fmml-verify-observers','false') == 'true') {
 69             if(!myhost.verifyAllObservers(true)) {
 70                 me.setLastError(new FM.DmGenericError({
 71                     messageId: 'UIVALIDATION',
 72                     text: 'Please enter all required fields'
 73                 }));
 74                 return false; // force validation of empty attributes
 75             }
 76         }
 77         
 78         // event host
 79         me.eventHost = null;
 80         me.eventHostNode = null;
 81         var evhostid = me.getAttr('data-fmml-event-host','');
 82         if(evhostid != '') {
 83             if(evhostid == 'parent' && myhostnode) {
 84                 me.eventHostNode = FM.findNodeWithAttr(myhostnode.parentNode, "fmmlHost");
 85             } else {
 86                 me.eventHostNode = FM.getNodeWithId(evhostid);
 87             }
 88             if (me.eventHostNode != null) {
 89                 me.eventHost = FM.isset(me.eventHostNode.fmmlHost) ? me.eventHostNode.fmmlHost : null;
 90             }
 91             if(me.eventHost == null) me.eventHostNode = null;            
 92         } else {
 93             me.eventHost = myhost;    
 94             me.eventHostNode = myhostnode;    
 95         }
 96     
 97         // data to send
 98         var evdmobj = null;
 99         var evdata = me.getAttr('data-fmml-event-data','');
100         if(evdata != '') {
101             if(FM.startsWith(evdata,'@')) {
102                 evdata = me.resolveAttrValue("-",evdata);
103             } 
104             evdmobj = FM.isset(evdata) && evdata && FM.isset(evdata.getAttr) ? evdata : new FM.DmGenericValue({
105                 value: evdata
106             });            
107         } else {
108             evdmobj = me.getDmObject();
109         }
110 
111         // event to send
112         var ev = me.getAttr('data-fmml-event-type','');
113         
114         // we are now ready to send
115         if(ev && ev != '') {
116             // set async class
117             if(me.getAttr('data-fmml-event-async','false') === 'true') {
118                 $(me.node).addClass("fmmlWaitButton");
119             }
120             
121             // clear old success & error hosts
122             me._runHostOnNode(document.getElementById(me.getAttr('data-fmml-run-on-success','')),null);
123             me._runHostOnNode(document.getElementById(me.getAttr('data-fmml-run-on-error','')),null);
124             
125             // send event
126             me.eventHost.onEvent(me,ev,{
127                 object: evdmobj,
128                 callback: function(isok, o) {
129                     me.eventCallback(isok, o);
130                 }
131             });
132         } else {
133             me.eventCallback(true, null);
134         }
135         return false;
136     });
137 }
138  
139 FM.MlObserverEvent.prototype._runHostOnNode = function(hostnode,dmobj) {
140     if(FM.isset(hostnode) && hostnode && FM.isset(hostnode.fmmlHost) && hostnode.fmmlHost) {
141         hostnode.fmmlHost.run(dmobj);
142     }
143 }
144 
145 FM.MlObserverEvent.prototype._redirectToPage = function(url,dmobj) {
146     var redirApl = FM.isset(url) && url && url != '' ?
147          this.resolveAttrValue('-',FM.isset(dmobj) && dmobj ? FM.applyTemplate(dmobj.getAttr(),url) : url) : 
148         ''
149     ;
150     window.location = redirApl;
151 }
152 
153 FM.MlObserverEvent.prototype._sendFormToPage = function(url,dmobj) {
154     if(!this.eventHost || !FM.isset(this.eventHostNode.submit)) return;
155     
156     var redirApl = FM.isset(url) && url && url != '' ?
157          this.resolveAttrValue('-',FM.isset(dmobj) && dmobj ? FM.applyTemplate(dmobj.getAttr(),url) : url) : 
158         ''
159     ;
160     
161     this.eventHostNode.action = redirApl;
162     this.eventHostNode.submit();    
163 }
164             
165 FM.MlObserverEvent.prototype._execCode = function(code,dmobj) {
166     try {
167         var val = FM.resolveAttrValue(this.options,"-", code,{
168             A: this.getApp(),
169             H: this.getHost(),
170             O: this,
171             D: dmobj
172         });
173     } catch(e) {
174         this.log(e,FM.logLevels.error,'MlObserverEvent._execCode');
175     }
176 }
177 
178 FM.MlObserverEvent.prototype.eventCallback = function(isok,oResponse) {
179     // remove async class 
180     $(this.node).removeClass("fmmlWaitButton");
181 
182     // exec
183     var code = isok ?
184         this.getAttr('data-fmml-exec-on-success','') :
185         this.getAttr('data-fmml-exec-on-error','')
186     ;
187     if(code != '') {
188         this._execCode(code,oResponse);
189     }
190     
191     // host run
192     var hostid = isok ?
193         this.getAttr('data-fmml-run-on-success','') :
194         this.getAttr('data-fmml-run-on-error','')
195     ;
196     if(hostid != '') {
197         this._runHostOnNode(document.getElementById(hostid),oResponse);
198     }        
199 
200     // redirect
201     var redir = isok ?
202         this.getAttr('data-fmml-redirect-on-success','') :
203         this.getAttr('data-fmml-redirect-on-error','')
204     ;
205     if(redir != '') {
206         if(this.getAttr('data-fmml-send-form','false') == 'true') {
207             this._sendFormToPage(redir,oResponse);
208         } else{
209             this._redirectToPage(redir,oResponse);
210         }
211     }
212 }

214 FM.MlObserver.addObserver("Event",FM.MlObserverEvent,'GLOBAL');
