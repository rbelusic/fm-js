1 /**
  2 * Attribute display ML observer class. 
  3 *
  4 * <table>
  5 * <th>List of ML node attributes</th>
  6 * <tr><td>data-fmml-attr-value-visible</td><td>type of attribute (string,number,date)</td></tr>
  7 * <tr><td>data-fmml-attr-value-hidden</td><td>set true if dates is in UTC</td></tr>
  8 * </table>
  9 * 
 10 * @class FM.MlObserverDisplay
 11 * @extends FM.MlObserver
 12 * @param {object} [attrs] DOM node attributes
 13 * @param {DOMnode} node DOM node
 14 */    
 15 FM.MlObserverDisplay = FM.defineClass('MlObserverDisplay',FM.MlObserver);
 16 
 17 FM.MlObserverDisplay.prototype._init = function(app,attrs,node) {            
 18     this._super("_init",app,attrs,node);
 19     this.objectSubClass ="Display";
 20 }

 23 FM.MlObserverDisplay.prototype.evalCondition = function(dmobj,v) {
 24     var ok = false;
 25     
 26     // if condition is undefined
 27     if(!FM.isset(v) || v == '' || !v) return ok;
 28     
 29     // prop or function in dmobject
 30     if(dmobj && FM.isset(dmobj[v])) { 
 31         if(FM.isFunction(dmobj[v])) {
 32             ok = dmobj[v]();
 33         } else {
 34             ok = dmobj[v];
 35         }
 36     } else if(dmobj && dmobj.isAttr(v)) { // attr in dmobject
 37         ok = dmobj.getAttr(v,false);
 38     } else {
 39         // eval value
 40         ok = this.resolveAttrValue("-",v);
 41     }
 42 
 43     // end
 44     return FM.isset(ok) && ok != null && ok != '' && ok != false && ok != 'false';    
 45 }

 69 FM.MlObserverDisplay.prototype.setNodeValue = function() {
 70     var visible = null;
 71     var visibilityCheck = this.getAttr('data-fmml-attr-value-visible','');    
 72     if(visibilityCheck != '') {
 73         visible = this.evalCondition(this.getDmObject(),visibilityCheck);
 74     } else {
 75         visibilityCheck = this.getAttr('data-fmml-attr-value-hidden','');
 76         if(visibilityCheck != '') {
 77             visible = !this.evalCondition(this.getDmObject(),visibilityCheck);
 78         }
 79     }
 80     
 81     // set
 82     if(this.getAttr('data-fmml-attr-name','') != '') {
 83         var value = this.getValue();
 84     
 85         if(FM.isset(this.node.value)) {
 86             this.node.value = value;
 87         } else if(this.node.nodeName == 'IMG') {
 88             this.node.setAttribute("src",value);
 89         } else {
 90             this.node.innerHTML = value;
 91         }
 92     }
 93     
 94     // show & hide
 95     if(visible) {
 96         $(this.node).show();
 97     } else {
 98         $(this.node).hide();
 99     }
100     
101 }
