1 /**
  2 * List selection ML observer class for {@link FM.MlHostGenericList} host. 
  3 * 
  4 * @class FM.MlObserverListSelection
  5 * @extends FM.MlObserver
  6 * @param {object} [attrs] DOM node attributes
  7 * @param {DOMnode} node DOM node
  8 */ 
  9 FM.MlObserverListSelection = FM.defineClass('MlObserverListSelection',FM.MlObserver);
 10     
 11 
 12 FM.extendClass(FM.MlObserverListSelection, FM.MlObserver);  
 13 
 14 FM.MlObserverListSelection.prototype._init = function(app,attrs,node) {            
 15     this._super("_init",app,attrs,node);
 16     this.objectSubClass ="ListSelection";
 17     
 18     var me = this;
 19     $(this.node).click(function(event) {
 20         var host = me.getHost();                
 21         var dmobj = me.getDmObject();
 22         if(dmobj && FM.isset(host.isSelected)) {
 23             host.onEvent(
 24                 me,
 25                 host.isSelected(dmobj) ? "onDeselected" : "onSelected", 
 26                 {object: dmobj}
 27             );
 28         }
 29         return true;
 30     });
 31 }
 32 
 33 FM.MlObserverListSelection.prototype.setNodeValue = function() {
 34     // provjeri
 35     var host = this.getHost();                
 36         
 37     // get dmobj && dmlist
 38     var dmobj = this.getDmObject();
 39     var issel = dmobj && FM.isset(host.isSelected) ?  host.isSelected(dmobj) : false;
 40     if(FM.isset(this.node.checked)) {
 41         if(this.node.checked != issel) this.node.checked = issel;
 42     } 
 43     
 44     if(issel) {
 45         $(this.node).closest(".fmmlClsListRowWrapper").addClass("fmmlSelected");
 46     } else {
 47         $(this.node).closest(".fmmlClsListRowWrapper").removeClass("fmmlSelected");
 48     }
 49 }
 50 
 51 FM.MlObserver.addObserver("ListSelection",FM.MlObserverListSelection,'GLOBAL');
