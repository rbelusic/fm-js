 86 FM.MlHostGenericCollection.prototype._nodeApplyTemplate = function(node,attrs) {

 93         }

 99 FM.MlHostGenericCollection.prototype._refreshItems = function() {
100     this._clearItems();
101     var dmList = this.getDmObject();
102     if(!dmList) return false;
103     var me = this;    
104     var curitm = -1;
105     var itmcnt = 0;
106     var maxitms = parseInt(this.getAttr('data-fmml-list-max-items','200'));
107     maxitms = FM.isset(maxitms) ? maxitms : 200;
108     dmList.forEachListElement(function(i,oObj) {                
109         var layout = me.getRegistryValue(
110             "/itemsLayout",
111             me.getAttr('data-fmml-list-items-layout','icon')
112             );
113         
114         var attrs = {
115             itemsLayout: layout
116         };
117         if (me._filterItemFromDisplay(oObj) == true) {
118             curitm++;
119             if(curitm < me.listOffset) return true;
120         
121             oObj.forEachAttr(function(pname,value) {
122                 attrs[pname] = value;
123                 return true;
124             });
125             attrs["objectClass"] = oObj.getSubClassName();
126             attrs["data_id"] = oObj.getDataID();
127             attrs["listIndex"] = curitm+1;
128             
129             itmcnt++;
130             if(me.listItemsTemplate) {
131                 me._createItmNode(oObj,attrs,me.listItemsTemplate);
132             } else {
133                 var tname = me.getAttr(
134                     'data-fmml-list-items-template-base',
135                     "ui.layouts.dm.[:objectClass].[:itemsLayout].html"
136                     );
137                 tname = FM.applyTemplate(attrs,tname,false,true);
138                 FM.UtTemplate.getTemplate(me.getApp(),attrs,tname,function(isok,templ) {            
139                     if(isok) {                
140                         me._createItmNode(oObj,attrs,templ);
141                     }
142                 });
143             }
144         }
145     
146         return maxitms > itmcnt;
147     });
148 
149     var emptyTempl = this.getAttr('data-fmml-list-items-template-empty','');
150     if(maxitms > itmcnt && (emptyTempl != '' || this.listEmptyTemplate)) {
151         curitm++;
152         if(this.listEmptyTemplate) {
153             while(maxitms > itmcnt) {
154                 curitm++;
155                 itmcnt++;
156                 var attrs = {
157                     itemsLayout: 'unknown',
158                     objectClass: 'unknown',
159                     data_id: 'unknown',
160                     listIndex: curitm
161                 };
162                 this._createItmNode(null,attrs,this.listEmptyTemplate);                                        
163             }            
164         } else {        
165             FM.UtTemplate.getTemplate(this.getApp(),{},emptyTempl,function(isok,templ) {                    
166                 if(isok) {
167                     while(maxitms > itmcnt) {
168                         curitm++;
169                         itmcnt++;
170                         var attrs = {
171                             itemsLayout: 'unknown',
172                             objectClass: 'unknown',
173                             data_id: 'unknown',
174                             listIndex: curitm
175                         };
176                         me._createItmNode(null,attrs,templ);                                        
177                     }
178                 }
179             });
180         }
181     }
182     
183     // empty list
184     if(itmcnt == 0 && emptyTempl == '') this.setListEmpty();
185     
186     // send event
187     this.sendEventToObservers(this,'onListRefreshCompleted');
188 }

190 FM.MlHostGenericCollection.prototype._createItmNode = function(oObj,attrs,templ) {
191     var curitm = parseInt(FM.getAttr(attrs,'listIndex','0'));
192     
193     var itm = $(templ);
194     if(itm) {
195         if(this.listItemsInner) {
196             var iNode = $(this.listItemsInner).clone();
197             if(FM.isset(iNode) && iNode && FM.isset(iNode[0])) {
198                 iNode = iNode[0];
199             }
200             this._nodeApplyTemplate(iNode,attrs);
201             $(iNode).append(itm);
202             itm = iNode;
203         }
204 
205         if(this.listItemsWrapper) {
206             var newNode = $(this.listItemsWrapper).clone();
207             if(FM.isset(newNode) && newNode && FM.isset(newNode[0])) {
208                 newNode = newNode[0];
209             }
210             this._nodeApplyTemplate(newNode,attrs);
211             $(newNode).append(itm);
212         } else {
213             newNode = itm;
214         }
215 
216 
217         $(newNode).attr('data-fmml-list-index',curitm);   
218         $(newNode).addClass(curitm & 1 ? 'fmmlRowOdd' : 'fmmlRowEven');
219         $(newNode).attr('data-fmml-item-data-id',oObj ? oObj.getDataID() : 'unknown');
220         
221         this._appendItmNode(oObj,newNode,attrs);                        
222     }
223 }

225 FM.MlHostGenericCollection.prototype._appendItmNode = function(oObj,node,attrs) {
226     var index = attrs["listIndex"];
227     var lastNode = null;
228     var lastNodeIndex = -1;
229     for(var i = index-1; i >=0 && lastNodeIndex == -1; i--) {
230         lastNode = $("[data-fmml-list-index='" + i + "']");
231         lastNode = lastNode.length ? lastNode[0]: null;
232         if(lastNode) {
233             lastNodeIndex = i;
234         }
235     }
236     if(lastNode) {
237         $(lastNode).after(node[0]);
238     } else {
239         $(this.listItemsContainer).prepend(node);
240     }
241     if(oObj) {
242         FM.MlHost.initChildNodes(this.getApp(),node,oObj,this.listItemsWrapper != null);
243     }
244 }

250 FM.MlHostGenericCollection.prototype._createList = function(oObj) {
251     var dmconfName = this.getAttr('data-fmml-list','');
252     var listOpt = {};
253     var pref = 'data-fmml-list-attr-';
254     var prefLen = pref.length;
255     
256     this.forEachAttr(function(pname,value) {
257         if(FM.startsWith(pname,pref)) {
258             listOpt[pname.substring(prefLen)] = value;
259         }                    
260         return true;
261     });
262     
263     // input param
264     oObj = FM.isset(oObj) && oObj ? oObj : null;
265     if(!oObj) {        
266         var lhost = this.getLinkedHost();
267         if(lhost) {
268             var node = document.getElementById(lhost);
269             if( node && FM.isset(node.fmmlHost) && node.fmmlHost) {
270                 oObj = node.fmmlHost.getDmObject();
271             }         
272         }
273     }
274     
275     if(FM.isset(oObj) && oObj && FM.isset(oObj.forEachAttr)) { // fm obect
276         oObj.forEachAttr(function(pname,value) {
277             listOpt[pname] = value;
278             return true;
279         });
280         listOpt["objectClass"] = oObj.getSubClassName();
281     } else if(FM.isArray(oObj) || FM.isObject(oObj)) { // array
282         FM.forEach(oObj, function(pname,value) {
283             listOpt[pname] = value;
284             return true;
285         });        
286     }
287     
288     dmconfName = FM.resolveAttrValue(null,"-",dmconfName,{
289         A: this.getApp(),
290         H: this
291     });
292     var dmList = FM.isObject(dmconfName) && dmconfName ? 
293         dmconfName :
294         new FM.DmList(listOpt,dmconfName,this.getApp())
295     ;
296     
297     return dmList;
298 }

334 FM.MlHostGenericCollection.prototype.addToSelection = function(o, node, sendevent) {    
335     if(this.maxSelected == 0) {
336         return;
337     } else if(this.maxSelected == 1) {
338         this.clearSelection(false);
339     } else if(this.maxSelected != -1) {
340         var cnt = 0;
341         FM.forEach(this.selectedItems,function(id,obj) {
342             if(obj == true) cnt++;
343             return true;
344         });
345         if(cnt >= this.maxSelected) return;
346     }
347     
348     if(FM.isArray(o)) {
349         for(var i= 0; i < o.length; i++) {
350             this.selectedItems[o[i].getDataID()]=true;
351         }
352     } else {
353         var oid = FM.isset(o) && o ? 
354         (FM.isString(o) ? o : o.getDataID()) :
355         ''
356         ;      
357         this.selectedItems[oid]=true;
358     }
359     if(FM.isset(node) && node) $(node).addClass("fmmlSelected");
360     if(FM.isset(sendevent) && sendevent) this.updateAllObservers();
361 }

363 FM.MlHostGenericCollection.prototype.removeFromSelection = function(o,node,sendevent) { 
364     if(FM.isArray(o)) {
365         for(var i= 0; i < o.length; i++) {
366             if(FM.isset(o[i]) && o[i] && FM.isset(this.selectedItems[o[i].getDataID()])) {
367                 this.selectedItems[o[i].getDataID()]=false;
368             }
369         }
370     } else {
371         var oid = FM.isset(o) && o ? 
372         (FM.isString(o) ? o : o.getDataID()) :
373         ''
374         ;      
375         if(FM.isset(this.selectedItems[oid])) this.selectedItems[oid]=false;
376     }
377     
378     if(FM.isset(node) && node) $(node).removeClass("fmmlSelected");
379     if(FM.isset(sendevent) && sendevent) this.updateAllObservers();
380 }

497 FM.MlHostGenericCollection.prototype.onChangeResource = function(oSender,evdata) {
498     var oObj = FM.getAttr(evdata,'object',null);
499     if(!oObj) return;
500     
501     var resurl = oObj.getAttr('resource_url',oObj.getAttr('value',''));
502     if(resurl != '') {
503         var resResolvFn = this.getAttr('data-fmml-list-attr-resource-parser','');
504         if(resResolvFn !== '') {
505             resResolvFn = FM.stringPtrToObject(resResolvFn, this, this.getApp());
506             if(resResolvFn) {
507                 try {
508                     resurl = resResolvFn(this,resurl);
509                 } catch(e) {
510                     this.log(e,FM.logLevels.error,'onChangeResource');
511                 }
512             }
513         }
514         this.listOffset = 0;
515         var dmList = this.getDmObject();
516         if(dmList) {
517             dmList.setAttr('resource_url',resurl,true);
518         }
519     }

653 FM.MlHostGenericCollection.prototype._getNodeListNode = function(node) {
654     var inode = node;
655     var rnode = null;
656     while(inode && !FM.isset($(inode).attr('data-fmml-item-data-id'))) {
657         inode = inode.parentNode;

677 FM.MlHostGenericCollection.prototype.onNextPage = function() {
678     var dmList = this.getDmObject();
679     if(!dmList) return (true);
680     var maxitms = parseInt(this.getAttr('data-fmml-list-max-items','200'));
681     var itemscnt = this.getFilteredCount();
682     maxitms = FM.isset(maxitms) && maxitms > 0 ? maxitms : 200;
683     var getMoreData = false;
684     var calcListOffset = this.listOffset + maxitms;
685     this.listOffset = calcListOffset;
686     if(this.listOffset > itemscnt) {
687         this.listOffset = Math.floor(itemscnt/maxitms) * maxitms;
688         getMoreData = true;
689     }
690     
691     if(this.listOffset >= itemscnt) {
692         this.listOffset = this.listOffset -= maxitms;
693         getMoreData = true;        
694     }
695     this._refreshItems();
696     
697     if(getMoreData && this.getAttr('data-fmml-list-get-more-at-end','false') == 'true') {
698         var fromrowAttr = this.getAttr('data-fmml-fromrow-attr-name',"fromrow");
699         if(fromrowAttr != '') {
700             dmList.setAttr(fromrowAttr,dmList.getListSize(),false);
701         }
702         this.getMore();
703     }
704     
705     // kraj
706     return true;
707 }
