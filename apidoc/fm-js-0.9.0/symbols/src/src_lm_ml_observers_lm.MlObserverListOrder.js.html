1 /**
  2 * List order ML observer class for {@link FM.MlHostGenericList} host. 
  3 * 
  4 * @class FM.MlObserverListOrder
  5 * @extends FM.MlObserver
  6 * @param {object} [attrs] DOM node attributes
  7 * @param {DOMnode} node DOM node
  8 */ 
  9 FM.MlObserverListOrder = FM.defineClass('MlObserverListOrder',FM.MlObserver);
 10 
 11 FM.MlObserverListOrder.prototype._init = function(app,attrs,node) {            
 12     this._super("_init",app,attrs,node);
 13     this.objectSubClass ="ListOrder";
 14     
 15     var me = this;
 16     $(this.node).click(function(event) { 
 17         event.preventDefault();
 18         var dmobj = me.getDmObject();
 19         if(dmobj) {
 20             var attrname = me.getAttr('data-fmml-attr-name',''); // mine attr
 21             var orderAttrname = me.getAttr('data-fmml-order-attr-name','order_attr'); // attr to set order (put attrname here)
 22             var orderDir = me.getAttr('data-fmml-order-dir-attr-name','order'); // order dir attr (ASC/DESC)
 23             var curOrderAtrr = dmobj.getAttr(orderAttrname,'');
 24             var curOrderDir = dmobj.getAttr(orderDir,'ASC');
 25             if(curOrderAtrr == attrname) {
 26                 dmobj.setAttr(orderDir, curOrderDir == 'DESC' ? 'ASC' : 'DESC',true);
 27             } else {
 28                 var attrs = {};
 29                 attrs[orderAttrname] = attrname;
 30                 attrs[orderDir] = 'ASC';
 31                 dmobj.setAttr(null,attrs,true);                
 32             }
 33         }
 34     });
 35 }        
 36 
 37 FM.MlObserverListOrder.prototype.setNodeValue = function() {    
 38     // provjeri
 39     var attrname = this.getAttr('data-fmml-attr-name',''); // mine attr
 40     var orderAttrname = this.getAttr('data-fmml-order-attr-name','order_attr'); // attr to set order (put attrname here)
 41     var orderDir = this.getAttr('data-fmml-order-dir-attr-name','order'); // order dir attr (ASC/DESC)
 42     var dmobj = this.getDmObject();
 43     if(dmobj && attrname != '') {
 44         var curOrderAttr = dmobj.getAttr(orderAttrname,'');
 45         var curOrderDir = dmobj.getAttr(orderDir,'ASC');
 46         if(curOrderAttr == attrname) {
 47             if(curOrderDir == 'DESC') {
 48                 $(this.node).removeClass("fmmlOrderAsc");
 49                 $(this.node).addClass("fmmlOrderDesc");
 50             } else {
 51                 $(this.node).addClass("fmmlOrderAsc");
 52                 $(this.node).removeClass("fmmlOrderDesc");                
 53             }
 54         } else {
 55             $(this.node).removeClass("fmmlOrderAsc");
 56             $(this.node).removeClass("fmmlOrderDesc");            
 57         }
 58     }
 59 }
 60 
 61 FM.MlObserver.addObserver("ListOrder",FM.MlObserverListOrder,'GLOBAL');
