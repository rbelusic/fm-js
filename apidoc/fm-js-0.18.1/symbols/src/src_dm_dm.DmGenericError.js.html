1 /**

 13 FM.DmGenericError = FM.defineClass('DmGenericError',FM.DmObject);
 14 
 15 // methods
 16 FM.DmGenericError.prototype._init = function(attrs) {
 17     this._super("_init", attrs, {messageId: "0",text: "No error"});
 18     this.objectSubClass = "GenericError";
 19 }
 20         
 21 FM.DmGenericError.prototype.getDataID = function() {
 22     return this.getID();
 23 }
 24 
 25 FM.DmGenericError.prototype.getErrorCode = function() {
 26     return this.getAttr('messageId','0');
 27 }
 28 
 29 FM.DmGenericError.prototype.setErrorCode = function(ec) {
 30     this.setAttr('messageId',ec);
 31 }
 32 
 33 FM.DmGenericError.prototype.getErrorText = function() {
 34     return this.getAttr('text','');
 35 }
 36 
 37 FM.DmGenericError.prototype.setErrorText = function(text) {
 38     return this.setAttr('text',text);
 39 }
 40 
 41 FM.DmGenericError.prototype.isError = function() {
 42     var errCode = this.getErrorCode();
 43     
 44     return errCode !== '' && errCode !== '0';
 45 }
