 35         var lX4, lY4, lX8, lY8, lResult;
 36         lX8 = (lX & 0x80000000);
 37         lY8 = (lY & 0x80000000);
 38         lX4 = (lX & 0x40000000);
 39         lY4 = (lY & 0x40000000);
 40         lResult = (lX & 0x3FFFFFFF) + (lY & 0x3FFFFFFF);
 41         if (lX4 & lY4) {
 42             return (lResult ^ 0x80000000 ^ lX8 ^ lY8);
 43         }
 44         if (lX4 | lY4) {

 46                 return (lResult ^ 0xC0000000 ^ lX8 ^ lY8);
 47             } else {

 88     var convertToWordArray = function(str) {

 93         var lNumberOfWords = (lNumberOfWords_temp2 + 1) * 16;
 94         var lWordArray = new Array(lNumberOfWords - 1);
 95         var lBytePosition = 0;
 96         var lByteCount = 0;
 97         while (lByteCount < lMessageLength) {
 98             lWordCount = (lByteCount - (lByteCount % 4)) / 4;
 99             lBytePosition = (lByteCount % 4) * 8;
100             lWordArray[lWordCount] = (lWordArray[lWordCount] | (str.charCodeAt(lByteCount) << lBytePosition));

102         }
103         lWordCount = (lByteCount - (lByteCount % 4)) / 4;
104         lBytePosition = (lByteCount % 4) * 8;
105         lWordArray[lWordCount] = lWordArray[lWordCount] | (0x80 << lBytePosition);
106         lWordArray[lNumberOfWords - 2] = lMessageLength << 3;
107         lWordArray[lNumberOfWords - 1] = lMessageLength >>> 29;

217 /**
218  * Decodes string encoded with MIME base64.
219  * Derived from <a href="http://phpjs.org/functions/base64_decode/">php.js</a> base64 decode function.

225 FM.base64_decode=function (data) {
226     // http://kevin.vanzonneveld.net
227     // +   original by: Tyler Akins (http://rumkin.com)
228     // +   improved by: Thunder.m
229     // +      input by: Aman Gupta
230     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
231     // +   bugfixed by: Onno Marsman
232     // +   bugfixed by: Pellentesque Malesuada
233     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
234     // +      input by: Brett Zamir (http://brett-zamir.me)
235     // +   bugfixed by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
236     // *     example 1: base64_decode('S2V2aW4gdmFuIFpvbm5ldmVsZA==');
237     // *     returns 1: 'Kevin van Zonneveld'
238     // mozilla has this native
239     // - but breaks in 2.0.0.12!
240     //if (typeof this.window['atob'] === 'function') {
241     //    return atob(data);
242     //}
243     var b64 = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";
244     var o1, o2, o3, h1, h2, h3, h4, bits, i = 0,
245     ac = 0,
246     dec = "",
247     tmp_arr = [];
248 
249     if (!data) {
250         return data;
251     }
252 
253     data += '';
254 
255     do { // unpack four hexets into three octets using index points in b64
256         h1 = b64.indexOf(data.charAt(i++));
257         h2 = b64.indexOf(data.charAt(i++));
258         h3 = b64.indexOf(data.charAt(i++));
259         h4 = b64.indexOf(data.charAt(i++));
260 
261         bits = h1 << 18 | h2 << 12 | h3 << 6 | h4;
262 
263         o1 = bits >> 16 & 0xff;
264         o2 = bits >> 8 & 0xff;
265         o3 = bits & 0xff;
266 
267         if (h3 == 64) {
268             tmp_arr[ac++] = String.fromCharCode(o1);
269         } else if (h4 == 64) {
270             tmp_arr[ac++] = String.fromCharCode(o1, o2);
271         } else {
272             tmp_arr[ac++] = String.fromCharCode(o1, o2, o3);
273         }
274     } while (i < data.length);
275 
276     dec = tmp_arr.join('');
277 
278     return dec;
279 }

298 FM.base64_encode = function(data) {
299     // http://kevin.vanzonneveld.net
300     // +   original by: Tyler Akins (http://rumkin.com)
301     // +   improved by: Bayron Guevara
302     // +   improved by: Thunder.m
303     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
304     // +   bugfixed by: Pellentesque Malesuada
305     // +   improved by: Kevin van Zonneveld (http://kevin.vanzonneveld.net)
306     // +   improved by: Rafał Kukawski (http://kukawski.pl)
307     // *     example 1: base64_encode('Kevin van Zonneveld');
308     // *     returns 1: 'S2V2aW4gdmFuIFpvbm5ldmVsZA=='
309     // mozilla has this native
310     // - but breaks in 2.0.0.12!
311     //if (typeof this.window['btoa'] === 'function') {
312     //    return btoa(data);
313     //}
314     var b64 = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=";
315     var o1, o2, o3, h1, h2, h3, h4, bits, i = 0,
316     ac = 0,
317     enc = "",
318     tmp_arr = [];
319 
320     if (!data) {
321         return data;
322     }
323 
324     do { // pack three octets into four hexets
325         o1 = data.charCodeAt(i++);
326         o2 = data.charCodeAt(i++);
327         o3 = data.charCodeAt(i++);
328 
329         bits = o1 << 16 | o2 << 8 | o3;

331         h1 = bits >> 18 & 0x3f;
332         h2 = bits >> 12 & 0x3f;
333         h3 = bits >> 6 & 0x3f;
334         h4 = bits & 0x3f;

339

341 
342     var r = data.length % 3;
343 
344     return (r ? enc.slice(0, r - 3) : enc) + '==='.slice(r || 3);
