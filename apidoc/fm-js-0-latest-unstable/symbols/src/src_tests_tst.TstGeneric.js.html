1 /**
  2 * Generic test class. 
  3 * 
  4 * @class FM.TstGeneric
  5 * @extends FM.Object
  6 * @param {object} config Test configuration
  7 */    
  8 FM.TstGeneric = function() {   
  9     this._init.apply(this, arguments); // new poziva _init()
 10 }
 11 
 12 FM.extendClass(FM.TstGeneric,FM.Object); 
 13 
 14 // properties
 15 FM.TstGeneric.prototype.objectSubClass = "";
 16 FM.TstGeneric.prototype.testsList = null;
 17 FM.TstGeneric.prototype.testsResults = null;
 18 
 19 // methods
 20 FM.TstGeneric.prototype._init = function(config) {            
 21     this._super("_init",config);
 22     this.objectSubClass = "TstGeneric";
 23     this.testsList = {};
 24     this.testsResults = {};
 25     this.runningTests = {};
 26 }
 27 
 28 FM.TstGeneric.prototype.run = function(test) {
 29     this.runningTests = {};
 30     this.testsResults = {};
 31     var tlist = {};    
 32     test = FM.isset(test) ? test : null;
 33     if(test) {
 34         this.log("Running tests [" + test + "] ...",FM.logLevels.info,this.getClassName());
 35         if(!FM.isset(this.testsList[test])) {
 36             this.log("Test [" + test + "] is not found!",FM.logLevels.error,this.getClassName());
 37             return false;
 38         }
 39         tlist[test] = true;        
 40     } else {
 41         this.log("Running all tests ...",FM.logLevels.info,this.getClassName());
 42         tlist = this.testsList;
 43     }
 44 
 45     for(var tname in tlist) {
 46         this.runningTests[tname] = false;
 47     }
 48 
 49     for(var tname in tlist) {
 50         this.log("Running test [" + tname + "] ...",FM.logLevels.info,this.getClassName());
 51         this[tname]();
 52     }
 53     
 54     return true;
 55 }     
 56 
 57 FM.TstGeneric.prototype.addResult = function(test,result) {
 58     this.testsResults[test] = result;
 59     this.runningTests[test] = true;
 60     
 61     for(var tname in this.runningTests) {
 62         if(!this.runningTests[tname]) return true;
 63     }
 64     return this.showResults();
 65 }
 66 
 67 FM.TstGeneric.prototype.showResults = function() {
 68     var cntok =0, cnterr = 0;
 69     this.print(" == Test results =================================================");
 70     for(var tname in this.testsResults) {
 71         if(this.testsResults[tname] == true) {
 72             this.print("Test [" + tname + "] is successfully finished.");
 73             cntok++;
 74         } else if(this.testsResults[tname] == false) {
 75             this.print("Test [" + tname + "] is in error!.");
 76             cnterr++;
 77         } else { // bilo sto osim true i false
 78             this.print("Test [" + tname + "] is invalid!.");
 79             cnterr++;
 80         }
 81     }
 82     this.print(" =================================================================");
 83     return true;
 84 }

 87 FM.TstGeneric.prototype.addTest = function(test) {
 88     if(FM.isset(this[test]) && FM.isFunction(this[test])) {
 89         this.testsList[test] = true;
 90         return true;
 91     }
 92     
 93     return false;
 94 }
 95 
 96 FM.TstGeneric.prototype.removeTest = function(test) {
 97     if(FM.isset(this.testsList[test])) {
 98         delete this.testsList[test];
 99         return true;
100     }
101     
102     return false;
103 }
104 
105 FM.TstGeneric.prototype.enableTest = function(test) {
106     if(FM.isset(this.testsList[test])) {
107         this.testsList[test] = true;
108         return true;
109     }
110     
111     return false;
112 }
113 
114 FM.TstGeneric.prototype.disableTest = function(test) {
115     if(FM.isset(this.testsList[test])) {
116         this.testsList[test] = false;
117         return true;
118     }
119     
120     return false;
121 }
122 
123 FM.TstGeneric.prototype.print = function(msg) {
124     if(FM.isObject(msg) || FM.isArray(msg)) {
125         console.dir(msg);
126     } else {
127         console.log(msg);
128     }
129 }
